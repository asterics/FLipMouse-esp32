83 void switch_radio(void)
84 {
85  //check if wifi was already used once.
86  if((xEventGroupGetBitsFromISR(connectionRoutingStatus) & WIFI_LOCKED) != 0)
87  {
88  ESP_LOGW(LOG_TAG,"Wifi was already used.");
89  return;
90  }
91 
92  //check if we enable/disable wifi.
93  if(isWifiOn == 0)
94  {
95  isWifiOn = 1;
96  LED(255,0,255,0);
97  taskWebGUIEnDisable(1,true);
98  } else {
99  uint8_t slotnr = halStorageGetCurrentSlotNumber();
100  slotnr++;
101  LED((slotnr%2)*0xFF,((slotnr/2)%2)*0xFF,((slotnr/4)%2)*0xFF,0);
102  isWifiOn = 0;
103  taskWebGUIEnDisable(0,true);
104  }
105 }

125 void app_main()

139  hid_ble = xQueueCreate(32,sizeof(hid_cmd_t));
140  hid_usb = xQueueCreate(32,sizeof(hid_cmd_t));
141  debouncer_in = xQueueCreate(32,sizeof(raw_action_t));
142 
143  //exit critical section & resume all tasks for initialising
144  xTaskResumeAll();
145 
146  //start IO continous task
147  if(halIOInit() == ESP_OK)
148  {
149  ESP_LOGD(LOG_TAG,"initialized halIOInit");
150  //set LED to first slot directly on startup

152  } else {
153  ESP_LOGE(LOG_TAG,"error initializing halIOInit");
154  }
155 
156  //initialize serial communication task
157  //(USB-HID & USB Serial for commands)
158  if(halSerialInit() == ESP_OK)
159  {
160  ESP_LOGD(LOG_TAG,"initialized halSerial");

162  ESP_LOGE(LOG_TAG,"error initializing halSerial");

164 
165  //start adc continous task
166  if(halAdcInit(NULL) == ESP_OK)

168  ESP_LOGD(LOG_TAG,"initialized halAdcInit");
169  } else {
170  ESP_LOGE(LOG_TAG,"error initializing halAdcInit");
171  }
172  //we need to calibrate here, otherwise sip/puff is not 512 in idle...
173  halAdcCalibrate();
174 
175  esp_event_loop_create_default();
176 
177  //init HID handler
178  if(handler_hid_init() == ESP_OK)
179  {

184

189  } else {
190  ESP_LOGE(LOG_TAG,"error adding VB handler");
191  }
192 
193  //command parser
194  if(taskCommandsInit() == ESP_OK)
195  {
196  ESP_LOGD(LOG_TAG,"initialized taskCommands");
197  } else {
198  ESP_LOGE(LOG_TAG,"error initializing taskCommands");
199  }
200 
201  //start debouncer

203  (void*)NULL,DEBOUNCER_TASK_PRIORITY, NULL) == pdPASS)

205  ESP_LOGD(LOG_TAG,"created new debouncer task");
206  } else {
207  ESP_LOGE(LOG_TAG,"error creating new debouncer task");
208  }
209 
210  //add the handler for long button presses (activates WiFi)
211  halIOAddLongPressHandler(switch_radio);
212 
213  //calibrate directly after start-up
214  //vTaskDelay(50/portTICK_PERIOD_MS);
215 
216 
217  //start config switcher
218  if(configSwitcherInit() == ESP_OK)
219  {
220  ESP_LOGD(LOG_TAG,"initialized configSwitcherInit");
221  } else {
222  ESP_LOGE(LOG_TAG,"error initializing configSwitcherInit");
223  }
224 
225  //delete this task after initializing.
226  ESP_LOGI(LOG_TAG,"Finished initializing!");
227 
228  //we wait 0.75s to activate WiFi+BT, seems to be hungry :-)
229  vTaskDelay(750/portTICK_PERIOD_MS);
230  //start BLE (mouse/keyboard interfaces active)
231  if(halBLEInit(1,1,0) == ESP_OK)
232  {
233  ESP_LOGD(LOG_TAG,"initialized halBle");
234  } else {
235  ESP_LOGE(LOG_TAG,"error initializing halBle");
236  }
237 
238  if(taskWebGUIInit() == ESP_OK)
239  {
240  ESP_LOGD(LOG_TAG,"initialized webserver/DNS server/webgui");
241  } else {
242  ESP_LOGE(LOG_TAG,"error initializing webserver/DNS server/webgui");
243  }
244 
245  //TESTING
246  #if 0
247  #warning "Heap tracing is enabled, this is normally NOT the case!!!"
248  ESP_ERROR_CHECK( heap_trace_init_standalone(trace_record, NUM_RECORDS) );
249  vTaskDelay(100); //1s
250  ESP_ERROR_CHECK( heap_trace_start(HEAP_TRACE_ALL) );
251  raw_action_t action;
252  action.vb = 1;
253  action.type = VB_RELEASE_EVENT;
254  for(uint8_t i = 0; i<10; i++)

261  heap_trace_dump();
262  #endif
263 
264 
265  vTaskDelete(NULL);
266 }
