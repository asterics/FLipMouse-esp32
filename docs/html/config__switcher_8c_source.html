116 esp_err_t configUpdate(TickType_t time)
117 {
118  //reload ADC
119  if(halAdcUpdateConfig(&currentConfigLoaded.adc) != ESP_OK)
120  {
121  ESP_LOGE(LOG_TAG,"error reloading adc config");
122  return ESP_FAIL;
123  }
124 
125  //set other config infos
126  ESP_LOGD(LOG_TAG,"setting connection bits (USB: %d, BLE: %d)",currentConfigLoaded.usb_active,currentConfigLoaded.ble_active);
127  if(currentConfigLoaded.ble_active != 0) xEventGroupSetBits(connectionRoutingStatus,DATATO_BLE);
128  else xEventGroupClearBits(connectionRoutingStatus,DATATO_BLE);
129  if(currentConfigLoaded.usb_active != 0) xEventGroupSetBits(connectionRoutingStatus,DATATO_USB);
130  else xEventGroupClearBits(connectionRoutingStatus,DATATO_USB);
131 
132  //reset HID channels (USB&BLE)
133  halBLEReset(0);
134  halSerialReset(0);
135  return ESP_OK;
136 }

150 void configSwitcherTask(void * params)
151 {
152  char command[SLOTNAME_LENGTH];
153 
154  uint32_t tid = 0;
155  uint8_t justupdate = 0;
156  esp_err_t ret;
157 
158  if(config_switcher == 0)
159  {
160  ESP_LOGE(LOG_TAG,"config_switcher queue uninitialized, exiting.");
161  vTaskDelete(NULL);
162  }
163 
164  while(1)
165  {
166  //wait for a command.
167  if(xQueueReceive(config_switcher,command,1000/portTICK_PERIOD_MS) == pdTRUE)
168  {
169  //still commands to be processed, wait for queue to get empty...
170  if((xEventGroupWaitBits(systemStatus,SYSTEM_EMPTY_CMD_QUEUE,pdFALSE, \
171  pdFALSE,1000/portTICK_PERIOD_MS) & SYSTEM_EMPTY_CMD_QUEUE) == 0)
172  {
173  ESP_LOGW(LOG_TAG,"Timeout waiting for empty CMD queue flag");
174  continue;
175  }
176  //signal system that we are updating config now.
177  xEventGroupSetBits(systemStatus, SYSTEM_LOADCONFIG);
178  xEventGroupClearBits(systemStatus, SYSTEM_STABLECONFIG);
179 
180  //request storage access
181  while(halStorageStartTransaction(&tid,100,LOG_TAG) != ESP_OK)
182  {
183  ESP_LOGE(LOG_TAG,"Cannot start storage transaction");
184  vTaskDelay(100/portTICK_PERIOD_MS);
185  }
186  //just to be sure: normally we are not updating...
187  justupdate = 0;
188 
189  //command received, load new slot:
190  //__NEXT, __PREV, __DEFAULT, __UPDATE, __RESTOREFACTORY
191  if(strcmp(command,"__NEXT") == 0)
192  {
193  ESP_LOGD(LOG_TAG,"Load next slot");
194  ret = halStorageLoad(NEXT,tid);
195  } else if(strcmp(command,"__PREV") == 0) {
196  ESP_LOGD(LOG_TAG,"Load prev slot");
197  ret = halStorageLoad(PREV,tid);
198  } else if(strcmp(command,"__DEFAULT") == 0) {
199  //load default slot (if not available, it will be created)
200  ret = halStorageLoad(DEFAULT,tid);
201  ESP_LOGD(LOG_TAG,"loading default");
202  } else if(strcmp(command,"__RESTOREFACTORY") == 0) {
203  ret = halStorageDeleteSlot(0,tid);
204  ret = halStorageLoad(DEFAULT,tid);
205  if(ret != ESP_OK)
206  {
207  ESP_LOGE(LOG_TAG,"Error deleting all slots");
208  } else {
209  ESP_LOGW(LOG_TAG,"Deleted all slots");
210  }
211  halStorageFinishTransaction(tid);
212  continue;
213  } else {
214  ret = halStorageLoadName(command,tid);
215  ESP_LOGD(LOG_TAG,"Load by name: %s",command);
216  }
217 
218  if(ret != ESP_OK)
219  {
220  ESP_LOGE(LOG_TAG,"Error loading general slot config!");
221  }
222 
223  //make one or more config tones (depending on slot number)
224  uint8_t slotnr = halStorageGetCurrentSlotNumber() + 1;
225  for(uint8_t i = 0; i<slotnr; i++)
226  {
227  //create a tone and a pause (values are from original firmware)
228  TONE(TONE_CHANGESLOT_FREQ_BASE + slotnr*TONE_CHANGESLOT_FREQ_SLOTNR, \
229  TONE_CHANGESLOT_DURATION);
230  TONE(0,TONE_CHANGESLOT_DURATION_PAUSE);
231  }
232 
233  //LED output on slot switch (steady color on Neopixel, short fading on RGB)
234  LED((slotnr%2)*0xFF,((slotnr/2)%2)*0xFF,((slotnr/4)%2)*0xFF,0);
235 
236  ESP_LOGD(LOG_TAG,"LED");
237 
238  //clean up
239  halStorageFinishTransaction(tid);
240  tid = 0;
241 
242  ESP_LOGD(LOG_TAG,"storage");
243 
244  //now we wait for finished processing of AT commands.
245  if((xEventGroupWaitBits(systemStatus,SYSTEM_EMPTY_CMD_QUEUE, \
246  pdFALSE,pdFALSE,10) & SYSTEM_EMPTY_CMD_QUEUE) == 0)
247  {
248  ESP_LOGW(LOG_TAG,"command queue not emptied in time!");
249  }
250 
251  ESP_LOGD(LOG_TAG,"wait for cmds");
252 
253  //calibrate
254  halAdcCalibrate();
255 
256  ESP_LOGD(LOG_TAG,"calibrate");
257 
258  //signal system that we are finished with updating
259  xEventGroupClearBits(systemStatus, SYSTEM_LOADCONFIG);
260  xEventGroupSetBits(systemStatus, SYSTEM_STABLECONFIG);
261 
262  ESP_LOGD(LOG_TAG,"bits set");
263 
264  //reload general config
265  configUpdate(0);
266 
267  ESP_LOGD(LOG_TAG,"cfg update");
268 
269  if(justupdate)
270  {
271  xSemaphoreGive(configUpdatePending);
272  ESP_LOGI(LOG_TAG,"----Config Update Complete, loaded slot %s----",currentConfigLoaded.slotName);
273  } else {
274  ESP_LOGI(LOG_TAG,"----Config Switch Complete, loaded slot %s----",currentConfigLoaded.slotName);
275  }
276  }
277  }
278 }

285 esp_err_t configSwitcherInit(void)
286 {
287  //set log level to given log level
288  esp_log_level_set(LOG_TAG,LOG_LEVEL_CFGSW);
289  //test if the config_switcher queue is initialized...
290  if(config_switcher == NULL)
291  {
292  ESP_LOGE(LOG_TAG,"Error init config switcher, please create \
293  config_switcher queue before calling configSwitcherInit()!");
294  return ESP_FAIL;
295  }
296 
297  //init update semaphore
298  configUpdatePending = xSemaphoreCreateBinary();
299  xSemaphoreGive(configUpdatePending);
300 
301  //start configSwitcherTask
302  if(xTaskCreate(configSwitcherTask,"configswitcher",CONFIGSWITCHERTASK_PERMANENT_STACKSIZE,(void *)NULL,
303  HAL_CONFIG_TASK_PRIORITY,configswitcher_handle) != pdPASS)
304  {
305  ESP_LOGE(LOG_TAG,"error creating config switcher task, cannot proceed.");
306  return ESP_FAIL;
307  } else {
308  ESP_LOGD(LOG_TAG,"configSwitcherTask created");
309  }
310 
311  //load the default slot by sending "__DEFAULT" to
312  // the config_switcher queue
313  char commandname[SLOTNAME_LENGTH];
314  strcpy(commandname,"__DEFAULT");
315  if(xQueueSend(config_switcher,commandname,10) == pdPASS)
316  {
317  ESP_LOGD(LOG_TAG,"requesting default slot switch on startup");
318  } else {
319  ESP_LOGE(LOG_TAG,"Error requesting default slot switch");
320  }
321 
322  return ESP_OK;
323 }
