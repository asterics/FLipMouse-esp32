505 void esp_hidd_prf_cb_hdl(esp_gatts_cb_event_t event, esp_gatt_if_t gatts_if,
506  esp_ble_gatts_cb_param_t *param)
507 {
508  switch(event) {
509  case ESP_GATTS_REG_EVT: {
510  esp_hidd_cb_param_t hidd_param;
511  hidd_param.init_finish.state = param->reg.status;
512  if(param->reg.app_id == HIDD_APP_ID) {
513  hidd_le_env.gatt_if = gatts_if;
514  if(hidd_le_env.hidd_cb != NULL) {
515  (hidd_le_env.hidd_cb)(ESP_HIDD_EVENT_REG_FINISH, &hidd_param);
516  hidd_le_create_service(hidd_le_env.gatt_if);
517  }
518  }
519  if(param->reg.app_id == BATTRAY_APP_ID) {
520  hidd_param.init_finish.gatts_if = gatts_if;
521  if(hidd_le_env.hidd_cb != NULL) {
522  (hidd_le_env.hidd_cb)(ESP_BAT_EVENT_REG, &hidd_param);
523  }
524 
525  }
526 
527  break;
528  }
529  case ESP_GATTS_CONF_EVT: {
530  break;
531  }
532  case ESP_GATTS_CREATE_EVT:
533  break;
534  case ESP_GATTS_CONNECT_EVT:
535  //if(param->connect.is_connected) {
536  if(param->connect.conn_id != 0) {
537  esp_hidd_cb_param_t cb_param = {0};
538  LOG_ERROR("the connection establish, conn_id = %x",param->connect.conn_id);
539  memcpy(cb_param.connect.remote_bda, param->connect.remote_bda, sizeof(esp_bd_addr_t));
540  cb_param.connect.conn_id = param->connect.conn_id;
541  hidd_clcb_alloc(param->connect.conn_id, param->connect.remote_bda);
542  //esp_ble_set_encryption(param->connect.remote_bda, ESP_BLE_SEC_ENCRYPT_NO_MITM);
543  if(hidd_le_env.hidd_cb != NULL) {
544  (hidd_le_env.hidd_cb)(ESP_HIDD_EVENT_BLE_CONNECT, &cb_param);
545  }
546  }
547  break;
548  case ESP_GATTS_DISCONNECT_EVT: {
549  if(hidd_le_env.hidd_cb != NULL) {
550  (hidd_le_env.hidd_cb)(ESP_HIDD_EVENT_BLE_DISCONNECT, NULL);
551  }
552  hidd_clcb_dealloc(param->disconnect.conn_id);
553  break;
554  }
555  case ESP_GATTS_CLOSE_EVT:
556  break;
557  case ESP_GATTS_WRITE_EVT: {
558  break;
559  }
560  case ESP_GATTS_CREAT_ATTR_TAB_EVT: {
561  if (param->add_attr_tab.num_handle == BAS_IDX_NB &&
562  param->add_attr_tab.svc_uuid.uuid.uuid16 == ESP_GATT_UUID_BATTERY_SERVICE_SVC &&
563  param->add_attr_tab.status == ESP_GATT_OK) {
564  incl_svc.start_hdl = param->add_attr_tab.handles[BAS_IDX_SVC];
565  incl_svc.end_hdl = incl_svc.start_hdl + BAS_IDX_NB -1;
566  LOG_ERROR("%s(), start added the hid service to the stack database. incl_handle = %d",
567  __func__, incl_svc.start_hdl);
568  esp_ble_gatts_create_attr_tab(hidd_le_gatt_db, gatts_if, HIDD_LE_IDX_NB, 0);
569  }
570  if (param->add_attr_tab.num_handle == HIDD_LE_IDX_NB &&
571  param->add_attr_tab.status == ESP_GATT_OK) {
572  memcpy(hidd_le_env.hidd_inst.att_tbl, param->add_attr_tab.handles,
573  HIDD_LE_IDX_NB*sizeof(uint16_t));
574  LOG_ERROR("hid svc handle = %x",hidd_le_env.hidd_inst.att_tbl[HIDD_LE_IDX_SVC]);
575  hid_add_id_tbl();
576  esp_ble_gatts_start_service(hidd_le_env.hidd_inst.att_tbl[HIDD_LE_IDX_SVC]);
577  } else {
578  esp_ble_gatts_start_service(param->add_attr_tab.handles[0]);
579  }
580  break;
581  }
582 
583  default:
584  break;
585  }
586 }
