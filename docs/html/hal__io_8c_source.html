127 static void gpio_isr_handler(void* arg)
128 {
129  uint32_t pin = (uint32_t) arg;
130  uint8_t vb = 0;
131  BaseType_t xResult, xHigherPriorityTaskWoken = pdFALSE;
132 
133  //determine pin of ISR reason
134  switch(pin)
135  {
136  case HAL_IO_PIN_BUTTON_EXT1: vb = VB_EXTERNAL1; break;
137  case HAL_IO_PIN_BUTTON_EXT2: vb = VB_EXTERNAL2; break;
138  #ifdef DEVICE_FABI
139  case HAL_IO_PIN_BUTTON_EXT3: vb = VB_EXTERNAL3; break;
140  case HAL_IO_PIN_BUTTON_EXT4: vb = VB_EXTERNAL4; break;
141  case HAL_IO_PIN_BUTTON_EXT5: vb = VB_EXTERNAL5; break;
142  case HAL_IO_PIN_BUTTON_EXT6: vb = VB_EXTERNAL6; break;
143  case HAL_IO_PIN_BUTTON_EXT7: vb = VB_EXTERNAL7; break;
144  #endif
145  case HAL_IO_PIN_BUTTON_INT1: vb = VB_INTERNAL1; break;
146  #ifdef DEVICE_FLIPMOUSE
147  case HAL_IO_PIN_BUTTON_INT2: vb = VB_INTERNAL2; break;
148  #endif
149  default: return;
150  }
151 
152  //press or release?
153  if(gpio_get_level(pin) == 0)
154  {
155  //set press flag
156  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)),&xHigherPriorityTaskWoken);
157  //clear release flag
158  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)));
159 
160  //extra handling for long press
161  if(pin == HAL_IO_PIN_LONGACTION)
162  {
163  //check if timer is initialized, if yes reset & start
164  if(longactiontimer != NULL)
165  {
166  xTimerResetFromISR(longactiontimer,&xHigherPriorityTaskWoken);
167  xTimerStartFromISR(longactiontimer,&xHigherPriorityTaskWoken);
168  }
169  }
170  } else {
171  //set release flag
172  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)),&xHigherPriorityTaskWoken);
173  //clear press flag
174  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)));
175  //extra handling for long press
176  if(pin == HAL_IO_PIN_LONGACTION)
177  {
178  //check if timer is initialized, if yes reset & start
179  if(longactiontimer != NULL) xTimerStopFromISR(longactiontimer,&xHigherPriorityTaskWoken);
180  }
181  }
182  if(xResult == pdPASS) portYIELD_FROM_ISR();
183 }

210 void halIOIRRecvTask(void * param)
211 {
212  halIOIR_t* recv;
213  RingbufHandle_t rb = NULL;
214  //get RMT RX ringbuffer
215  rmt_get_ringbuf_handle(4, &rb);
216 
217  if(halIOIRRecvQueue == NULL)
218  {
219  ESP_LOGW(LOG_TAG, "halIOIRRecvQueue not initialised");
220  while(halIOIRRecvQueue == NULL) vTaskDelay(1000/portTICK_PERIOD_MS);
221  }
222 
223  while(1)
224  {
225  //wait for updates (triggered receiving)
226  if(xQueueReceive(halIOIRRecvQueue,&recv,portMAX_DELAY))
227  {
228  ESP_LOGI(LOG_TAG,"IR recv triggered.");
229 
230  //acquire buffer for ALL possible receiving elements
231  uint16_t offset = 0;
232  size_t rx_size = 0;
233  uint16_t local_timeout = 0;
234 
235  //check buffer
236  if(recv->buffer == NULL)
237  {
238  ESP_LOGE(LOG_TAG,"Please provide a buffer for receiving IR!");
239  continue;
240  }
241 
242  //start receiving on channel 4, flush all buffer elements
243  rmt_rx_start(4, 1);
244  //set target struct
245  recv->status = IR_RECEIVING;
246 
247  do{
248  //wait for one item until timeout or data is valid
249  rmt_item32_t* item = (rmt_item32_t*) xRingbufferReceive(rb, &rx_size, TASK_HAL_IR_RECV_EDGE_TIMEOUT/portTICK_PERIOD_MS);
250  //got one item
251  if(item != NULL) {
252  //put data into buffer
253  memcpy(&recv->buffer[offset], item, sizeof(rmt_item32_t)*rx_size);
254 
255  //invert active mode by inverting bits level0 and level1 in each item
256  for(size_t i = 0; i<rx_size;i++)
257  {
258  recv->buffer[offset+i].level0 = !recv->buffer[offset+i].level0;
259  recv->buffer[offset+i].level1 = !recv->buffer[offset+i].level1;
260  }
261  //increase offset for next received item
262  offset+= rx_size;
263 
264  //give item back to ringbuffer
265  vRingbufferReturnItem(rb, (void*) item);
266  //too much
267  if(offset >= TASK_HAL_IR_RECV_MAXIMUM_EDGES)
268  {
269  ESP_LOGE(LOG_TAG,"Too much IR edges, finished");
270  recv->status = IR_OVERFLOW;
271  recv->count = 0;
272  break;
273  }
274  } else {
275  local_timeout += TASK_HAL_IR_RECV_EDGE_TIMEOUT;
276  //check if timeout is the long one and no edges received...
277  if(local_timeout >= TASK_HAL_IR_RECV_TIMEOUT && offset == 0)
278  {
279  //timeout, cancel
280  rmt_rx_stop(4);
281  recv->status = IR_TOOSHORT;
282  recv->count = 0;
283  ESP_LOGE(LOG_TAG,"No cmd");
284  break;
285  }
286  //if we received already something and timeout triggers ->
287  //command finished
288  if(offset != 0)
289  {
290  //update status accordingly
291  if(offset > TASK_HAL_IR_RECV_MINIMUM_EDGES)
292  {
293  //save everything necessary to pointer from queue
294  recv->count = offset;
295  recv->status = IR_FINISHED;
296  ESP_LOGI(LOG_TAG,"Recorded @%d %d edges",(uint32_t)recv->buffer,offset);
297  } else {
298  //timeout, cancel
299  recv->status = IR_TOOSHORT;
300  recv->count = 0;
301  ESP_LOGE(LOG_TAG,"IR cmd too short");
302  }
303  rmt_rx_stop(4);
304  break;
305  }
306  }
307  } while(1);
308  }
309  }
