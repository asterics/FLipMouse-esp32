67 static void gpio_isr_handler(void* arg)
68 {
69  uint32_t pin = (uint32_t) arg;
70  uint8_t vb = 0;
71  BaseType_t xResult, xHigherPriorityTaskWoken = pdFALSE;
72 
73  switch(pin)
74  {
75  case HAL_IO_PIN_BUTTON_EXT1: vb = VB_EXTERNAL1; break;
76  case HAL_IO_PIN_BUTTON_EXT2: vb = VB_EXTERNAL2; break;
77  case HAL_IO_PIN_BUTTON_INT1: vb = VB_INTERNAL1; break;
78  case HAL_IO_PIN_BUTTON_INT2: vb = VB_INTERNAL2; break;
79  default: return;
80  }
81 
82  if(gpio_get_level(pin) == 0)
83  {
84  //set press flag
85  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)),&xHigherPriorityTaskWoken);
86  //clear release flag
87  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)));
88  } else {
89  //set release flag
90  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)),&xHigherPriorityTaskWoken);
91  //clear press flag
92  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)));
93  }
94  if(xResult == pdPASS) portYIELD_FROM_ISR();
95 }

163 void halIOIRRecvTask(void * param)
164 {
165  halIOIR_t* recv;
166  RingbufHandle_t rb = NULL;
167  //get RMT RX ringbuffer
168  rmt_get_ringbuf_handle(4, &rb);
169 
170  if(halIOIRRecvQueue == NULL)
171  {
172  ESP_LOGW(LOG_TAG, "halIOIRRecvQueue not initialised");
173  while(halIOIRRecvQueue == NULL) vTaskDelay(1000/portTICK_PERIOD_MS);
174  }
175 
176  while(1)
177  {
178  //wait for updates (triggered receiving)
179  if(xQueueReceive(halIOIRRecvQueue,&recv,10000))
180  {
181  //acquire buffer for ALL receiving elements
182  rmt_item32_t *buf = malloc(sizeof(rmt_item32_t)*TASK_HAL_IR_RECV_MAXIMUM_EDGES);
183  uint16_t offset = 0;
184  size_t rx_size = 0;
185 
186  //check buffer
187  if(buf == NULL)
188  {
189  ESP_LOGE(LOG_TAG,"Not enough memory for receiving IR commands!");
190  continue;
191  }
192 
193  //start receiving on channel 4, flush all buffer elements
194  rmt_rx_start(4, 1);
195  //set target struct
196  recv->status = IR_RECEIVING;
197  //wait for one item until timeout or data is valid
198  rmt_item32_t* item = (rmt_item32_t*) xRingbufferReceive(rb, &rx_size, TASK_HAL_IR_RECV_TIMEOUT/portTICK_PERIOD_MS);
199  //got one item
200  if(item) {
201  //put data into buffer
202  memcpy(&buf[offset], item, sizeof(rmt_item32_t)*rx_size);
203  offset+= rx_size;
204  //give item back to ringbuffer
205  vRingbufferReturnItem(rb, (void*) item);
206  //too much
207  if(offset == TASK_HAL_IR_RECV_MAXIMUM_EDGES)
208  {
209  ESP_LOGE(LOG_TAG,"Too much IR edges, finished");
210  recv->status = IR_OVERFLOW;
211  continue;
212  }
213  } else {
214  //timeout, cancel
215  rmt_rx_stop(4);
216  //update status accordingly
217  if(offset > TASK_HAL_IR_RECV_MINIMUM_EDGES)
218  {
219  //save everything necessary to pointer from queue
220  recv->buffer = buf;
221  recv->count = offset;
222  recv->status = IR_FINISHED;
223  } else {
224  recv->status = IR_TOOSHORT;
225  ESP_LOGE(LOG_TAG,"Too short, no cmd");
226  continue;
227  }
228  }
229  }
230  }

285 void halIOLEDTask(void * param)
286 {
287  uint32_t recv = 0;
288  uint32_t duty = 0;
289  uint32_t fade = 0;
290  if(halIOLEDQueue == NULL)
291  {
292  ESP_LOGW(LOG_TAG, "halIOLEDQueue not initialised");
293  while(halIOLEDQueue == NULL) vTaskDelay(1000/portTICK_PERIOD_MS);
294  }
295  while(1)
296  {
297  //wait for updates
298  if(xQueueReceive(halIOLEDQueue,&recv,10000))
299  {
300  //updates received, sending to ledc driver
301 
302  //get fading time, unit is 10ms
303  fade = ((recv & 0xFF000000) >> 24) * 10;
304 
305  //set fade with time (target duty and fading time)
306 
307  //1.) RED: map to 10bit and set to fading unit
308  duty = (recv & 0x000000FF) * 2 * 2;
309  ledc_set_fade_with_time(LEDC_HIGH_SPEED_MODE,LEDC_CHANNEL_0, \
310  duty, fade);
311 
312  //2.) GREEN: map to 10bit and set to fading unit
313  duty = ((recv & 0x0000FF00) >> 8) * 2 * 2;
314  ledc_set_fade_with_time(LEDC_HIGH_SPEED_MODE,LEDC_CHANNEL_1, \
315  duty, fade);
316 
317  //3.) BLUE: map to 10bit and set to fading unit
318  duty = ((recv & 0x00FF0000) >> 16) * 2 * 2;
319  ledc_set_fade_with_time(LEDC_HIGH_SPEED_MODE,LEDC_CHANNEL_2, \
320  duty, fade);
321 
322  //start fading for RGB
323  ledc_fade_start(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_0, LEDC_FADE_NO_WAIT);
324  ledc_fade_start(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_1, LEDC_FADE_NO_WAIT);
325  ledc_fade_start(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_2, LEDC_FADE_NO_WAIT);
