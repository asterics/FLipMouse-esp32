121 static void hidd_event_callback(esp_hidd_cb_event_t event, esp_hidd_cb_param_t *param)
122 {
123  switch(event) {
124  case ESP_HIDD_EVENT_REG_FINISH:
125  if (param->init_finish.state == ESP_HIDD_INIT_OK) {
126  //esp_bd_addr_t rand_addr = {0x04,0x11,0x11,0x11,0x11,0x05};
127  #ifdef DEVICE_FLIPMOUSE
128  esp_ble_gap_set_device_name("FLipMouse");
129  #else
130  esp_ble_gap_set_device_name("FABI");
131  #endif
132  esp_ble_gap_config_adv_data(&hidd_adv_data);
133  }
134  break;
135 
136  case ESP_BAT_EVENT_REG: break;
137  case ESP_HIDD_EVENT_DEINIT_FINISH: break;
138  case ESP_HIDD_EVENT_BLE_CONNECT:
139  ESP_LOGI(LOG_TAG, "ESP_HIDD_EVENT_BLE_CONNECT");
140  hid_conn_id = param->connect.conn_id;
141  break;
142  case ESP_HIDD_EVENT_BLE_DISCONNECT:
143  sec_conn = false;
144  ESP_LOGI(LOG_TAG, "ESP_HIDD_EVENT_BLE_DISCONNECT");
145  esp_ble_gap_start_advertising(&hidd_adv_params);
146  break;
147  case ESP_HIDD_EVENT_BLE_VENDOR_REPORT_WRITE_EVT:
148  ESP_LOGI(LOG_TAG, "%s, ESP_HIDD_EVENT_BLE_VENDOR_REPORT_WRITE_EVT", __func__);
149  ESP_LOG_BUFFER_HEX(LOG_TAG, param->vendor_write.data, param->vendor_write.length);
150  break;
151  default:
152  break;
153  }
154  return;

158 static void gap_event_handler(esp_gap_ble_cb_event_t event, esp_ble_gap_cb_param_t *param)
159 {
160  switch (event)
161  {
162  case ESP_GAP_BLE_ADV_DATA_SET_COMPLETE_EVT:
163  esp_ble_gap_start_advertising(&hidd_adv_params);
164  break;
165  case ESP_GAP_BLE_SEC_REQ_EVT:
166  for(int i = 0; i < ESP_BD_ADDR_LEN; i++) {
167  ESP_LOGD(LOG_TAG, "%x:",param->ble_security.ble_req.bd_addr[i]);
168  }
169  esp_ble_gap_security_rsp(param->ble_security.ble_req.bd_addr, true);
170  break;
171  case ESP_GAP_BLE_AUTH_CMPL_EVT:
172  sec_conn = true;
173  esp_bd_addr_t bd_addr;
174  memcpy(bd_addr, param->ble_security.auth_cmpl.bd_addr, sizeof(esp_bd_addr_t));
175  ESP_LOGI(LOG_TAG, "remote BD_ADDR: %08x%04x",\
176  (bd_addr[0] << 24) + (bd_addr[1] << 16) + (bd_addr[2] << 8) + bd_addr[3],
177  (bd_addr[4] << 8) + bd_addr[5]);
178  ESP_LOGI(LOG_TAG, "address type = %d", param->ble_security.auth_cmpl.addr_type);
179  ESP_LOGI(LOG_TAG, "pair status = %s",param->ble_security.auth_cmpl.success ? "success" : "fail");
180  if(!param->ble_security.auth_cmpl.success) {
181  ESP_LOGE(LOG_TAG, "fail reason = 0x%x",param->ble_security.auth_cmpl.fail_reason);
182  }
183  break;
184  default:
185  break;
186  }

195 void halBLETask(void * params)
196 {
197  hid_cmd_t rx;
198 
199  //Empty queue if initialized (there might be something left from last connection)
200  if(hid_ble != NULL) xQueueReset(hid_ble);
201 
202  //check if queue is initialized
203  if(hid_ble != NULL)
204  {
205  while(1)
206  {
207  //pend on MQ, if timeout triggers, just wait again.
208  if(xQueueReceive(hid_ble,&rx,portMAX_DELAY))
209  {
210  //if we are not connected, discard.
211  if(sec_conn == false) continue;
212 
213  //parse command (similar to usb_bridge controller)
214  switch(rx.cmd[0] & 0xF0)
215  {
216  //general cmds
217  case 0x00:
218  switch(rx.cmd[0] & 0x0F)
219  {
220  //reset report
221  case 0:
222  halBLEReset(0);
223  break;
224  //mouse X/Y report
225  case 1:
226  mouse_report[1] = rx.cmd[1];
227  mouse_report[2] = rx.cmd[2];
228  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
229  HID_RPT_ID_MOUSE_IN, HID_REPORT_TYPE_INPUT, HID_MOUSE_IN_RPT_LEN, mouse_report);
230  //reset the mouse_report's relative values (X/Y/wheel/pan)
231  mouse_report[1] = 0;
232  mouse_report[2] = 0;
233  break;
234  default: break;
235  }
236  break;
237 
238  //mouse handling
239  case 0x10:
240  switch(rx.cmd[0] & 0x0F)
241  {
242  case 0: //move X
243  mouse_report[1] = rx.cmd[1];
244  break;
245  case 1: //move Y
246  mouse_report[2] = rx.cmd[1];
247  break;
248  case 2: //move wheel
249  mouse_report[3] = rx.cmd[1];
250  break;
251  /* Press & release */
252  case 3: //left
253  mouse_report[0] |= (1<<0);
254  //send press report, wait for free EP (sending release is done after switch)
255  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
256  HID_RPT_ID_MOUSE_IN, HID_REPORT_TYPE_INPUT, HID_MOUSE_IN_RPT_LEN, mouse_report);
257  mouse_report[0] &= ~(1<<0);
258  break;
259  case 4: //right
260  mouse_report[0] |= (1<<1);
261  //send press report, wait for free EP (sending release is done after switch)
262  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
263  HID_RPT_ID_MOUSE_IN, HID_REPORT_TYPE_INPUT, HID_MOUSE_IN_RPT_LEN, mouse_report);
264  mouse_report[0] &= ~(1<<1);
265  break;
266  case 5: //middle
267  mouse_report[0] |= (1<<2);
268  //send press report, wait for free EP (sending release is done after switch)
269  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
270  HID_RPT_ID_MOUSE_IN, HID_REPORT_TYPE_INPUT, HID_MOUSE_IN_RPT_LEN, mouse_report);
271  mouse_report[0] &= ~(1<<2);
272  break;
273  /* Press */
274  case 6: //left
275  mouse_report[0] |= (1<<0);
276  break;
277  case 7: //right
278  mouse_report[0] |= (1<<1);
279  break;
280  case 8: //middle
281  mouse_report[0] |= (1<<2);
282  break;
283  /* Release */
284  case 9: //left
285  mouse_report[0] &= ~(1<<0);
286  break;
287  case 10: //right
288  mouse_report[0] &= ~(1<<1);
289  break;
290  case 11: //middle
291  mouse_report[0] &= ~(1<<2);
292  break;
293  /* Toggle */
294  case 12: //left
295  mouse_report[0] ^= (1<<0);
296  break;
297  case 13: //right
298  mouse_report[0] ^= (1<<1);
299  break;
300  case 14: //middle
301  mouse_report[0] ^= (1<<2);
302  break;
303  case 15: //reset mouse (excepting keyboard & joystick)
304  halBLEReset((1<<0)|(1<<1));
305  break;
306  }
307  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
308  HID_RPT_ID_MOUSE_IN, HID_REPORT_TYPE_INPUT, HID_MOUSE_IN_RPT_LEN, mouse_report);
309  //reset the mouse_report's relative values (X/Y/wheel/pan)
310  mouse_report[1] = 0;
311  mouse_report[2] = 0;
312  mouse_report[3] = 0;
313  mouse_report[4] = 0;
314  break;
315  //Keyboard handling
316  case 0x20:
317  switch(rx.cmd[0] & 0x0F)
318  {
319  case 0: //Press & release a key
320  //press key & send
321  add_keycode(rx.cmd[1], &keyboard_report[2]);
322  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
323  HID_RPT_ID_KEY_IN, HID_REPORT_TYPE_INPUT, HID_KEYBOARD_IN_RPT_LEN, keyboard_report);
324  //remove keycode
325  //sending the second report is done after this switch
326  remove_keycode(rx.cmd[1], &keyboard_report[2]);
327  break;
328  case 1: //Press a key
329  add_keycode(rx.cmd[1], &keyboard_report[2]);
330  break;
331  case 2: //Release a key
332  remove_keycode(rx.cmd[1], &keyboard_report[2]);
333  break;
334  case 3:
335  if(is_in_keycode_arr(rx.cmd[1],&keyboard_report[2])) remove_keycode(rx.cmd[1], &keyboard_report[2]);
336  else add_keycode(rx.cmd[1], &keyboard_report[2]);
337  break;
338  case 4: //Press & release a modifier (mask!)
339  keyboard_report[0] |= rx.cmd[1];
340  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
341  HID_RPT_ID_KEY_IN, HID_REPORT_TYPE_INPUT, HID_KEYBOARD_IN_RPT_LEN, keyboard_report);
342  //remove modifier
343  //sending the second report is done after this switch
344  keyboard_report[0] &= ~rx.cmd[1];
345  break;
346  case 5: //Press a modifier (mask!)
347  keyboard_report[0] |= rx.cmd[1];
348  break;
349  case 6: //Release a modifier (mask!)
350  keyboard_report[0] &= ~rx.cmd[1];
351  break;
352  case 7: //Toggle a modifier (mask!)
353  keyboard_report[0] ^= rx.cmd[1];
354  break;
355  case 15: //reset mouse (excepting mouse & joystick)
356  halBLEReset((1<<1)|(1<<2));
357  break;
358  }
359  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
360  HID_RPT_ID_KEY_IN, HID_REPORT_TYPE_INPUT, HID_KEYBOARD_IN_RPT_LEN, keyboard_report);
361  break;
362  case 0x30:
363  switch(rx.cmd[0] & 0x0F)
364  {
365  case 0: //Press & release button/hat
366  //test if it is buttons or hat?
367  if((rx.cmd[1] & (1<<7)) == 0)
368  {
369  //buttons, map to corresponding bits in 4 bytes
370  if(rx.cmd[1] <= 7) joystick_report[0] |= (1<<rx.cmd[1]);
371  else if(rx.cmd[1] <= 15) joystick_report[1] |= (1<<(rx.cmd[1]-8));
372  else if(rx.cmd[1] <= 24) joystick_report[2] |= (1<<(rx.cmd[1]-16));
373  else if(rx.cmd[1] <= 31) joystick_report[3] |= (1<<(rx.cmd[1]-24));
374  } else {
375  //hat, remove bit 7 and set to report (don't touch 4 bits of X)
376  joystick_report[4] = (joystick_report[4] & 0xF0) | (rx.cmd[1] & 0x0F);
377  }
378  //send press action
379  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
380  HID_RPT_ID_JOY_IN, HID_REPORT_TYPE_INPUT, HID_JOYSTICK_IN_RPT_LEN, joystick_report);
381  //release button/hat
382  //test if it is buttons or hat?
383  if((rx.cmd[1] & (1<<7)) == 0)
384  {
385  //buttons, map to corresponding bits in 4 bytes
386  if(rx.cmd[1] <= 7) joystick_report[0] &= ~(1<<rx.cmd[1]);
387  else if(rx.cmd[1] <= 15) joystick_report[1] &= ~(1<<(rx.cmd[1]-8));
388  else if(rx.cmd[1] <= 24) joystick_report[2] &= ~(1<<(rx.cmd[1]-16));
389  else if(rx.cmd[1] <= 31) joystick_report[3] &= ~(1<<(rx.cmd[1]-24));
390  } else {
391  //hat release means always 15.
392  joystick_report[4] = (joystick_report[4] & 0xF0) | 0x0F;
393  }
394  break;
395  case 1: //Press button/hat
396  //test if it is buttons or hat?
397  if((rx.cmd[1] & (1<<7)) == 0)
398  {
399  //buttons, map to corresponding bits in 4 bytes
400  if(rx.cmd[1] <= 7) joystick_report[0] |= (1<<rx.cmd[1]);
401  else if(rx.cmd[1] <= 15) joystick_report[1] |= (1<<(rx.cmd[1]-8));
402  else if(rx.cmd[1] <= 24) joystick_report[2] |= (1<<(rx.cmd[1]-16));
403  else if(rx.cmd[1] <= 31) joystick_report[3] |= (1<<(rx.cmd[1]-24));
404  } else {
405  //hat, remove bit 7 and set to report (don't touch 4 bits of X)
406  joystick_report[4] = (joystick_report[4] & 0xF0) | (rx.cmd[1] & 0x0F);
407  }
408  break;
409  case 2: //Release button/hat
410  //test if it is buttons or hat?
411  if((rx.cmd[1] & (1<<7)) == 0)
412  {
413  //buttons, map to corresponding bits in 4 bytes
414  if(rx.cmd[1] <= 7) joystick_report[0] &= ~(1<<rx.cmd[1]);
415  else if(rx.cmd[1] <= 15) joystick_report[1] &= ~(1<<(rx.cmd[1]-8));
416  else if(rx.cmd[1] <= 24) joystick_report[2] &= ~(1<<(rx.cmd[1]-16));
417  else if(rx.cmd[1] <= 31) joystick_report[3] &= ~(1<<(rx.cmd[1]-24));
418  } else {
419  //hat release means always 15.
420  joystick_report[4] = (joystick_report[4] & 0xF0) | 0x0F;
421  }
422  break;
423  case 4: //X Axis
424  //preserve 4 bits of hat
425  joystick_report[4] = (joystick_report[4] & 0x0F) | ((rx.cmd[1] & 0x0F) << 4);
426  //preserve 2 bits of Y
427  joystick_report[5] = (joystick_report[5] & 0xC0) | ((rx.cmd[1] & 0xF0) >> 4) | ((rx.cmd[2] & 0x03) << 4);
428  break;
429  case 5: //Y Axis
430  //preserve 6 bits of X
431  joystick_report[5] = (joystick_report[5] & 0x3F) | ((rx.cmd[1] & 0x03) << 6);
432  //save remaining Y
433  joystick_report[6] = ((rx.cmd[1] & 0xFC) >> 2) | ((rx.cmd[2] & 0x03) << 6);
434  break;
435  case 6: //Z Axis
436  joystick_report[7] = rx.cmd[1];
437  joystick_report[8] = (joystick_report[8] & 0xFC) | (rx.cmd[2] & 0x03);
438  break;
439  case 7: //Z-rotate
440  //preserve 2 bits of Z-axis
441  joystick_report[8] = (joystick_report[8] & 0x03) | ((rx.cmd[1] & 0x3F) << 2);
442  //preserve slider left & combine 2 bits of LSB & MSB to one nibble
443  joystick_report[9] = (joystick_report[9] & 0xF0) | ((rx.cmd[1] & 0xC0) >> 6) | ((rx.cmd[2] & 0x03) << 2);
444  break;
445  case 8: //slider left
446  //preserve 4 bits of Z-rotate, add low nibble of first byte
447  joystick_report[9] = (joystick_report[9] & 0x0F) | ((rx.cmd[1] & 0x0F) << 4);
448  //preserve 2 bits of slider right, add high nibble of first byte and second byte
449  joystick_report[10] = (joystick_report[10] & 0xC0) | ((rx.cmd[1] & 0xF0) >> 4) | ((rx.cmd[2] & 0x03) << 4);
450  break;
451  case 9: //slider right
452  //preserve 6 bits of slider left, add 2 bits for slider right
453  joystick_report[10] = (joystick_report[10] & 0x3F) | ((rx.cmd[1] & 0x03) << 6);
454  //save remaining slider right
455  joystick_report[11] = ((rx.cmd[1] & 0xFC) >> 2) | ((rx.cmd[2] & 0x03) << 6);
456  break;
457  case 15: //reset joystick (excepting mouse & keyboard)
458  halBLEReset((1<<0)|(1<<2));
459  break;
460  }
461  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
462  HID_RPT_ID_JOY_IN, HID_REPORT_TYPE_INPUT, HID_JOYSTICK_IN_RPT_LEN, joystick_report);
463  break;
464  }
465  }
466  }
467  } else {
468  ESP_LOGE(LOG_TAG,"ble hid queue not initialized, retry in 1s");
469  vTaskDelay(1000/portTICK_PERIOD_MS);
470  }
471 }

515 void halBLEReset(uint8_t exceptDevice)
516 {
517  uint8_t send = 0;
518 
519  //check if we need to reset the mouse report
520  if(!(exceptDevice & (1<<2))) {
521  //compare the global mouse report against an empty one
522  uint8_t m[sizeof(mouse_report)] = {0};
523  if(memcmp(mouse_report,m,sizeof(mouse_report)) != 0) {
524  //if the global one wasn't empty before, we empty and send it
525  memset(mouse_report,0,sizeof(mouse_report));
526  send = 1;
527  }
528  }
529  //same procedure as for the mouse....
530  if(!(exceptDevice & (1<<0))) {
531  uint8_t k[sizeof(keyboard_report)] = {0};
532  if(memcmp(keyboard_report,k,sizeof(keyboard_report)) != 0) {
533  memset(keyboard_report,0,sizeof(keyboard_report));
534  send = 1;
535  }
536  }
537  //and once again for the joystick.
538  if(!(exceptDevice & (1<<1))) {
539  uint8_t j[sizeof(keyboard_report)] = {0};
540  if(memcmp(joystick_report,j,sizeof(joystick_report)) != 0) {
541  memset(joystick_report,0,sizeof(joystick_report));
542  send = 1;
543  }
544  }
545 
546  //we don't need to send empty reports all the time, just if they
547  //weren't empty before.
548  if(send == 0) return;
549 
550  if(sec_conn && activateMouse) {
551  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
552  HID_RPT_ID_MOUSE_IN, HID_REPORT_TYPE_INPUT, HID_MOUSE_IN_RPT_LEN, mouse_report);
553  }
554 
555  if(sec_conn && activateKeyboard) {
556  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
557  HID_RPT_ID_KEY_IN, HID_REPORT_TYPE_INPUT, HID_KEYBOARD_IN_RPT_LEN, keyboard_report);
558  }
559 
560  if(sec_conn && activateJoystick) {
561  hid_dev_send_report(hidd_le_env.gatt_if, hid_conn_id,
562  HID_RPT_ID_JOY_IN, HID_REPORT_TYPE_INPUT, HID_JOYSTICK_IN_RPT_LEN, joystick_report);
563  }
564 }

568 esp_err_t halBLEInit(uint8_t enableKeyboard, uint8_t enableMouse, uint8_t enableJoystick)
569 {
570  activateKeyboard = enableKeyboard;
571  activateMouse = enableMouse;
572  activateJoystick = enableJoystick;
573 
574  // Initialize NVS.
575  esp_err_t ret = nvs_flash_init();
576 
577  esp_bt_controller_mem_release(ESP_BT_MODE_CLASSIC_BT);
578  esp_bt_controller_config_t bt_cfg = BT_CONTROLLER_INIT_CONFIG_DEFAULT();
579  ret = esp_bt_controller_init(&bt_cfg);
580  if (ret) {
581  ESP_LOGE(LOG_TAG, "%s initialize controller failed\n", __func__);
582  return ESP_FAIL;
583  }
584 
585  ret = esp_bt_controller_enable(ESP_BT_MODE_BLE);
586  if (ret) {
587  ESP_LOGE(LOG_TAG, "%s enable controller failed\n", __func__);
588  return ESP_FAIL;
589  }
590 
591  ret = esp_bluedroid_init();
592  if (ret) {
593  ESP_LOGE(LOG_TAG, "%s init bluedroid failed\n", __func__);
594  return ESP_FAIL;
595  }
596 
597  ret = esp_bluedroid_enable();
598  if (ret) {
599  ESP_LOGE(LOG_TAG, "%s init bluedroid failed\n", __func__);
600  return ESP_FAIL;
601  }
602 
603  if(!hidd_le_env.enabled) {
604  memset(&hidd_le_env, 0, sizeof(hidd_le_env_t));
605  hidd_le_env.enabled = true;
606  }
607
