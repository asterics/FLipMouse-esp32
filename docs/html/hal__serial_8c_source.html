126 void halSerialRXTask(void *pvParameters)
127 {
128  uint16_t cmdoffset = 0;
129  uint8_t *buf;
130  uint8_t *bufstatic = malloc(ATCMD_LENGTH);
131  uint8_t data;
132  atcmd_t currentcmd;
133 
134  if(bufstatic == NULL)
135  {
136  ESP_LOGE(LOG_TAG,"Cannot allocate mem");
137  vTaskDelete(NULL);
138  }
139 
140  while(1)
141  {
142  int len = uart_read_bytes(HAL_SERIAL_UART, &data, 1, portMAX_DELAY);
143 
144  //only parse if a valid byte is received
145  if(len == 1)
146  {
147  //check if we got a \r or \n at the beginning
148  //(which is a remaining from the previous command)
149  if((data == '\r' || data == '\n') && (cmdoffset == 0))
150  {
151  //do nothing with it, just discard
152  continue;
153  }
154 
155  //now read data until we reach \r or \n
156  if(data == '\r' || data == '\n')
157  {
158  //terminate string
159  bufstatic[cmdoffset] = 0;
160  //allocate a new buffer for this command to be processed later
161  //free is done in halSerialReceiveUSBSerial.
162  buf = malloc(cmdoffset+1);
163  if(buf == NULL)
164  {
165  ESP_LOGE(LOG_TAG,"Cannot allocate %d B buffer for new AT cmd",cmdoffset+1);
166  cmdoffset = 0;
167  break;
168  }
169  memcpy(buf,bufstatic,cmdoffset+1);
170 
171  //send buffer to queue
172  currentcmd.buf = buf;
173  currentcmd.len = cmdoffset+1;
174  if(halSerialATCmds != NULL)
175  {
176  if(xQueueSend(halSerialATCmds,(void*)&currentcmd,10) != pdTRUE)
177  {
178  ESP_LOGE(LOG_TAG,"AT cmd queue is full, cannot send cmd");
179  free(buf);
180  } else {
181  #if LOG_LEVEL_SERIAL >= ESP_LOG_INFO
182  ESP_LOGI(LOG_TAG,"Sent AT cmd with len %d to queue: %s",cmdoffset,bufstatic);
183  #endif
184  }
185  } else {
186  ESP_LOGE(LOG_TAG,"AT cmd queue is NULL, cannot send cmd");
187  free(buf);
188  }
189  cmdoffset = 0;
190  continue;
191  }
192 
193  //if everything is fine, just save this byte.
194  bufstatic[cmdoffset] = data;
195  cmdoffset++;
196 
197  //check for memory length
198  if(cmdoffset == ATCMD_LENGTH)
199  {
200  ESP_LOGW(LOG_TAG,"AT cmd too long, discarding");
201  cmdoffset = 0;
202  }
203  }
204  }
205 
206  //we should never be here...
207  vTaskDelete(NULL);
208 }

279 void halSerialHIDTask(void *param)
280 {
281  hid_cmd_t rx;
282  while(1)
283  {
284  //check if queue is initialized
285  if(hid_usb != NULL)
286  {
287  //pend on MQ, if timeout triggers, just wait again.
288  if(xQueueReceive(hid_usb,&rx,portMAX_DELAY))
289  {
290  //output if debug
291  #if LOG_LEVEL_SERIAL >= ESP_LOG_DEBUG
292  ESP_LOGD(LOG_TAG,"HID: %02X:%02X:%02X",rx.cmd[0],rx.cmd[1],rx.cmd[2]);
293  #endif
294  i2c_cmd_handle_t cmd = i2c_cmd_link_create();
295  i2c_master_start(cmd);
296  i2c_master_write_byte(cmd, (HAL_SERIAL_I2C_ADDR_LPC << 1) | WRITE_BIT, ACK_CHECK_EN);
297  i2c_master_write(cmd, rx.cmd, 3, ACK_CHECK_EN);
298  i2c_master_stop(cmd);
299  esp_err_t ret = i2c_master_cmd_begin(I2C_NUM_0, cmd, 1000 / portTICK_RATE_MS);
300  i2c_cmd_link_delete(cmd);
301  //we don't care about return code.
302  //I2C driver sometimes return TIMEOUT...
303  (void) ret;
304 
305  if(ret != ESP_OK)
306  {
307  ESP_LOGW(LOG_TAG,"I2C didn't succeed: 0x%X",ret);
308  halSerialInitI2C(true);
309  } else {
310  #if LOG_LEVEL_SERIAL >= ESP_LOG_DEBUG
311  ESP_LOGD(LOG_TAG,"I2C succeed");
312  #endif
313  }
314  }
315  } else {
316  ESP_LOGW(LOG_TAG,"usb hid queue not initialized, retry in 1s");
317  vTaskDelay(1000/portTICK_PERIOD_MS);
318  }
319  }

331 int halSerialReceiveI2CADC(uint8_t **data)
332 {
333  //we define 10bytes to be read.
334  int size = 10;
335  i2c_cmd_handle_t cmd = i2c_cmd_link_create();
336  i2c_master_start(cmd);
337  i2c_master_write_byte(cmd, (HAL_SERIAL_I2C_ADDR_LPC << 1) | READ_BIT, ACK_CHECK_EN);
338  if (size > 1) {
339  i2c_master_read(cmd, *data, size - 1, ACK_VAL);
340  }
341  i2c_master_read_byte(cmd, *data + size - 1, NACK_VAL);
342  i2c_master_stop(cmd);
343  esp_err_t ret = i2c_master_cmd_begin(I2C_NUM_0, cmd, 1000 / portTICK_RATE_MS);
344  i2c_cmd_link_delete(cmd);
345  if(ret == ESP_OK) return size;
346  else
347  {
348  //try to re-initialize for next call.
349  halSerialInitI2C(true);
350  return -1;
351  }
352 }

409 int halSerialSendUSBSerial(char *data, uint32_t length, TickType_t ticks_to_wait)
410 {
411  //check if sem is already initialized
412  if(serialsendingsem == NULL)
413  {
414  ESP_LOGW(LOG_TAG,"Sem not ready, waiting for 1s");
415  vTaskDelay(1000/portTICK_PERIOD_MS);
416  }
417 
418  //if an additional stream is registered, send data there as well
419  //no HID commands will be sent there.
420  if(outputcb != NULL)
421  {
422  if(outputcb(data,length) != ESP_OK)
423  {
424  ESP_LOGE(LOG_TAG,"Additional stream cannot be sent,removing stream!");
425  outputcb = NULL;
426  }
427  }
428 
429  //acquire mutex to have TX permission on UART
430  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
431  {
432  //send the data
433  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
434  uart_write_bytes(HAL_SERIAL_UART, HAL_SERIAL_LINE_ENDING, strnlen(HAL_SERIAL_LINE_ENDING,4));
435  //release mutex
436  xSemaphoreGive(serialsendingsem);
437 
438  return txBytes;
439  } else return -1;
440 }

525 esp_err_t halSerialInit(void)
526 {
527  //set log level to given log level
528  esp_log_level_set(LOG_TAG,LOG_LEVEL_SERIAL);
529 
530  /*++++ UART config (UART to CDC) ++++*/
531 
532  esp_err_t ret = ESP_OK;
533  const uart_config_t uart_config = {
534  .baud_rate = 115200,
535  .data_bits = UART_DATA_8_BITS,
536  .parity = UART_PARITY_DISABLE,
537  .stop_bits = UART_STOP_BITS_1,
538  .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
539  };
540 
541  //update UART config
542  ret = uart_param_config(HAL_SERIAL_UART, &uart_config);
543  if(ret != ESP_OK)
544  {
545  ESP_LOGE(LOG_TAG,"UART param config failed");
546  return ret;
547  }
548 
549  //set IO pins
550  ret = uart_set_pin(HAL_SERIAL_UART, HAL_SERIAL_TXPIN, HAL_SERIAL_RXPIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
551  if(ret != ESP_OK)
552  {
553  ESP_LOGE(LOG_TAG,"UART set pin failed");
554  return ret;
555  }
556 
557  //Install UART driver with RX and TX buffers
558  ret = uart_driver_install(HAL_SERIAL_UART, BUF_SIZE_RX, BUF_SIZE_TX, 0, NULL, 0);
559  if(ret != ESP_OK)
560  {
561  ESP_LOGE(LOG_TAG,"UART driver install failed");
562  return ret;
563  }
564 
565  //create mutex for all tasks sending to serial TX queue.
566  //avoids splitting of different packets due to preemption
567  serialsendingsem = xSemaphoreCreateMutex();
568  if(serialsendingsem == NULL)
569  {
570  ESP_LOGE(LOG_TAG,"Cannot create semaphore for TX");
571  return ESP_FAIL;
572  }
573 
574  //create the AT command queue
575  halSerialATCmds = xQueueCreate(CMDQUEUE_SIZE,sizeof(atcmd_t));
576 
577  /*++++ I2C config (sending HID commands; receiving ADC data) ++++*/
578  halSerialInitI2C(false);
579 
580 
581  /*++++ task setup ++++*/
582  //task for sending HID commands via I2C
583  xTaskCreate(halSerialHIDTask, "serialHID", HAL_SERIAL_TASK_STACKSIZE+256, NULL, configMAX_PRIORITIES-3, NULL);
584 
585  //Create a task to handler UART event from ISR
586  xTaskCreate(halSerialRXTask, "serialRX", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-3, NULL);
587 
588  //everything went fine
589  ESP_LOGI(LOG_TAG,"Driver installation complete");
590  return ESP_OK;
591 }
