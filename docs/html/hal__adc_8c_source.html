162 void halAdcProcessStrongMode(adcData_t *D)
163 {
164  raw_action_t evt;
165  //on a FABI device, we cannot do this combination with analog values.
166  //maybe it will be done on another firmware part.
167  #ifdef DEVICE_FABI
168  D->strongmode = STRONG_NORMAL;
169  return;
170  #endif
171 
172  //cannot do anything here if we are not in a special mode.
173  if(D->strongmode == STRONG_NORMAL) return;
174 
175  #ifdef DEVICE_FLIPMOUSE
176  if(adcStrongTimerHandle == NULL || adcStrongTimeoutTimerHandle == NULL)
177  {
178  ESP_LOGE(LOG_TAG,"Strong mode timer uninitialized!!!");
179  return;
180  }
181 
182  //timer is not started
183  if(xTimerIsTimerActive(adcStrongTimeoutTimerHandle) == pdFALSE)
184  {
185  //than we need to start a timer for the timeout moving back to NORMAL
186  xTimerReset(adcStrongTimeoutTimerHandle,0);
187  //in addition, we start a timer to wait for a defined delay
188  xTimerReset(adcStrongTimerHandle,0);
189  //return in this case & wait for next iteration
190  return;
191  }
192 
193  //check if we have a movement in any direction
194  if(D->x != 0 || D->y != 0)
195  {
196  //check if it is possible to trigger the next action
197  //(delay has passed -> adcStrongSem is free)
198  if(uxSemaphoreGetCount(adcStrongSem) == 0) return;
199  xSemaphoreTake(adcStrongSem,0);
200 
201  //cancel timer for idle time
202  xTimerStop(adcStrongTimerHandle,0);
203 
204  //if yes, trigger action (depending on SIP/PUFF mode)
205  if(D->strongmode == STRONG_PUFF)
206  {
207  if(abs(D->x) > abs(D->y))
208  {
209  //x has higher values -> use LEFT/RIGHT
210  if(D->x > 0) evt.vb = VB_STRONGPUFF_RIGHT;
211  else evt.vb = VB_STRONGPUFF_LEFT;
212  evt.type = VB_PRESS_EVENT;
213  xQueueSendToBack(debouncer_in,&evt,0);
214  ESP_LOGI(LOG_TAG,"Exit STRONG: PUFF + LEFT/RIGHT");
215  } else {
216  //y has higher values -> use UP/DOWN
217  if(D->y > 0) evt.vb = VB_STRONGPUFF_DOWN;
218  else evt.vb = VB_STRONGPUFF_UP;
219  evt.type = VB_PRESS_EVENT;
220  xQueueSendToBack(debouncer_in,&evt,0);
221  ESP_LOGI(LOG_TAG,"Exit STRONG: PUFF + UP/DOWN");
222 
223  }
224  //stop the timeout timer
225  xTimerStop(adcStrongTimeoutTimerHandle,0);
226  //reset strong mode after sending the action
227  D->strongmode = STRONG_NORMAL;
228  }
229  if(D->strongmode == STRONG_SIP)
230  {
231  if(abs(D->x) > abs(D->y))
232  {
233  //x has higher values -> use LEFT/RIGHT
234  if(D->x > 0) evt.vb = VB_STRONGSIP_RIGHT;
235  else evt.vb = VB_STRONGSIP_LEFT;
236  evt.type = VB_PRESS_EVENT;
237  xQueueSendToBack(debouncer_in,&evt,0);
238  ESP_LOGI(LOG_TAG,"Exit STRONG: SIP + LEFT/RIGHT");
239  } else {
240  //y has higher values -> use UP/DOWN
241  if(D->y > 0) evt.vb = VB_STRONGSIP_DOWN;
242  else evt.vb = VB_STRONGSIP_UP;
243  evt.type = VB_PRESS_EVENT;
244  xQueueSendToBack(debouncer_in,&evt,0);
245  ESP_LOGI(LOG_TAG,"Exit STRONG: SIP + UP/DOWN");
246  }
247  //stop the timeout timer
248  xTimerStop(adcStrongTimeoutTimerHandle,0);
249  //reset strong mode after sending the action
250  D->strongmode = STRONG_NORMAL;
251  }
252  }
253  #endif

338 int halAdcReadData(adcData_t *values)
339 {
340  //read all sensors
341  int32_t tmp = 0;
342  int32_t x,y;
343  int32_t up,down,left,right,pressure;
344  static int32_t up_p,down_p,left_p,right_p,pressure_p;
345  up = down = left = right = pressure = 0;
347  static uint32_t debug_out_cnt = 0;
348 
349  //read sensor data via I2C.
350  uint8_t adc[10];
351  uint8_t *adc2 = adc;
352  int rcv = halSerialReceiveI2CADC(&adc2);
353  if(rcv != 10)
354  {
355  ESP_LOGW(LOG_TAG,"I2C recv: 0x%X",rcv);
356  }
357  left = (int32_t)((uint16_t)(adc[2] + (adc[3] << 8)));
358  right = (int32_t)((uint16_t)(adc[6] + (adc[7] << 8)));
359  up = (int32_t)((uint16_t)(adc[4] + (adc[5] << 8)));
360  down = (int32_t)((uint16_t)(adc[0] + (adc[1] << 8)));
361  pressure = (int32_t)((uint16_t)(adc[8] + (adc[9] << 8)));
362  //ESP_LOGE(LOG_TAG,"%d,%d,%d,%d,%d",up,down,left,right,pressure);
363 
364  //before any further processing, check if we have an unusual
365  //sensor reading (deviation to last measuring > 200)
366  //in this case, discard this measurement and wait for next iteration
367  //if it is a wanted high mouthpiece movement, next iteration will be
368  //triggering the action (we save the last value, even if discarded)
369  int toohigh = 0;
370  if((abs(left-left_p) > 200) || (abs(right-right_p) > 200) || (abs(up-up_p) > 200) || \
371  (abs(down-down_p) > 200) || (abs(pressure-pressure_p) > 200)) { toohigh = 1; }
372 
373  left_p = left;
374  right_p = right;
375  up_p = up;
376  down_p = down;
377  pressure_p = pressure;
378  if(toohigh != 0)
379  {
380  ESP_LOGW(LOG_TAG,"sensor deviation over rate,discarding");
381  return -1;
382  }
383 
384  //do the mouse rotation
385  switch (adc_conf.orientation) {
386  case 90: tmp=up; up=left; left=down; down=right; right=tmp; break;
387  case 180: tmp=up; up=down; down=tmp; tmp=right; right=left; left=tmp; break;
388  case 270: tmp=up; up=right; right=down; down=left; left=tmp;break;
389  }
390 
391  //save raw values to data struct
392  values->left = left;
393  values->right = right;
394  values->up = up;
395  values->down = down;
396 
397  //process pressure
398  values->pressure = pressure;
399 
400  //calculate X and Y values
401  x = (left - right) - offsetx;
402  y = (up - down) - offsety;
403 
404  #if HAL_IO_ADC_ELLIPTIC_DEADZONE == 0
405 
406  //in this case, we use a rectangle deadzone
407 
408  if (x<-adc_conf.deadzone_x) x+=adc_conf.deadzone_x; // apply deadzone values x direction
409  else if (x>adc_conf.deadzone_x) x-=adc_conf.deadzone_x;
410  else x=0;
411 
412  if (y<-adc_conf.deadzone_y) y+=adc_conf.deadzone_y; // apply deadzone values y direction
413  else if (y>adc_conf.deadzone_y) y-=adc_conf.deadzone_y;
414  else y=0;
415 
416  //otherwise, x&y is 0
417  values->x = x;
418  values->y = y;
419 
420  #else
421 
422  //in this case, we use an elliptic deadzone
423 
424  //apply elliptic deadzone
425  //formula:
426  //https://sebadorn.de/2012/01/02/herausfinden-ob-ein-punkt-in-einer-ellipse-liegt
427  //A point in an elliptic curve:
428  //(px - cx) / rx + (py - cy) / ry <= 1
429  //cx/cy is 0 (we start at 0/0)
430  //py/py is our x/y
431  //rx/ry are deadzone values
432  //to shorten the formulas
433  uint8_t a = adc_conf.deadzone_x;
434  uint8_t b = adc_conf.deadzone_y;
435  float status = pow(x,2) / pow(a,2) + pow(y,2) / pow(b,2);
436 
437  //check if point is outside deadzone
438  if(status > 1.0)
439  {
440  //if outside deadzone, subtract ellipse itself to start with
441  //0 for a movement
442 
443  //formula for ellipse point:
444  //https://math.stackexchange.com/questions/22064/calculating-a-point-that-lies-on-an-ellipse-given-an-angle
445  float deadzoneX = 0;
446  float deadzoneY = 0;
447 
448  //angle of the given mouthpiece
449  //calculate only if x&y is != 0
450  if((x < 0 || x > 0) && (y < 0 || y > 0))
451  {
452  float angle = atan(y/x);
453  deadzoneX = abs((a*b)/sqrt(pow(b,2)+pow(a,2)*pow(tan(angle),2)));
454  deadzoneY = abs((a*b)/sqrt(pow(a,2)+ pow(b,2)/pow(tan(angle),2)));
455  }
456 
457  //subtract calculated ellipse coordinates from output X/Y values
458  if(x > 0)
459  {
460  values->x = x - (int)deadzoneX;
461  } else {
462  values->x = x + (int)deadzoneX;
463  }
464  if(y > 0)
465  {
466  values->y = y - (int)deadzoneY;
467  } else {
468  values->y = y + (int)deadzoneY;
469  }
470  } else {
471  //otherwise, x&y is 0
472  values->x = 0;
473  values->y = 0;
474  }
475  #endif
476  if(debug_out_cnt++%HAL_ADC_RAW_DIVIDER == 0)
477  {
478  ESP_LOGD(LOG_TAG,"raw x/y %d/%d; ",values->x,values->y);
479  }
480  return 0;
481 }

501 void halAdcProcessPressure(adcData_t *D)
502 {
503  //track changes and fire event only if not issued already.
504  //this is done individually for SIP, PUFF, STRONG_SIP & STRONG_PUFF
505  //0 means nothing is fired, 1 is a fired press action, 2 is a fired release action
506  static uint8_t fired[4] = {0,0,0,0};
507  //currently measured pressure value
508  uint32_t pressurevalue = D->pressure;
509  //currently active general config
510  generalConfig_t *cfg = configGetCurrent();
511  //the raw event, sent to the debouncer
512  raw_action_t evt;
513  //cannot proceed if no global config is available
514  if(cfg == NULL) return;
515 
516  //if we are in a special mode don't process further (only on FLipMouse)
517  #ifdef DEVICE_FLIPMOUSE
518  if(D->strongmode != STRONG_NORMAL) return;
519  #endif
520 
521  //SIP triggered
522  if(pressurevalue < cfg->adc.threshold_sip && \
523  pressurevalue > cfg->adc.threshold_strongsip)
524  {
525  if(fired[0] != 1)
526  {
527  //set/clear VBs
528  evt.type = VB_PRESS_EVENT;
529  evt.vb = VB_SIP;
530  xQueueSendToBack(debouncer_in,&evt,0);
531  //save fired state
532  fired[0] = 1;
533  }
534  } else {
535  if(fired[0] == 1)
536  {
537  //set/clear VBs
538  evt.type = VB_RELEASE_EVENT;
539  evt.vb = VB_SIP;
540  xQueueSendToBack(debouncer_in,&evt,0);
541  //track fired state
542  fired[0] = 0;
543  }
544  }
545 
546  //STRONGSIP triggered
547  if(pressurevalue < cfg->adc.threshold_strongsip)
548  {
549  //check if strong sip + up/down/left/right is set.
550  //if this is the case, we will proceed with strong mode.
551  //Otherwise the VB_STRONGSIP will be triggered.
552  #ifdef DEVICE_FLIPMOUSE
553  if(handler_hid_active(VB_STRONGSIP_UP) || handler_vb_active(VB_STRONGSIP_UP) || \
554  handler_hid_active(VB_STRONGSIP_DOWN) || handler_vb_active(VB_STRONGSIP_DOWN) || \
555  handler_hid_active(VB_STRONGSIP_LEFT) || handler_vb_active(VB_STRONGSIP_LEFT) || \
556  handler_hid_active(VB_STRONGSIP_RIGHT) || handler_vb_active(VB_STRONGSIP_RIGHT))
557  {
558  //if at least one strong action is defined, enter strong sip mode
559  D->strongmode = STRONG_SIP;
560  ESP_LOGI(LOG_TAG,"Enter STRONG SIP");
561  TONE(TONE_STRONGSIP_ENTER_FREQ,TONE_STRONGSIP_ENTER_DURATION);
562  } else {
563  #endif
564  if(fired[1] != 1)
565  {
566  //make a tone
567  //TONE(TONE_STRONGSIP_ENTER_FREQ,TONE_STRONGSIP_ENTER_DURATION);
568  //no strong sip + <yy> action is defined, trigger strong sip VB.
569 
570  evt.type = VB_PRESS_EVENT;
571  evt.vb = VB_STRONGSIP;
572  xQueueSendToBack(debouncer_in,&evt,0);
573  //save fired stated
574  fired[1] = 1;
575  }
576  #ifdef DEVICE_FLIPMOUSE
577  }
578  #endif
579  } else {
580  if(fired[1] != 2)
581  {
582  //set/clear VBs
583  evt.type = VB_RELEASE_EVENT;
584  evt.vb = VB_STRONGSIP;
585  xQueueSendToBack(debouncer_in,&evt,0);
586  //track fired state
587  fired[1] = 2;
588  }
589  }
590 
591  //PUFF triggered
592  if(pressurevalue > cfg->adc.threshold_puff && \
593  pressurevalue < cfg->adc.threshold_strongpuff)
594  {
595 
596  if(fired[2] != 1)
597  {
598  //set/clear VBs
599  evt.type = VB_PRESS_EVENT;
600  evt.vb = VB_PUFF;
601  xQueueSendToBack(debouncer_in,&evt,0);
602  //save fired state
603  fired[2] = 1;
604  }
605  } else {
606  if(fired[2] == 1)
607  {
608  //set/clear VBs
609  evt.type = VB_RELEASE_EVENT;
610  evt.vb = VB_PUFF;
611  xQueueSendToBack(debouncer_in,&evt,0);
612  //track fired state
613  fired[2] = 0;
614  }
615  }
616 
617  //STRONGPUFF triggered
618  if(pressurevalue > cfg->adc.threshold_strongpuff)
619  {
620  //check if strong puff + up/down/left/right is set.
621  //if this is the case, we will proceed with strong mode.
622  //Otherwise the VB_STRONGPUFF will be triggered.
623  #ifdef DEVICE_FLIPMOUSE

648  #ifdef DEVICE_FLIPMOUSE
649  }

651  } else {
652  if(fired[3] != 2)
653  {
654  //set/clear VBs
655  evt.type = VB_RELEASE_EVENT;
656  evt.vb = VB_STRONGPUFF;
657  xQueueSendToBack(debouncer_in,&evt,0);

679 void halAdcTaskMouse(void * pvParameters)
680 {
681  //analog values
682  adcData_t D;
683  D.strongmode = STRONG_NORMAL;
684  //int32_t x,y;
685  static uint16_t accelTimeX=0,accelTimeY=0;
686  int32_t tempX,tempY;
687  float moveVal, accumXpos = 0, accumYpos = 0;
688  float accelFactor= 20 / 100000000.0f;
689  hid_cmd_t command,command2;
690  TickType_t xLastWakeTime = xTaskGetTickCount();
691  //set adc data reference for timer
692  vTimerSetTimerID(adcStrongTimeoutTimerHandle,&D);
693  uint32_t debug_out_cnt = 0;
694 
695  while(1)
696  {
697  //get mutex
698  if(xSemaphoreTake(adcSem, (TickType_t) 30) != pdTRUE)
699  {
700  ESP_LOGW(LOG_TAG,"Cannot obtain mutex for reading");
701  continue;
702  }
703 
704  //read out the analog voltages from all 5 channels (including deadzone)
705  //& set calibrate request to 0 before
706  D.calibrate_request = 0;
707  uint8_t retry = 0;
708  while((halAdcReadData(&D) != 0) && (retry++ < 10));
709  if(retry == 10)
710  {
711  ESP_LOGE(LOG_TAG,"Cannot read ADC");
712  vTaskDelay(1000/portTICK_PERIOD_MS);
713  xSemaphoreGive(adcSem);
714  }
715 
716 
717  //if you want to slow down, uncomment following two lines
718  //ESP_LOGD(LOG_TAG,"X/Y square, X/Y ellipse: %d/%d, %d/%d",tempX,tempY,D.x,D.y);
719  //vTaskDelay(10);
720 
721  //report raw values.
722  halAdcReportRaw(D.up, D.down, D.left, D.right, D.pressure, D.x, D.y);
723 
724  //if we are in a special strong mode, do NOT send accumulated data
725  //to USB/BLE. Instead, call halAdcProcessStrongMode
726  //if in normal mode, proceed with mouse
727  if(D.strongmode == STRONG_NORMAL)
728  {
729  //apply acceleration
730  if (D.x==0) accelTimeX=0;
731  else if (accelTimeX < ACCELTIME_MAX) accelTimeX+=adc_conf.acceleration;
732  if (D.y==0) accelTimeY=0;
733  else if (accelTimeY < ACCELTIME_MAX) accelTimeY+=adc_conf.acceleration;
734 
735  //calculate the current X movement by using acceleration, accel factor and sensitivity
736  moveVal = D.x * adc_conf.sensitivity_x * accelFactor * accelTimeX;
737  //limit value
738  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
739  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
740  //add to accumulated movement value
741  accumXpos+=moveVal;
742 
743  //do the same calculations for Y axis
744  moveVal = D.y * adc_conf.sensitivity_y * accelFactor * accelTimeY;
745  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
746  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
747  accumYpos+=moveVal;
748 
749  //cast to int again
750  tempX = accumXpos;
751  tempY = accumYpos;
752 
753  //limit to int8 values (to fit into mouse report)
754  if(tempX > 127) tempX = 127;
755  if(tempX < -127) tempX = -127;
756  if(tempY > 127) tempY = 127;
757  if(tempY < -127) tempY = -127;
758 
759  #if LOG_LEVEL_ADC >= ESP_LOG_DEBUG
760  if(debug_out_cnt++%HAL_ADC_RAW_DIVIDER == 0)
761  {
762  ESP_LOGD(LOG_TAG,"mouse x/y %d/%d; ",tempX,tempY);
763  }
764  #endif
765 
766  if(tempX != 0 && tempY != 0)
767  {
768  command.cmd[0] = 0x01;
769  command.cmd[1] = tempX;
770  command.cmd[2] = tempY;
771 
772  accumXpos -= tempX;
773  accumYpos -= tempY;
774

922 void halAdcCalibrate(void)
923 {
924  //check for initialized mutex
925  if(adcSem == NULL) return;
926 
927  //if we are updating now, do not calibrate.
928  if((xEventGroupGetBits(systemStatus) & SYSTEM_STABLECONFIG) == 0) return;
929 
930  //get mutex
931  if(xSemaphoreTake(adcSem, (TickType_t) 20))
932  {
933  uint8_t retry = 0;
934  do {
935  TONE(TONE_CALIB_FREQ,TONE_CALIB_DURATION);
936  vTaskDelay(100/portTICK_PERIOD_MS);
937 
938  if((xTaskGetTickCount() - adcCalibLast) < (HAL_ADC_CALIB_LOCKTIME / portTICK_PERIOD_MS))
939  {
940  ESP_LOGI(LOG_TAG,"Calibration lock time not passed yet");
941  //give mutex to enable tasks again
942  xSemaphoreGive(adcSem);
943  return;
944  }
945  ESP_LOGI(LOG_TAG,"Starting calibration, offsets: %d/%d",offsetx,offsety);
946  adcData_t D;
947 
948  uint32_t up = 0;
949  uint32_t down = 0;
950  uint32_t left = 0;
951  uint32_t right = 0;
952 
953  //save for next iteration
954  adcCalibLast = xTaskGetTickCount();
955 
956  //read values itself & accumulate (8sensor readings)
957  uint8_t i = 0;
958  do{
959  //use read data for acquiring offset correction data
960  if(halAdcReadData(&D) != 0) continue;
961  i++;
962 
963  //accumulate data
964  up+= D.up;
965  left+= D.left;
966  right+= D.right;
967  down+= D.down;
968 
969  //wait 2 ticks
970  vTaskDelay(2);
971  } while(i<8);
972 
973  //divide by 8 readings
974  up = up / 8;
975  down = down / 8;
976  left = left / 8;
977  right = right / 8;
978  //set as offset values
979  offsetx = left - right;
980  offsety = up - down;
981  //increase retry count
982  retry++;
983  } while(((abs(offsetx) > 1000) || (abs(offsety) > 1000)) && (retry < 10));
984 
985  if(retry < 10)
986  {
987  ESP_LOGI(LOG_TAG,"Finished calibration, offsets: %d/%d",offsetx,offsety);
988  } else {
989  ESP_LOGE(LOG_TAG,"Cannot calibrate, sensor defect!");
990  while(1)
991  {
992  TONE(TONE_CALIB_FREQ,TONE_CALIB_DURATION);
993  vTaskDelay(1000/portTICK_PERIOD_MS);
994  }
995  }
996  //give mutex to enable tasks again
997  xSemaphoreGive(adcSem);
998  } else {
999  ESP_LOGE(LOG_TAG,"Cannot calibrate, no mutex");
1000  }
1001 
1002  return;
1003 }
