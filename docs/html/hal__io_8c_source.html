67 static void gpio_isr_handler(void* arg)
68 {
69  uint32_t pin = (uint32_t) arg;
70  uint8_t vb = 0;
71  BaseType_t xResult, xHigherPriorityTaskWoken = pdFALSE;
72 
73  switch(pin)
74  {
75  case HAL_IO_PIN_BUTTON_EXT1: vb = VB_EXTERNAL1; break;
76  case HAL_IO_PIN_BUTTON_EXT2: vb = VB_EXTERNAL2; break;
77  case HAL_IO_PIN_BUTTON_INT1: vb = VB_INTERNAL1; break;
78  case HAL_IO_PIN_BUTTON_INT2: vb = VB_INTERNAL2; break;
79  default: return;
80  }
81 
82  if(gpio_get_level(pin) == 0)
83  {
84  //set press flag
85  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)),&xHigherPriorityTaskWoken);
86  //clear release flag
87  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)));
88  } else {
89  //set release flag
90  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)),&xHigherPriorityTaskWoken);
91  //clear press flag
92  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)));
93  }
94  if(xResult == pdPASS) portYIELD_FROM_ISR();
95 }

104 void halIOLEDTask(void * param)
105 {
106  uint32_t recv = 0;
107  uint32_t duty = 0;
108  uint32_t fade = 0;
109  if(halIOLEDQueue == NULL)
110  {
111  ESP_LOGW(LOG_TAG, "halIOLEDQueue not initialised");
112  while(halIOLEDQueue == NULL) vTaskDelay(1000/portTICK_PERIOD_MS);
113  }
114  while(1)
115  {
116  //wait for updates
117  if(xQueueReceive(halIOLEDQueue,&recv,10000))
118  {
119  //updates received, sending to ledc driver
120 
121  //get fading time, unit is 10ms
122  fade = ((recv & 0xFF000000) >> 24) * 10;
123 
124  //set fade with time (target duty and fading time)
125 
126  //1.) RED: map to 10bit and set to fading unit
127  duty = (recv & 0x000000FF) * 2 * 2;
128  ledc_set_fade_with_time(LEDC_HIGH_SPEED_MODE,LEDC_CHANNEL_0, \
129  duty, fade);
130 
131  //2.) GREEN: map to 10bit and set to fading unit
132  duty = ((recv & 0x0000FF00) >> 8) * 2 * 2;
133  ledc_set_fade_with_time(LEDC_HIGH_SPEED_MODE,LEDC_CHANNEL_1, \
134  duty, fade);
135 
136  //3.) BLUE: map to 10bit and set to fading unit
137  duty = ((recv & 0x00FF0000) >> 16) * 2 * 2;
138  ledc_set_fade_with_time(LEDC_HIGH_SPEED_MODE,LEDC_CHANNEL_2, \
139  duty, fade);
140 
141  //start fading for RGB
142  ledc_fade_start(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_0, LEDC_FADE_NO_WAIT);
143  ledc_fade_start(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_1, LEDC_FADE_NO_WAIT);
144  ledc_fade_start(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_2, LEDC_FADE_NO_WAIT);
145  }
146  }
147 }
