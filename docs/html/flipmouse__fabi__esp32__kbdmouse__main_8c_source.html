80 void switch_radio(void)
81 {
82  switch(radio)
83  {
84  case BLE:
85  ESP_LOGI(LOG_TAG,"Switching from BLE to BLE_PAIRING");
86  halBLEEnDisable(2);
87  radio = BLE_PAIRING;
88  LED(127,255,0,0);
89  break;
90  case BLE_PAIRING:
91  ESP_LOGI(LOG_TAG,"Switching from BLE_PAIRING to WIFI");
92  halBLEEnDisable(0);
93  taskWebGUIEnDisable(1);
94  radio = WIFI;
95  LED(0,127,255,0);
96  break;
97  case WIFI:
98  ESP_LOGI(LOG_TAG,"Switching from WIFI to BLE");
99  taskWebGUIEnDisable(0);
100  halBLEEnDisable(1);
101  radio = BLE;
102  LED(255,0,127,0);
103  break;
104  case UNINITIALIZED:
105  default:
106  ESP_LOGE(LOG_TAG,"Error, radio is in mode uninitialized. Trying with BLE...");
107  taskWebGUIEnDisable(0);
108  halBLEEnDisable(1);
109  radio = BLE;
110  break;
111  }
112 }

152  ESP_LOGD(LOG_TAG,"created new debouncer task");
153  } else {
154  ESP_LOGE(LOG_TAG,"error creatin new debouncer task");
155  }
156 
157  //start adc continous task
158  if(halAdcInit(NULL) == ESP_OK)
159  {
160  ESP_LOGD(LOG_TAG,"initialized halAdcInit");

177  }

185  }
186 
187  //initialize serial communication task

191  ESP_LOGD(LOG_TAG,"initialized halSerial");

201  ESP_LOGE(LOG_TAG,"error initializing taskCommands");
202  }
203 
204  //initialize web framework
205  if(taskWebGUIInit() == ESP_OK)
206  {
207  ESP_LOGD(LOG_TAG,"initialized webserver/DNS server/webgui");
208  } else {
209  ESP_LOGE(LOG_TAG,"error initializing webserver/DNS server/webgui");

211  //start wifi

213  //disable wifi & enable bluetooth

215  taskWebGUIEnDisable(0);
216  halBLEEnDisable(1);
217 
218  //calibrate directly after start-up
219  halAdcCalibrate();
220 
221  //delete this task after initializing.
222  ESP_LOGI(LOG_TAG,"Finished intializing, deleting app_main");
223  vTaskDelete(NULL);
224 }
