76 static void handler_hid(void *event_handler_arg, esp_event_base_t event_base, int32_t event_id, void *event_data)
77 {
78  //if we don't have a stable config, simply return...
79  if((xEventGroupGetBits(systemStatus) & SYSTEM_STABLECONFIG) == 0) return;
80  //still commands to be processed, shouldn't continue
81  if((xEventGroupGetBits(systemStatus) & SYSTEM_EMPTY_CMD_QUEUE) == 0) return;
82 
83  //use the mutex to ensure a valid chained list.
84  if(xSemaphoreTake(hidCmdSem,4) != pdTRUE)
85  {
86  ESP_LOGW(LOG_TAG,"HID mutex not free for handler");
87  return;
88  }
89  //if command chain is empty, we cannot do anything.
90  if(cmd_chain == NULL)
91  {
92  xSemaphoreGive(hidCmdSem);
93  return;
94  }
95 
96  uint8_t vb = 0;
97  uint32_t count = 0;
98  switch(event_id)
99  {
100  case VB_PRESS_EVENT: vb |= 0x80; //set uppermost bit for press event.
101  case VB_RELEASE_EVENT:
102  break;
103  default: //might be another type of event, we don't care of.
104  return;
105  }
106 
107  //we need an VB number as event data.
108  if(event_data == 0)
109  {
110  ESP_LOGE(LOG_TAG,"Empty event data, cannot proceed!");
111  return;
112  }
113 
114  vb |= (*((uint32_t*) event_data)) & 0x7F;
115  //begin with head of chain
116  hid_cmd_t *current = cmd_chain;
117  hid_cmd_t *firsttriggered = NULL;
118  //iterate through all available hid cmds
119  while(current != NULL)
120  {
121  //send HID command(s), if VB matches
122  //this way, we can do more button presses on one VB (e.g. AT KW, AT KP KEY_SHIFT KEY_A)
123  if(current->vb == vb)
124  {
125  count++;
126  //save the first triggered action for log output
127  if(firsttriggered ==NULL) firsttriggered = current;
128  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
129  { xQueueSend(hid_usb,current,2); }
130  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
131  { xQueueSend(hid_ble,current,2); }
132  }
133  current = current->next;
134  }
135  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
136  if(count == 0) ESP_LOGD(LOG_TAG,"Sent %d cmds for VB %d", count, vb & 0x7F);
137  #endif
138  if(count != 0) ESP_LOGI(LOG_TAG,"Sent %d cmds for VB %d: 0x%02X:0x%02X:0x%02X", \
139  count, vb & 0x7F,firsttriggered->cmd[0],firsttriggered->cmd[1],firsttriggered->cmd[2]);
140  xSemaphoreGive(hidCmdSem);
141 }

147 esp_err_t handler_hid_init(void)
148 {
149  //init HID mutex
150  if(hidCmdSem != NULL) vSemaphoreDelete(hidCmdSem);
151  hidCmdSem = xSemaphoreCreateMutex();
152  if(hidCmdSem == NULL)
153  {
154  ESP_LOGE(LOG_TAG,"Cannot create mutex, exiting!");
155  return ESP_FAIL;
156  }
157  xSemaphoreGive(hidCmdSem);
158  //set log level to given log level
159  esp_log_level_set(LOG_TAG,LOG_LEVEL_HID);

171 esp_err_t handler_hid_delCmd(uint8_t vb)
172 {
173  //existing chain, add to end
174  hid_cmd_t *current = cmd_chain;
175  hid_cmd_t *prev = NULL;
176  uint count = 0;
177 
178  //do as long as we don't have a null pointer
179  while(current != NULL)
180  {
181  //if the VB number matches (discarding press/release flag)
182  if((current->vb & 0x7F) == (vb & 0x7F))
183  {
184  //set pointer from previous element to next one
185  //but only if we are not at the head (no previous element)
186  if(prev != NULL) prev->next = current->next;
187  //if no previous element -> replace
188  else cmd_chain = current->next;
189  //free an AT string
190  if(current->atoriginal != NULL) free(current->atoriginal);
191  //free this element
192  free(current);
193 
194  //just begin at the front again (easiest way if we removed the head)
195  current = cmd_chain;
196  prev = NULL;
197  if((vb & 0x7F) <= 63) vb_active &= ~(1<<(vb & 0x7F)); //delete active flag
198  count++;
199  } else { //does not match
200  //set pointers to next element
201  prev = current;
202  current = current->next;
203  }
204  }
205  if(count != 0) return ESP_OK;
206  else return ESP_FAIL;
207 }

220 esp_err_t handler_hid_addCmd(hid_cmd_t *newCmd, uint8_t replace)
221 {
222  //sanitizing...
223  if(newCmd == NULL)
224  {
225  ESP_LOGE(LOG_TAG,"newCmd is NULL");
226  return ESP_FAIL;
227  }
228  if(hidCmdSem == NULL)
229  {
230  ESP_LOGE(LOG_TAG,"hidCmdSem is NULL");
231  return ESP_FAIL;
232  }
233  if((newCmd->vb & 0x7F) >= VB_MAX)
234  {
235  ESP_LOGE(LOG_TAG,"newCmd->vb out of range");
236  return ESP_FAIL;
237  }
238 
239  //take mutex for modifying
240  if(xSemaphoreTake(hidCmdSem,50) != pdTRUE)
241  {
242  ESP_LOGE(LOG_TAG,"HID mutex not free for adding");
243  return ESP_FAIL;
244  }
245 
246  //set pointer of next element to NULL, so we have a defined
247  //invalid pointer to check for.
248  newCmd->next = NULL;
249 
250  //existing chain, add to end
251  hid_cmd_t *current = cmd_chain;
252  int count = 0;
253 
254  //debugging...
255  #if LOG_LEVEL_HID >= ESP_LOG_DEBUG
256  while(current!=NULL)
257  {
258  count++;
259  ESP_LOGD(LOG_TAG,"%d:%2d:0x%X",count,current->vb,(uint32_t)current);
260  current = current->next;
261  }
262  current = cmd_chain;
263  count = 0;
264  #endif
265 
266  //if set, remove any previously set commands.
267  if(replace) handler_hid_delCmd(newCmd->vb);
268 
269  //allocate new command
270  current = cmd_chain;
271  hid_cmd_t *new;
272  new = malloc(sizeof(*new));
273 
274  if(new != NULL)
275  {
276  memcpy(new, newCmd, sizeof(*new));
277  //save pointer of new config to end of chain
278  new->next = NULL;
279  //use as head if the head was not here before.
280  if(cmd_chain == NULL) {
281  cmd_chain = new;
282  } else {
283  //otherwise append to the tail.
284  while(current->next != NULL)
285  {
286  #if LOG_LEVEL_HID >= ESP_LOG_DEBUG
287  ESP_LOGD(LOG_TAG,"Nr %d, @0x%8X",count,(uint32_t)current);
288  #endif
289  count++;
290  current = current->next;
291  }
292  current->next = new;
293  }
294  if((new->vb & 0x7F) <= 63) vb_active |= (1<<(new->vb & 0x7F)); //set active flag
295  count++;
296  #if LOG_LEVEL_HID >= ESP_LOG_DEBUG
297  ESP_LOGI(LOG_TAG,"Added new cmd nr %d, new: 0x%8X, prev: 0x%8X",count,(uint32_t)new,(uint32_t)current);
298  #endif
299 
300  current = new;
301  xSemaphoreGive(hidCmdSem);
302  return ESP_OK;
303  } else {
304  ESP_LOGE(LOG_TAG,"Cannot allocate memory for new HID cmd!");
305  xSemaphoreGive(hidCmdSem);
306  return ESP_FAIL;
307  }
308 
309  //should not be here, but return ok anyway
310  xSemaphoreGive(hidCmdSem);
311  return ESP_OK;
312 }

329 esp_err_t handler_hid_setCmdChain(hid_cmd_t *chain)
330 {
331  esp_err_t ret = ESP_OK;
332 
333  //check if chain is currently active,
334  //if yes clear it before setting a new one
335  if(cmd_chain != NULL)
336  {

340  ESP_LOGE(LOG_TAG,"Cannot clear old chain");
341  return ret;
342  }
343  }
344 
345  //enter critical section for setting new cmd chain
346  if(hidCmdSem == NULL)
347  {
348  ESP_LOGE(LOG_TAG,"hidCmdSem is NULL");
349  return ESP_FAIL;
350  }
351  //take mutex for modifying
352  if(xSemaphoreTake(hidCmdSem,50) != pdTRUE)
353  {
354  ESP_LOGE(LOG_TAG,"cannot enter critical section");
355  return ESP_FAIL;
356  }
357  //save the pointer
358  cmd_chain = chain;
359 
360  //release mutex
361  xSemaphoreGive(hidCmdSem);
362  return ESP_OK;

371 esp_err_t handler_hid_clearCmds(void)
372 {
373  if(cmd_chain == NULL)
374  {
375  ESP_LOGW(LOG_TAG,"HID cmds already empty");
376  return ESP_FAIL;
377  }
378  if(hidCmdSem == NULL)
379  {
380  ESP_LOGE(LOG_TAG,"hidCmdSem is NULL");
381  return ESP_FAIL;
382  }
383 
384  //take mutex for modifying
385  if(xSemaphoreTake(hidCmdSem,50) != pdTRUE)
386  {
387  ESP_LOGE(LOG_TAG,"HID mutex not free for clearing");
388  return ESP_FAIL;
389  }
390 
391  //pointers for next and current command
392  hid_cmd_t *next = NULL;
393  hid_cmd_t *current = cmd_chain;
394  int count = 0;
395 
396  do {
397  //load next block
398  next = current->next;
399  //if set, free the original AT command
400  if(current->atoriginal != NULL) free(current->atoriginal);
401  current->atoriginal = NULL;
402  //free the current one
403  free(current);
404  //count for statistics
405  count++;
406  //previous next is current for next while iteration
407  current = next;
408  //break the loop if current is NULL (we reached end of chain)
409  } while(current != NULL);
410 
411  #if LOG_LEVEL_HID >= ESP_LOG_INFO
412  ESP_LOGI(LOG_TAG,"Cleared %d HID cmds",count);
413  #endif
414 
415  cmd_chain = NULL;
416  vb_active = 0;
417  //release mutex
418  xSemaphoreGive(hidCmdSem);
419  return ESP_OK;
420 }

430 esp_err_t handler_hid_getAT(char* output, uint8_t vb)
431 {
432  if(hidCmdSem == NULL)
433  {
434  ESP_LOGE(LOG_TAG,"hidCmdSem is NULL");
435  return ESP_FAIL;
436  }
437 
438  if(cmd_chain == NULL)
439  {
440  ESP_LOGE(LOG_TAG,"Chain empty!");
441  return ESP_FAIL;
442  }
443  //take mutex for reading
444  if(xSemaphoreTake(hidCmdSem,50) != pdTRUE)
445  {
446  ESP_LOGE(LOG_TAG,"HID mutex not free for getting");
447  return ESP_FAIL;
448  }
449 
450  //pointers for current command
451  hid_cmd_t *current = cmd_chain;
452 
453  //iterate through all available hid cmds
454  while(current != NULL)
455  {
456  //is this the requested button?
457  if((current->vb & 0x7F) == vb)
458  {
459  //check if we found an AT string
460  if(current->atoriginal != NULL)
461  {
462  strncpy(output,current->atoriginal,ATCMD_LENGTH);
463  #if LOG_LEVEL_HID >= ESP_LOG_INFO
464  ESP_LOGI(LOG_TAG,"BM%02d: %s",vb,output);
465  #endif
466  xSemaphoreGive(hidCmdSem);
467  return ESP_OK;
468  }
469  }
470  current = current->next;
471  }
472  ESP_LOGD(LOG_TAG,"No AT command found");
473  xSemaphoreGive(hidCmdSem);
474  return ESP_FAIL;
475 }

486 bool handler_hid_active(uint8_t vb)
487 {
488  //we check for VB_MAX (firmware specfic) and 63 (size of vb_active)
489  if(vb >= VB_MAX || vb >= 63)
