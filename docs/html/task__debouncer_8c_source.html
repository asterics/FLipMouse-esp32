139 int cancelTimer(uint32_t virtualButton, uint8_t stop)
140 {
141  //if we have VB_MAX here, cancel all of them
142  if(virtualButton >= VB_MAX)
143  {
144  for(int i = 0; i<VB_MAX; i++)
145  {
146  if(xTimers[i].handle != NULL)
147  {
148  if(stop != 0) esp_timer_stop(xTimers[i].handle);
149  esp_timer_delete(xTimers[i].handle);
150  xTimers[i].handle = NULL;
151  }
152  xTimers[i].dir = TIMER_IDLE;
153  }
154  ESP_LOGD(LOG_TAG,"Canceled ALL timers");
155  return -1;
156  }
157 
158  //else: cancel only requested timer
159  if(xTimers[virtualButton].handle != NULL)
160  {
161  if(stop != 0)
162  {
163  if(esp_timer_stop(xTimers[virtualButton].handle) != ESP_OK)
164  { ESP_LOGW(LOG_TAG,"Error stopping timer %d",virtualButton); }
165  }
166  if(esp_timer_delete(xTimers[virtualButton].handle) != ESP_OK)
167  { ESP_LOGW(LOG_TAG,"Error deleting timer %d",virtualButton); }
168  xTimers[virtualButton].handle = NULL; //set handle to NULL, so we remember a stopped timer
169  xTimers[virtualButton].dir = TIMER_IDLE;
170  return virtualButton;
171  } else {
172  ESP_LOGD(LOG_TAG,"Cannot cancel, no timer");
173  //no timer found...
174  return -1;
175  }
176 }

218 void debouncerCallback(void *arg) {
219  //get own ID (virtual button)
220  debouncer_cfg_t *debcfg = (debouncer_cfg_t *)arg;
221 
222  if(debcfg == NULL)
223  {
224  ESP_LOGE(LOG_TAG,"Illegal argument");
225  return;
226  }
227 
228  uint32_t virtualButton = debcfg->vb;
229  generalConfig_t *cfg = configGetCurrent();
230  uint16_t deadtime = 0;
231 
232  if(cfg == NULL)
233  {
234  ESP_LOGE(LOG_TAG,"Cannot do deadtime for VB %d, global config is NULL!",virtualButton);
235  deadtime = 0;
236  } else {
237  //check if an individual deadtime is set
238  if(cfg->debounce_idle_vb[virtualButton] != 0) deadtime = cfg->debounce_idle_vb[virtualButton];
239  //check if we have to use global deadtime
240  if(deadtime == 0 && cfg->debounce_idle != 0) deadtime = cfg->debounce_idle;
241  }
242 
243  switch(debcfg->dir)
244  {
245  case TIMER_IDLE:
246  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but timer should be idle %d",virtualButton);
247  return;
248  case TIMER_PRESS:
249  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for press VB%d",virtualButton);
250  //map in to out and clear from in...
251  if(esp_event_post(VB_EVENT,VB_PRESS_EVENT,(void*)&virtualButton,sizeof(uint32_t),0) != ESP_OK)
252  {
253  ESP_LOGW(LOG_TAG,"Cannot post event!");
254  }
255  //stop timer
256  if(cancelTimer(virtualButton,0) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer (CB)!");
257  //send feedback to host, if enabled
258  sendButtonLearn(virtualButton,VB_PRESS_EVENT,cfg);
259  //create tones, according to issued VB
261  switch(virtualButton & 0x7F)
262  {
263  //tones for sip/puff
264  case VB_SIP: TONE(TONE_SIP_FREQ,TONE_SIP_DURATION); break;
265  case VB_PUFF: TONE(TONE_PUFF_FREQ,TONE_PUFF_DURATION); break;
266  //tones for StrongPuff + XXX
267  case VB_STRONGPUFF_UP:
268  case VB_STRONGPUFF_DOWN:
269  case VB_STRONGPUFF_LEFT:
270  case VB_STRONGPUFF_RIGHT: TONE(TONE_STRONGPUFF_EXIT_FREQ,TONE_STRONGPUFF_EXIT_DURATION); break;
271  //tones for StrongSip + XXX
272  case VB_STRONGSIP_UP:
273  case VB_STRONGSIP_DOWN:
274  case VB_STRONGSIP_LEFT:
275  case VB_STRONGSIP_RIGHT: TONE(TONE_STRONGSIP_EXIT_FREQ,TONE_STRONGSIP_EXIT_DURATION); break;
276  default: break;
277  }
278  break;
279  case TIMER_RELEASE:
280  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for release VB%d",virtualButton);
281  //map in to out and clear from in...
282  if(esp_event_post(VB_EVENT,VB_RELEASE_EVENT,(void*)&virtualButton,sizeof(uint32_t),0) != ESP_OK)
283  {
284  ESP_LOGW(LOG_TAG,"Cannot post event!");
285  }
286  //stop timer
287  if(cancelTimer(virtualButton,0) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer (CB)!");
288  //send feedback to host, if enabled
289  sendButtonLearn(virtualButton,VB_RELEASE_EVENT,cfg);
290  break;
291  case TIMER_DEADTIME:
292  ESP_LOGD(LOG_TAG,"Deadtime finished, ready");
293  if(cancelTimer(virtualButton,0) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer (CB)!");
294  return;
295  default: return;
296  }
297  //if necessary, start timer again for deadtime.
298  if(deadtime != 0)
299  {
300  if(cancelTimer(virtualButton,0) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer (CB)!");
301  debcfg->dir = TIMER_DEADTIME;
302  if(startTimer(debcfg,deadtime) != ESP_OK)
303  {
304  ESP_LOGE(LOG_TAG,"Cannot start timer... (CB)");
305  } else {
306  ESP_LOGD(LOG_TAG,"Deadtime start for VB %d",virtualButton);
307  }
308  }
309 }

315 esp_err_t startTimer(debouncer_cfg_t *cfg, uint16_t debounceTime)
316 {
317  //check if there is a valid VB number (used as index)
318  if(cfg->vb >= VB_MAX) return ESP_FAIL;
319 
320  //if there is a timer running, cancel it before.
321  if(isDebouncerActive(cfg->vb) != TIMER_IDLE) cancelTimer(cfg->vb,1);
322 
323  //create the timer
324  esp_timer_create_args_t args;
325  xTimers[cfg->vb].dir = cfg->dir;
326  args.callback = debouncerCallback; //always the same callback
327  args.arg = (void *)&xTimers[cfg->vb]; //we assign a pointer to this array member
328  args.dispatch_method = ESP_TIMER_TASK; //no other option possible
329  esp_err_t ret = esp_timer_create(&args,&xTimers[cfg->vb].handle);
330 
331  if(ret != ESP_OK)
332  {
333  ESP_LOGE(LOG_TAG,"Cannot create timer, ret: %d",ret);
334  return ESP_FAIL;
335  }
336 
337  //now start this timer (only one-shot)
338  //we need to multiply the given time (in [ms]) to have the
339  //necessary [us] parameter.
340  ret = esp_timer_start_once(xTimers[cfg->vb].handle, debounceTime * 1000);
341 
342  if(ret != ESP_OK)
343  {
344  ESP_LOGE(LOG_TAG,"Cannot start timer, ret: %d",ret);
345  return ESP_FAIL;
346  }
347 
348  //yeah, everything fine...
349  return ESP_OK;
350 }

366 void task_debouncer(void *param)
367 {
368  generalConfig_t *cfg = configGetCurrent();
369  raw_action_t evt;
370  debouncer_cfg_t debcfg;
371  debcfg.handle = NULL;
372  uint16_t time = 0;
373  esp_log_level_set(LOG_TAG,LOG_LEVEL_DEBOUNCE);
374 
375  //test if eventgroup is created
376  while(debouncer_in == NULL)
377  {
378  ESP_LOGE(LOG_TAG,"Eventgroup uninitialized, retry in 1s");
379  vTaskDelay(1000/portTICK_PERIOD_MS);
380  }
381  //clear the xTimers array
382  for(int i = 0; i<VB_MAX; i++)
383  {
384  xTimers[i].dir = TIMER_IDLE;
385  xTimers[i].handle = NULL;
386  xTimers[i].vb = i;
387  }
388 
389  //wait until config is valid
390  while(cfg == NULL)
391  {
392  ESP_LOGE(LOG_TAG,"Global config uninitialized, retry in 1s");
393  vTaskDelay(1000/portTICK_PERIOD_MS);
394  cfg = configGetCurrent();
395  }
396 
397  ESP_LOGI(LOG_TAG,"Debouncer started");
398 
399  while(1)
400  {
401  //if config updates are running, cancel all timers and wait for stable config
402  if((xEventGroupGetBits(systemStatus) & SYSTEM_STABLECONFIG) == 0)
403  {
404  //cancel all timers
405  cancelTimer(VB_MAX,1);
406  //clear all VB events
407  xQueueReset(debouncer_in);
408  //wait 5 ticks to check again
409  //If not set in time, wait again
410  if((xEventGroupWaitBits(systemStatus,SYSTEM_STABLECONFIG, \
411  pdFALSE,pdFALSE,5) & SYSTEM_STABLECONFIG) == 0)
412  {
413  ESP_LOGD(LOG_TAG,"Waiting for config");
414  continue;
415  }
416  }
417 
418  if(xQueueReceive(debouncer_in,&evt,portMAX_DELAY) == pdTRUE)
419  {
420  if(evt.vb >= VB_MAX)
421  {
422  ESP_LOGE(LOG_TAG,"VB out of range!");
423  continue;
424  }
425  //if timer is not running, start one with the corresponding
426  //edge and set xTimerDirection.
427  if(isDebouncerActive(evt.vb) == TIMER_IDLE)
428  {
429  //check which time to use (either VB, global value or default)
430  uint8_t t_type = TIMER_IDLE;
431 
432  switch(evt.type)
433  {
434  case VB_PRESS_EVENT:
435  t_type = TIMER_PRESS;
436  //is a VB value set in config?
437  if(cfg->debounce_press_vb[evt.vb] != 0) time = cfg->debounce_press_vb[evt.vb];
438  //is a global value set in config?
439  if(time == 0 && cfg->debounce_press != 0) time = cfg->debounce_press;
440  //no? just use the default value
441  if(time == 0) time = DEBOUNCETIME_MS;
442  break;
443  case VB_RELEASE_EVENT:
444  t_type = TIMER_RELEASE;
445  //is a VB value set in config?
446  if(cfg->debounce_release_vb[evt.vb] != 0) time = cfg->debounce_release_vb[evt.vb];
447  //is a global value set in config?
448  if(time == 0 && cfg->debounce_release != 0) time = cfg->debounce_release;
449  //no? just use the default value
450  if(time == 0) time = DEBOUNCETIME_MS;
451  break;
452  default: break;
453  }
454  if(time > DEBOUNCETIME_MIN_MS)
455  {
456  debcfg.vb = evt.vb;
457  debcfg.dir = t_type;
458  if(startTimer(&debcfg,time) != ESP_OK) ESP_LOGE(LOG_TAG,"Cannot start timer...");
459  else {
460  ESP_LOGD(LOG_TAG,"Debounce started for VB%d / T: %d",evt.vb,t_type);
461  }
462  continue;
463  //note: currently, following branch is unused, but maybe we need a directly mapped VB.
464  } else {
465  //if no debounce time is used
466  ESP_LOGD(LOG_TAG,"Map VB%d / T: %d",evt.vb,evt.type);
467  if(esp_event_post(VB_EVENT,evt.type,(void*)&evt.vb,sizeof(evt.vb),0) != ESP_OK)
468  {
469  ESP_LOGW(LOG_TAG,"Cannot post event!");
470  }
471  continue;
472  }
473  } else {
474  //if timer is running, check if this flag requests the
475  //opposite direction OR the same direction is cleared
476  //if yes -> stop & delete this timer
477  switch(xTimers[evt.vb].dir)
478  {
479  case TIMER_PRESS:
480  //if release is wanted, but press timer is running
481  //->cancel timer
482  if(evt.type == VB_RELEASE_EVENT)
483  {
484  ESP_LOGD(LOG_TAG,"Press canceled for VB%d, sending release",evt.vb);
485  if(cancelTimer(evt.vb,1) == -1) //stop current press debouncer
486  { ESP_LOGE(LOG_TAG,"Cannot cancel press timer!"); }
489  if(esp_event_post(VB_EVENT,evt.type,(void*)&evt.vb,sizeof(evt.vb),0) != ESP_OK)
490  {
491  ESP_LOGW(LOG_TAG,"Cannot post event!");
492  }
493  }
494  break;
495  case TIMER_RELEASE:
496  //if press is wanted, but release timer is running
497  //->cancel timer
501  time = 0;
502  //is a VB value set in config?
503  if(cfg->debounce_release_vb[evt.vb] != 0) time = cfg->debounce_release_vb[evt.vb];
504  //is a global value set in config?
505  if(time == 0 && cfg->debounce_release != 0) time = cfg->debounce_release;
506  if(evt.type == VB_PRESS_EVENT && time != 0)
507  {
508  ESP_LOGD(LOG_TAG,"Release canceled for VB%d",evt.vb);
509  if(cancelTimer(evt.vb,1) == -1) //stop current press debouncer
510  { ESP_LOGE(LOG_TAG,"Cannot cancel release timer!"); }
511  }
512  break;
513  case TIMER_IDLE:
514  ESP_LOGE(LOG_TAG,"Timer is idle but a valid ID?");
515  break;
516  case TIMER_DEADTIME:
517  ESP_LOGD(LOG_TAG,"Deadtime active, waiting.");
518  break;
519  case TIMER_ERROR:
520  ESP_LOGE(LOG_TAG,"Timer is in error state [%d]",evt.vb);
521  default:
522  ESP_LOGE(LOG_TAG,"Unknown status in xTimers[%d].dir",evt.vb);
523  break;
524  }
525  } /* else -> timerId != TIMER_IDLE */
526  } /* if(xQueueReceive... */
527  } /* while(1) */
528 } /* task_debouncer */
