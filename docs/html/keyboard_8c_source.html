2052 uint8_t parse_for_keycode(uint8_t inputdata, uint8_t locale, uint8_t *keycode_modifier, uint8_t *deadkey_first_keycode)
2053 {

2055  static uint16_t unicode_wchar = 0;
2056  static uint16_t keycode = 0;
2057  static uint8_t deadkey = 0;
2058 
2059  //avoid accessing arrays out of bound
2060  if(locale >= LAYOUT_MAX) return 0;
2061 
2062  if (inputdata < 0x80) {
2063  // single byte encoded, 0x00 to 0x7F
2064  utf8_state = 0;
2065  keycode = unicode_to_keycode(inputdata, locale);
2066  deadkey = deadkey_to_keycode(keycode,locale);
2067  *keycode_modifier = keycode_to_modifier(keycode,locale);
2068  if(deadkey) *deadkey_first_keycode = keycode_to_key(deadkey);
2069  else *deadkey_first_keycode = 0;
2070  return keycode_to_key(keycode & keycodes_masks[locale][3]);
2071  } else if (inputdata < 0xC0) {
2072  // 2nd, 3rd or 4th byte, 0x80 to 0xBF
2073  inputdata &= 0x3F;
2074  if (utf8_state == 1) {
2075  utf8_state = 0;
2076  keycode = unicode_to_keycode(unicode_wchar | inputdata, locale);
2077  deadkey = deadkey_to_keycode(keycode,locale);
2078  *keycode_modifier = keycode_to_modifier(keycode,locale);
2079  if(deadkey) *deadkey_first_keycode = keycode_to_key(deadkey);
2080  else *deadkey_first_keycode = 0;
2081  return keycode_to_key(keycode & keycodes_masks[locale][3]);
2082  } else if (utf8_state == 2) {
2083  unicode_wchar |= ((uint16_t)inputdata << 6);
2084  utf8_state = 1;
2085  }
2086  } else if (inputdata < 0xE0) {
2087  // begin 2 byte sequence, 0xC2 to 0xDF
2088  // or illegal 2 byte sequence, 0xC0 to 0xC1
2089  unicode_wchar = (uint16_t)(inputdata & 0x1F) << 6;
2090  utf8_state = 1;
2091  } else if (inputdata < 0xF0) {
2092  // begin 3 byte sequence, 0xE0 to 0xEF
2093  unicode_wchar = (uint16_t)(inputdata & 0x0F) << 12;
2094  utf8_state = 2;
2095  } else {
2096  // begin 4 byte sequence (not supported), 0xF0 to 0xF4
2097  // or illegal, 0xF5 to 0xFF
2098  utf8_state = 255;
2099  }
2100  return 0;
2101 }
