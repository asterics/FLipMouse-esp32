122 //Parses a label into a C-string containing a dotted
123 //Returns pointer to start of next fields in packet
124 static char* labelToStr(char *packet, char *labelPtr, int packetSz, char *res, int resMaxLen) {
125  int i, j, k;
126  char *endPtr=NULL;
127  i=0;
128  do {
129  if ((*labelPtr&0xC0)==0) {
130  j=*labelPtr++; //skip past length
131  //Add separator period if there already is data in res
132  if (i<resMaxLen && i!=0) res[i++]='.';
133  //Copy label to res
134  for (k=0; k<j; k++) {
135  if ((labelPtr-packet)>packetSz) return NULL;
136  if (i<resMaxLen) res[i++]=*labelPtr++;
137  }
138  } else if ((*labelPtr&0xC0)==0xC0) {
139  //Compressed label pointer
140  endPtr=labelPtr+2;
141  int offset=my_ntohs(((uint16_t *)labelPtr))&0x3FFF;
142  //Check if offset points to somewhere outside of the packet
143  if (offset>packetSz) return NULL;
144  labelPtr=&packet[offset];
145  }
146  //check for out-of-bound-ness
147  if ((labelPtr-packet)>packetSz) return NULL;
148  } while (*labelPtr!=0);
149  res[i]=0; //zero-terminate
150  if (endPtr==NULL) endPtr=labelPtr+1;
151  return endPtr;
152 }

155 static char *strToLabel(char *str, char *label, int maxLen) {
156  char *len=label; //ptr to len byte
157  char *p=label+1; //ptr to next label byte to be written
158  while (1) {
159  if (*str=='.' || *str==0) {
160  *len=((p-len)-1); //write len of label bit
161  len=p; //pos of len for next part
162  p++; //data ptr is one past len
163  if (*str==0) break; //done
164  str++;
165  } else {
166  *p++=*str++; //copy byte
167  //if ((p-label)>maxLen) return NULL; //check out of bounds
168  }
169  }
170  *len=0;
171  return p; //ptr to first free byte in resp
172 }

174 //Receive a DNS packet and maybe send a response back
175 static void captdnsRecv(struct sockaddr_in *premote_addr, char *pusrdata, unsigned short length) {
176 
177  char buff[DNS_LEN];
178  char reply[DNS_LEN];
179  int i;
180  char *rend=&reply[length];
181  char *p=pusrdata;
182  DnsHeader *hdr=(DnsHeader*)p;
183  DnsHeader *rhdr=(DnsHeader*)&reply[0];
184  p+=sizeof(DnsHeader);
185  // printf("DNS packet: id 0x%X flags 0x%X rcode 0x%X qcnt %d ancnt %d nscount %d arcount %d len %d\n",
186  // my_ntohs(&hdr->id), hdr->flags, hdr->rcode, my_ntohs(&hdr->qdcount), my_ntohs(&hdr->ancount), my_ntohs(&hdr->nscount), my_ntohs(&hdr->arcount), length);
187  //Some sanity checks:
188  if (length>DNS_LEN) return; //Packet is longer than DNS implementation allows
189  if (length<sizeof(DnsHeader)) return; //Packet is too short
190  if (hdr->ancount || hdr->nscount || hdr->arcount) return; //this is a reply, don't know what to do with it
191  if (hdr->flags&FLAG_TC) return; //truncated, can't use this
192  //Reply is basically the request plus the needed data
193  memcpy(reply, pusrdata, length);
194  rhdr->flags|=FLAG_QR;
195 
196  for (i=0; i<my_ntohs(&hdr->qdcount); i++)
197  {
198  //Grab the labels in the q string
199  p=labelToStr(pusrdata, p, length, buff, sizeof(buff));
200  if (p==NULL) return;
201  DnsQuestionFooter *qf=(DnsQuestionFooter*)p;
202  p+=sizeof(DnsQuestionFooter);
203 
204  if (my_ntohs(&qf->type)==QTYPE_A) {
205  //They want to know the IPv4 address of something.
206  //Build the response.
207 
208  rend=strToLabel(buff, rend, sizeof(reply)-(rend-reply)); //Add the label
209  if (rend==NULL) return;
210  DnsResourceFooter *rf=(DnsResourceFooter *)rend;
211  rend+=sizeof(DnsResourceFooter);
212  setn16(&rf->type, QTYPE_A);
213  setn16(&rf->class, QCLASS_IN);
214  setn32(&rf->ttl, 0);
215  setn16(&rf->rdlength, 4); //IPv4 addr is 4 bytes;
216  //Grab the current IP of the softap interface
217 
218  tcpip_adapter_ip_info_t info;
219  //tcpip_adapter_get_ip_info(TCPIP_ADAPTER_IF_ETH, &info);
220  tcpip_adapter_get_ip_info(TCPIP_ADAPTER_IF_AP, &info);
221  //struct ip_info info;
222  //wifi_get_ip_info(SOFTAP_IF, &info);
223  *rend++=ip4_addr1(&info.ip);
224  *rend++=ip4_addr2(&info.ip);
225  *rend++=ip4_addr3(&info.ip);
226  *rend++=ip4_addr4(&info.ip);
227  setn16(&rhdr->ancount, my_ntohs(&rhdr->ancount)+1);
228  #if (LOG_LEVEL_DNS>=ESP_LOG_DEBUG)
229  ESP_LOGD(LOG_TAG,"DNS: Q (type 0x%X class 0x%X) for %s, resp: %s", my_ntohs(&qf->type), my_ntohs(&qf->class), buff,ip4addr_ntoa(&info.ip));
230  #endif
231 
232  } else if (my_ntohs(&qf->type)==QTYPE_NS) {
233  //Give ns server. Basically can be whatever we want because it'll get resolved to our IP later anyway.
234  rend=strToLabel(buff, rend, sizeof(reply)-(rend-reply)); //Add the label
235  DnsResourceFooter *rf=(DnsResourceFooter *)rend;
236  rend+=sizeof(DnsResourceFooter);
237  setn16(&rf->type, QTYPE_NS);
238  setn16(&rf->class, QCLASS_IN);
239  setn16(&rf->ttl, 0);
240  setn16(&rf->rdlength, 4);
241  *rend++=2;
242  *rend++='n';
243  *rend++='s';
244  *rend++=0;
245  setn16(&rhdr->ancount, my_ntohs(&rhdr->ancount)+1);
246  //printf("Added NS rec to resp. Resp len is %d\n", (rend-reply));
247  } else if (my_ntohs(&qf->type)==QTYPE_URI) {
248  //Give uri to us
249  rend=strToLabel(buff, rend, sizeof(reply)-(rend-reply)); //Add the label
250  DnsResourceFooter *rf=(DnsResourceFooter *)rend;
251  rend+=sizeof(DnsResourceFooter);
252  DnsUriHdr *uh=(DnsUriHdr *)rend;
253  rend+=sizeof(DnsUriHdr);
254  setn16(&rf->type, QTYPE_URI);
255  setn16(&rf->class, QCLASS_URI);
256  setn16(&rf->ttl, 0);
257  setn16(&rf->rdlength, 4+16);
258  setn16(&uh->prio, 10);
259  setn16(&uh->weight, 1);
260  memcpy(rend, "http://esp.nonet", 16);
261  rend+=16;
262  setn16(&rhdr->ancount, my_ntohs(&rhdr->ancount)+1);
263  //printf("Added NS rec to resp. Resp len is %d\n", (rend-reply));
264  }
265  }
266  //Send the response
267  //printf("Send response\n");
268  sendto(sockFd,(uint8_t*)reply, rend-reply, 0, (struct sockaddr *)premote_addr, sizeof(struct sockaddr_in));
269 
270 }

272 static void captdnsTask(void *pvParameters) {
273  struct sockaddr_in server_addr;
274  uint32_t ret;
275  struct sockaddr_in from;
276  socklen_t fromlen;
277  //struct tcpip_adapter_ip_info_t ipconfig;
278  char udp_msg[DNS_LEN];
279 
280  //memset(&ipconfig, 0, sizeof(ipconfig));
281  memset(&server_addr, 0, sizeof(server_addr));
282  server_addr.sin_family = AF_INET;
283  server_addr.sin_addr.s_addr = INADDR_ANY;
284  server_addr.sin_port = htons(53);
285  server_addr.sin_len = sizeof(server_addr);
286 
287  do {
288  sockFd=socket(AF_INET, SOCK_DGRAM, 0);
289  if (sockFd==-1) {
290  ESP_LOGE(LOG_TAG,"failed to create sock!");
291  vTaskDelay(1000/portTICK_RATE_MS);
292  }
293  } while (sockFd==-1);
294 
295  do {
296  ret=bind(sockFd, (struct sockaddr *)&server_addr, sizeof(server_addr));
297  if (ret!=0) {
298  ESP_LOGE(LOG_TAG,"failed to bind sock!");
299  vTaskDelay(1000/portTICK_RATE_MS);
300  }
301  } while (ret!=0);
302 
303  #if (LOG_LEVEL_DNS>=ESP_LOG_INFO)
304  ESP_LOGI(LOG_TAG,"DNS task finished init");
305  #endif
306 
307  while(1) {
308  memset(&from, 0, sizeof(from));
309  fromlen=sizeof(struct sockaddr_in);
310  ret=recvfrom(sockFd, (uint8_t *)udp_msg, DNS_LEN, 0,(struct sockaddr *)&from,(socklen_t *)&fromlen);
311  if (ret>0) captdnsRecv(&from,udp_msg,ret);
312  }
313 
314  close(sockFd);
315  vTaskDelete(NULL);
316 }
