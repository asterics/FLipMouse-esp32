118 static void hidd_event_callback(esp_hidd_cb_event_t event, esp_hidd_cb_param_t *param)
119 {
120  switch(event) {
121  case ESP_HIDD_EVENT_REG_FINISH: {
122  if (param->init_finish.state == ESP_HIDD_INIT_OK) {
123  //esp_bd_addr_t rand_addr = {0x04,0x11,0x11,0x11,0x11,0x05};
124  if(deviceIdentifier == 1)
125  {
126  esp_ble_gap_set_device_name(hid_device_name_fabi);

128  esp_ble_gap_set_device_name(hid_device_name_flipmouse);
129  }
130  esp_ble_gap_config_adv_data(&hidd_adv_data);
131 
132  }
133  break;
134  }
135  case ESP_BAT_EVENT_REG: {
136  break;
137  }
138  case ESP_HIDD_EVENT_DEINIT_FINISH:
139  break;
140  case ESP_HIDD_EVENT_BLE_CONNECT: {
141  hid_conn_id = param->connect.conn_id;
142  sec_conn = true; //TODO: right here?!?
143  LOG_ERROR("%s(), ESP_HIDD_EVENT_BLE_CONNECT", __func__);
144  break;
145  }
146  case ESP_HIDD_EVENT_BLE_DISCONNECT: {
147  sec_conn = false;
148  hid_conn_id = 0;
149  LOG_ERROR("%s(), ESP_HIDD_EVENT_BLE_DISCONNECT", __func__);
150  esp_ble_gap_start_advertising(&hidd_adv_params);
151  break;
152  }
153  default:
154  break;
155  }
156  return;

161  switch (event) {
162  case ESP_GAP_BLE_ADV_DATA_SET_COMPLETE_EVT:

176  break;

181 void halBleTaskKeyboardPress(void *param)
182 {
183  uint16_t rxK;
184  uint8_t keycode;
185  uint8_t keycodesLocal[6] = {0,0,0,0,0,0};
186  static uint8_t unconnectedWarning = 0;
187  generalConfig_t* currentConfig = configGetCurrent();
188 
189  if(keyboard_ble_press == NULL)
190  {
191  ESP_LOGE(LOG_TAG,"queue keyboard_ble_press uninitialized, cannot continue");
192  vTaskDelete(NULL);
193  }
194 
195  while(1)
196  {
197  //wait for a keyboard press command
198  if(xQueueReceive(keyboard_ble_press, &rxK, 2000/portTICK_PERIOD_MS ))
199  {
200  //we cannot send if we are not connected
201  if(!sec_conn) {
202  if(!unconnectedWarning) ESP_LOGW(LOG_TAG,"Not connected, cannot send keyboard");
203  unconnectedWarning = 1;
204  continue;
205  } else unconnectedWarning = 0;
206 
207  //check if this is a unicode/ascii byte or a keycode
208  if((rxK & 0xFF00) == 0)
209  {
210  //single bytes are ascii or unicode bytes, sent to keycode parser
211  keycode = parse_for_keycode((uint8_t) rxK, currentConfig->locale, &keycode_modifier, &keycode_deadkey_first);
212  if(keycode != 0)
213  {
214  //if a deadkey is issued (no release necessary), we send the deadkey before:
215  if(keycode_deadkey_first != 0)
216  {
217  ESP_LOGD(LOG_TAG,"Sending deadkey first: %d",keycode_deadkey_first);
218  keycodesLocal[0] = keycode_deadkey_first;
219  //send to device
220  esp_hidd_send_keyboard_value(hid_conn_id,keycode_modifier,keycodesLocal,sizeof(keycodesLocal));
221  //wait 1 tick to release key
222  vTaskDelay(1); //suspend for 1 tick
223  //TODO: should we release the key here?!? -> TEST
224  keycodesLocal[0] = keycode_deadkey_first;
225  keycodesLocal[0] = 0;
226  //send to device
227  esp_hidd_send_keyboard_value(hid_conn_id,keycode_modifier,keycodesLocal,sizeof(keycodesLocal));
228  }
229  } //we need another byte to calculate keycode...
230 
231  } else {
232  keycode = rxK & 0x00FF;
233  }
234 
235  //keycodes, add directly to the keycode array
236  switch(add_keycode(keycode,keycode_arr))
237  {
238  case 0:
239  esp_hidd_send_keyboard_value(hid_conn_id,keycode_modifier,keycode_arr,sizeof(keycode_arr));
240  ESP_LOGD(LOG_TAG,"Kbd: mod:%d, other keycodes: ",keycode_modifier);
241  esp_log_buffer_hex(LOG_TAG,keycode_arr,6);
242  break;
243  case 1: ESP_LOGW(LOG_TAG,"keycode already in queue"); break;
244  case 2: ESP_LOGE(LOG_TAG,"no space in keycode arr!"); break;
245  default: ESP_LOGE(LOG_TAG,"add_keycode return unknown code..."); break;
246  }
247 
248  keycode = 0;
249  }
250  }
251 }

305 void halBleTaskMouse(void *param)
306 {
307  mouse_command_t rxM;
308  static uint8_t unconnectedWarning = 0;
309  static uint8_t emptySent = 0;
310  mouse_command_t emptyReport = {0,0,0,0};
311 
312  if(mouse_movement_ble == NULL)
313  {
314  ESP_LOGE(LOG_TAG,"queue mouse_movement_ble uninitialized, cannot continue");
315  vTaskDelete(NULL);
316  }
317 
318  while(1)
319  {
320  //wait for a mouse command
321  if(xQueueReceive(mouse_movement_ble, &rxM, 2000/portTICK_PERIOD_MS ))
322  {
323  if(!sec_conn) {
324  if(!unconnectedWarning) ESP_LOGW(LOG_TAG,"Not connected, cannot send mouse");
325  unconnectedWarning = 1;
326  } else {
327  unconnectedWarning = 0;
328  //send report always if not empty
329  if(memcmp(&rxM,&emptyReport,sizeof(mouse_command_t)) != 0)
330  {
331  esp_hidd_send_mouse_value(hid_conn_id,rxM.buttons, \
332  rxM.x, rxM.y,rxM.wheel);
333  ESP_LOGD(LOG_TAG,"Mouse: %d,%d,%d,%d",rxM.buttons, \
334  rxM.x, rxM.y, rxM.wheel);
335  //after one non-empty report, we can resend an empty one
336  emptySent = 0;
337  } else {
338  //if empty, check if it was sent once, if not: send empty
339  if(emptySent == 0)
340  {
341  esp_hidd_send_mouse_value(hid_conn_id,rxM.buttons, \
342  rxM.x, rxM.y,rxM.wheel);
343  ESP_LOGD(LOG_TAG,"Mouse: %d,%d,%d,%d",rxM.buttons, \
344  rxM.x, rxM.y, rxM.wheel);
345  //remember we sent an empty report
346  emptySent = 1;
347  }
348  }
349  }
350  }
351  }

398  ESP_LOGW(LOG_TAG,"Not connected, cannot reset reports");
399  } else {
400  ESP_LOGD(LOG_TAG,"BLE reset reports");
401  //reset mouse

403  //reset keyboard

410  }

426 esp_err_t halBLEInit(uint8_t deviceIdentifier)
427 {
428  generalConfig_t* currentConfig = configGetCurrent();
429  // Initialize NVS.
430  esp_err_t ret = nvs_flash_init();
431  if (ret == ESP_ERR_NVS_NO_FREE_PAGES) {
432  ESP_ERROR_CHECK(nvs_flash_erase());
433  ret = nvs_flash_init();
434  }
435  ESP_ERROR_CHECK( ret );
436  // Read config
437  if(deviceIdentifier > 1)
438  {
439  ESP_LOGE(LOG_TAG,"deviceIdentifier is invalid, using FABI");
440  deviceIdentifier = 1;
441  }
442 
443  if(currentConfig->locale >= LAYOUT_MAX)
444  {
445  ESP_LOGE(LOG_TAG,"locale is invalid: %d, using US_INTERNATIONAL",currentConfig->locale);
446  currentConfig->locale = LAYOUT_US_INTERNATIONAL;
447  }
448 
449  esp_bt_controller_config_t bt_cfg = BT_CONTROLLER_INIT_CONFIG_DEFAULT();
450  ret = esp_bt_controller_init(&bt_cfg);
451  if (ret) {
452  ESP_LOGE(LOG_TAG, "initialize controller failed\n");
453  return ESP_FAIL;
454  }
455 
456  ret = esp_bt_controller_enable(ESP_BT_MODE_BTDM);
457  if (ret) {
458  ESP_LOGE(LOG_TAG, "enable controller failed\n");
459  return ESP_FAIL;
460  }
461 
462  ret = esp_bluedroid_init();
463  if (ret) {
464  ESP_LOGE(LOG_TAG,"init bluedroid failed\n");
465  return ESP_FAIL;
466  }
467 
468  ret = esp_bluedroid_enable();
469  if (ret) {
470  ESP_LOGE(LOG_TAG,"init bluedroid failed\n");
471  return ESP_FAIL;
472  }
473 
474  //load HID country code for locale before initialising HID
475  hidd_set_countrycode(get_hid_country_code(currentConfig->locale));
476 
477  if((ret = esp_hidd_profile_init()) != ESP_OK) {
478  ESP_LOGE(LOG_TAG,"init bluedroid failed\n");
479  return ESP_FAIL;
480  }
481 
483  esp_ble_gap_register_callback(gap_event_handler);
484  esp_hidd_register_callbacks(hidd_event_callback);
485 
486  /* set the security iocap & auth_req & key size & init key response key parameters to the stack*/
487  esp_ble_auth_req_t auth_req = ESP_LE_AUTH_BOND; //bonding with peer device after authentication
488  esp_ble_io_cap_t iocap = ESP_IO_CAP_NONE; //set the IO capability to No output No input
489  uint8_t key_size = 16; //the key size should be 7~16 bytes
490  uint8_t init_key = ESP_BLE_ENC_KEY_MASK | ESP_BLE_ID_KEY_MASK;
491  uint8_t rsp_key = ESP_BLE_ENC_KEY_MASK | ESP_BLE_ID_KEY_MASK;
492  esp_ble_gap_set_security_param(ESP_BLE_SM_AUTHEN_REQ_MODE, &auth_req, sizeof(uint8_t));
493  esp_ble_gap_set_security_param(ESP_BLE_SM_IOCAP_MODE, &iocap, sizeof(uint8_t));
494  esp_ble_gap_set_security_param(ESP_BLE_SM_MAX_KEY_SIZE, &key_size, sizeof(uint8_t));
495  /* If your BLE device act as a Slave, the init_key means you hope which types of key of the master should distribut to you,
496  and the response key means which key you can distribut to the Master;
497  If your BLE device act as a master, the response key means you hope which types of key of the slave should distribut to you,
498  and the init key means which key you can distribut to the slave. */
499  esp_ble_gap_set_security_param(ESP_BLE_SM_SET_INIT_KEY, &init_key, sizeof(uint8_t));
500  esp_ble_gap_set_security_param(ESP_BLE_SM_SET_RSP_KEY, &rsp_key, sizeof(uint8_t));
501 
502  //start tasks, pending for all 4 input queues
503  for(uint8_t i = 0; i<4; i++)
504  {
505  if(xTaskCreate(halbletaskfunctions[i],halbletasknames[i],TASK_HALBLE_STACKSIZE,
506  (void*)NULL,HAL_BLE_TASK_PRIORITY_BASE+i, &halbletasks[i]) == pdPASS)
507  {
508  ESP_LOGD(LOG_TAG,"created hal_ble task %d",i);
509  } else {
510  ESP_LOGE(LOG_TAG,"error creating new debouncer task");
511  }
512  }
513  return ESP_OK;
514 }
