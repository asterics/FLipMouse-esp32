175 void sendHIDCmd(hid_cmd_t *sendCmd, uint8_t vb, uint8_t* atorig, uint8_t replace)
176 {
177  //check if this cmd contains data
178  hid_cmd_t emptyhid; memset(&emptyhid,0,sizeof(hid_cmd_t));
179  if(sendCmd == NULL) return;
180  if(memcmp(sendCmd,&emptyhid,sizeof(hid_cmd_t)) == 0) return;
181 
182  //send it directly, if singleshot is active
183  if(requestVBUpdate == VB_SINGLESHOT)
184  {
185  //post values to mouse queue (USB and/or BLE)
186  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
187  { xQueueSend(hid_usb,sendCmd,0); }
188 
189  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
190  { xQueueSend(hid_ble,sendCmd,0); }
191 
192  if(sendCmd->atoriginal != NULL) free(sendCmd->atoriginal);
193  } else {
194  //update HID command (set VB, add original string)
195  sendCmd->vb = vb;
196  if(atorig != NULL)
197  {
198  sendCmd->atoriginal = strdup((char*)atorig);
199  if(sendCmd->atoriginal == NULL)
200  {
201  ESP_LOGE(LOG_TAG,"Error allocating AT cmd string");
202  }
203  } else {
204  sendCmd->atoriginal = NULL;
205  }
206  //add to HID cmd, remove from VB cmd
207  handler_vb_delCmd(sendCmd->vb);
208  handler_hid_addCmd(sendCmd,replace);
209  }
210 }

429 esp_err_t keyboard_helper_parsekeycode(char t, uint8_t *buf)
430 {
431  hid_cmd_t cmd; //this would be the press or press&release action
432  memset(&cmd,0,sizeof(hid_cmd_t));
433  //if the first HID cmd is sent, this flag is set.
434  uint8_t deleted = 0;
435  char *pch;
436  uint8_t cnt = 0;
437  uint16_t keycode = 0;
438  uint16_t releaseArr[16];
439 
440  //remove trailing \r/\n
441  strip((char*)buf);
442  pch = (char*)&buf[5];
443  while (pch != NULL)
444  {
445  ESP_LOGD(LOG_TAG,"Token: %s",pch+1);
446  //check if token seems to be a KEY_*
447  if(memcmp(pch+1,"KEY_",4) != 0)
448  {
449  if(cnt == 0) ESP_LOGW(LOG_TAG,"Not a valid KEY_* identifier");
450  } else {
451  //parse identifier to keycode
452  keycode = parseIdentifierToKeycode(pch+1);
453 
454  if(keycode != 0)
455  {
456  //if found, save...
457  ESP_LOGD(LOG_TAG,"Keycode: 0x%4X",keycode);
458 
459  //because we have KEY identifiers, no deadkey processing is
460  //done here.
461 
462  //KEY_ * identifiers are either a key or a modifier
463  //determine which type and save accordingly:
464  if(keycode_is_modifier(keycode))
465  {
466  //we are working with a modifier key here
467  cmd.cmd[1] = keycode & 0xFF;
468  switch(t)
469  {
470  case 'H': //KH + KP press on first action
471  case 'P': cmd.cmd[0] = 0x25; break;
472  case 'R': cmd.cmd[0] = 0x26; break; //KR release on first action
473  case 'T': cmd.cmd[0] = 0x27; break; //KT toggle on first action
474  }
475  } else {
476  cmd.cmd[1] = keycode_to_key(keycode);
477  switch(t)
478  {
479  case 'H':
480  case 'P': cmd.cmd[0] = 0x21; break;
481  case 'R': cmd.cmd[0] = 0x22; break;
482  case 'T': cmd.cmd[0] = 0x23; break;
483  }
484  }
485 
486  //send the cmd either directly or save it to the HID task
487  //we do the press event here.
488  if(deleted == 0) sendHIDCmd(&cmd,requestVBUpdate|0x80,buf,1);
489  else sendHIDCmd(&cmd,requestVBUpdate|0x80,NULL,0);
490  deleted = 1;
491  ESP_LOGI(LOG_TAG,"Press action 0x%2X, keycode/modifier: 0x%2X",cmd.cmd[0],cmd.cmd[1]);
492  //save for later release
493  releaseArr[cnt] = keycode;
494  cnt++;
495 
496  if(cnt == 14) //maximum 6 keycodes + 8 modifier bits
497  {
498  sendErrorBack("AT KP/KH/KR parameter too long");
499  return ESP_FAIL;
500  }
501  } else {
502  ESP_LOGW(LOG_TAG,"No keycode found for this token.");
503  }
504  }
505  //split into tokens
506  pch = strpbrk(pch+1, " ");
507  }
508  //check if any key identifiers were found
509  if(cnt == 0)
510  {
511  return ESP_FAIL;
512  }
513  //now we need to add the release actions.
514  //this is done ONLY for AT KP & AT KH:
515  //AT KP releases all keys immediately after pressing them.
516  //This is valid in VB mode and via AT commands on serial interface
517  //AT KH releases all keys on a VB release trigger. If we
518  //got that command via the serial interface, we don't do this (AT KR is needed)
519 
520  if((t=='P') || ((t=='H')&&(requestVBUpdate != VB_SINGLESHOT)))
521  {
522  //this has to be a press action too, but at the end
523  if(t=='P') cmd.vb = requestVBUpdate | 0x80;
524  //we need to have this action on a VB release trigger
525  if(t=='H') cmd.vb = requestVBUpdate;
526 
527  //now either send directly or add to HID task...
528  for(uint8_t i = 0; i<cnt; i++)
529  {
530  //check which command we need (release key or modifier)
531  if(keycode_is_modifier(releaseArr[i]))
532  {
533  cmd.cmd[0] = 0x26; //release the modifier
534  //get key for this one
535  cmd.cmd[1] = releaseArr[i] & 0xFF;
536  } else {
537  cmd.cmd[0] = 0x22; //release the key
538  //get key for this one
539  cmd.cmd[1] = keycode_to_key(releaseArr[i]);
540  }
541 
542  if(deleted == 0) sendHIDCmd(&cmd,requestVBUpdate,buf,1);
543  else sendHIDCmd(&cmd,requestVBUpdate,NULL,0);
544  deleted = 1;
545  ESP_LOGI(LOG_TAG,"Release action 0x%2X, keycode/modifier: 0x%2X",cmd.cmd[0],cmd.cmd[1]);
546  }
547  }
548  return ESP_OK;
549 }
