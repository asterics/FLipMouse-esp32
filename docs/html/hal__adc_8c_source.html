151 void halAdcProcessStrongMode(adcData_t *D)
152 {
153  raw_action_t evt;
154  //on a FABI device, we cannot do this combination with analog values.
155  //maybe it will be done on another firmware part.
156  #ifdef DEVICE_FABI
157  D->strongmode = STRONG_NORMAL;
158  return;
159  #endif
160 
161  #ifdef DEVICE_FLIPMOUSE
162  if(adcStrongTimerHandle == NULL)
163  {
164  ESP_LOGE(LOG_TAG,"Strong mode timer uninitialized!!!");
165  return;
166  }
167 
168  //timer is not started, and we have a mode != NORMAL
169  if((xTimerIsTimerActive(adcStrongTimerHandle) == pdFALSE) && (D->strongmode != STRONG_NORMAL))
170  {
171  //than we need to start a timer for the timeout moving back to NORMAL
172  xTimerStart(adcStrongTimerHandle,0);
173  //return in this case & wait for next iteration
174  return;
175  }
176 
177  //if we have a mode != NORMAL
178  if(D->strongmode != STRONG_NORMAL)
179  {
180  //check if we have a movement in any direction
181  if(D->x != 0 || D->y != 0)
182  {
183  //if yes, trigger action (depending on SIP/PUFF mode)
184  if(D->strongmode == STRONG_PUFF)
185  {
186  if(abs(D->x) > abs(D->y))
187  {
188  //x has higher values -> use LEFT/RIGHT
189  if(D->x > 0) evt.vb = VB_STRONGPUFF_RIGHT;
190  else evt.vb = VB_STRONGPUFF_LEFT;
191  evt.type = VB_PRESS_EVENT;
192  xQueueSendToBack(debouncer_in,&evt,0);
193  } else {
194  //y has higher values -> use UP/DOWN
195  if(D->y > 0) evt.vb = VB_STRONGPUFF_DOWN;
196  else evt.vb = VB_STRONGPUFF_UP;
197  evt.type = VB_PRESS_EVENT;
198  xQueueSendToBack(debouncer_in,&evt,0);
199  }
200  }
201  if(D->strongmode == STRONG_SIP)
202  {
203  if(abs(D->x) > abs(D->y))
204  {
205  //x has higher values -> use LEFT/RIGHT
206  if(D->x > 0) evt.vb = VB_STRONGSIP_RIGHT;
207  else evt.vb = VB_STRONGSIP_LEFT;
208  evt.type = VB_PRESS_EVENT;
209  xQueueSendToBack(debouncer_in,&evt,0);
210  } else {
211  //y has higher values -> use UP/DOWN
212  if(D->y > 0) evt.vb = VB_STRONGSIP_DOWN;
213  else evt.vb = VB_STRONGSIP_UP;
214  evt.type = VB_PRESS_EVENT;
215  xQueueSendToBack(debouncer_in,&evt,0);
216  }
217  }
218  //cancel timer
219  xTimerStop(adcStrongTimerHandle,0);
220  xTimerReset(adcStrongTimerHandle,0);
221 
222  //and set back to normal
223  D->strongmode = STRONG_NORMAL;
224  }
225  }
226  #endif
227 }

310 void halAdcReadData(adcData_t *values)
311 {
312  //read all sensors
313  int32_t tmp = 0;
314  int32_t x,y;
315  int32_t up,down,left,right,pressure;
316  up = down = left = right = pressure = 0;
318  static uint32_t debug_out_cnt = 0;
319 
320  //read sensor data via I2C.
321  uint8_t adc[10];
322  uint8_t *adc2 = adc;
323  int rcv = halSerialReceiveI2CADC(&adc2);
324  if(rcv != 10)
325  {
326  ESP_LOGW(LOG_TAG,"I2C recv: 0x%X",rcv);
327  }
328  left = (int32_t)((uint16_t)(adc[2] + (adc[3] << 8)));
329  right = (int32_t)((uint16_t)(adc[6] + (adc[7] << 8)));
330  up = (int32_t)((uint16_t)(adc[4] + (adc[5] << 8)));
331  down = (int32_t)((uint16_t)(adc[0] + (adc[1] << 8)));
332  pressure = (int32_t)((uint16_t)(adc[8] + (adc[9] << 8)));
333  //ESP_LOGE(LOG_TAG,"%d,%d,%d,%d,%d",up,down,left,right,pressure);
334 
335  //do the mouse rotation
336  switch (adc_conf.orientation) {
337  case 90: tmp=up; up=left; left=down; down=right; right=tmp; break;
338  case 180: tmp=up; up=down; down=tmp; tmp=right; right=left; left=tmp; break;
339  case 270: tmp=up; up=right; right=down; down=left; left=tmp;break;
340  }
341 
342  //save raw values to data struct
343  values->left = left;
344  values->right = right;
345  values->up = up;
346  values->down = down;
347 
348  //process pressure
349  values->pressure = pressure;
350 
351  //calculate X and Y values
352  x = (left - right) - offsetx;
353  y = (up - down) - offsety;
354 
355  #if HAL_IO_ADC_ELLIPTIC_DEADZONE == 0
356 
357  //in this case, we use a rectangle deadzone
358 
359  if (x<-adc_conf.deadzone_x) x+=adc_conf.deadzone_x; // apply deadzone values x direction
360  else if (x>adc_conf.deadzone_x) x-=adc_conf.deadzone_x;
361  else x=0;
362 
363  if (y<-adc_conf.deadzone_y) y+=adc_conf.deadzone_y; // apply deadzone values y direction
364  else if (y>adc_conf.deadzone_y) y-=adc_conf.deadzone_y;
365  else y=0;
366 
367  //otherwise, x&y is 0
368  values->x = x;
369  values->y = y;
370 
371  #else
372 
373  //in this case, we use an elliptic deadzone
374 
375  //apply elliptic deadzone
376  //formula:
377  //https://sebadorn.de/2012/01/02/herausfinden-ob-ein-punkt-in-einer-ellipse-liegt
378  //A point in an elliptic curve:
379  //(px - cx) / rx + (py - cy) / ry <= 1
380  //cx/cy is 0 (we start at 0/0)
381  //py/py is our x/y
382  //rx/ry are deadzone values
383  //to shorten the formulas
384  uint8_t a = adc_conf.deadzone_x;
385  uint8_t b = adc_conf.deadzone_y;
386  float status = pow(x,2) / pow(a,2) + pow(y,2) / pow(b,2);
387 
388  //check if point is outside deadzone
389  if(status > 1.0)
390  {
391  //if outside deadzone, subtract ellipse itself to start with
392  //0 for a movement
393 
394  //formula for ellipse point:
395  //https://math.stackexchange.com/questions/22064/calculating-a-point-that-lies-on-an-ellipse-given-an-angle
396  float deadzoneX = 0;
397  float deadzoneY = 0;
398 
399  //angle of the given mouthpiece
400  //calculate only if x&y is != 0
401  if((x < 0 || x > 0) && (y < 0 || y > 0))
402  {
403  float angle = atan(y/x);
404  deadzoneX = abs((a*b)/sqrt(pow(b,2)+pow(a,2)*pow(tan(angle),2)));
405  deadzoneY = abs((a*b)/sqrt(pow(a,2)+ pow(b,2)/pow(tan(angle),2)));
406  }
407 
408  //subtract calculated ellipse coordinates from output X/Y values
409  if(x > 0)
410  {
411  values->x = x - (int)deadzoneX;
412  } else {
413  values->x = x + (int)deadzoneX;
414  }
415  if(y > 0)
416  {
417  values->y = y - (int)deadzoneY;
418  } else {
419  values->y = y + (int)deadzoneY;
420  }
421  } else {
422  //otherwise, x&y is 0
423  values->x = 0;
424  values->y = 0;
425  }
426  #endif
427  if(debug_out_cnt++%HAL_ADC_RAW_DIVIDER == 0)
428  {
429  ESP_LOGD(LOG_TAG,"raw x/y %d/%d; ",values->x,values->y);
430  }
431 }
432 #endif /* DEVICE_FLIPMOUSE */

438 void halAdcProcessPressure(adcData_t *D)
439 {
440  //track changes and fire event only if not issued already.
441  //this is done individually for SIP, PUFF, STRONG_SIP & STRONG_PUFF
442  //0 means nothing is fired, 1 is a fired press action, 2 is a fired release action
443  static uint8_t fired[4] = {0,0,0,0};
444  //currently measured pressure value
445  uint32_t pressurevalue = D->pressure;
446  //currently active general config
447  generalConfig_t *cfg = configGetCurrent();
448  //the raw event, sent to the debouncer
449  raw_action_t evt;
450  //cannot proceed if no global config is available
451  if(cfg == NULL) return;
452 
453  //if we are in a special mode don't process further (only on FLipMouse)
454  #ifdef DEVICE_FLIPMOUSE
455  if(D->strongmode != STRONG_NORMAL) return;
456  #endif
457 
458  //SIP triggered
459  if(pressurevalue < cfg->adc.threshold_sip && \
460  pressurevalue > cfg->adc.threshold_strongsip)
461  {
462  if(fired[0] != 1)
463  {
464  //set/clear VBs
465  evt.type = VB_PRESS_EVENT;
466  evt.vb = VB_SIP;
467  xQueueSendToBack(debouncer_in,&evt,0);
468  //save fired state
469  fired[0] = 1;
470  }
471  } else {
472  if(fired[0] == 1)
473  {
474  //set/clear VBs
475  evt.type = VB_RELEASE_EVENT;
476  evt.vb = VB_SIP;
477  xQueueSendToBack(debouncer_in,&evt,0);
478  //track fired state
479  fired[0] = 0;
480  }
481  }
482 
483  //STRONGSIP triggered
484  if(pressurevalue < cfg->adc.threshold_strongsip)
485  {
486  //check if strong sip + up/down/left/right is set and
487  //strong sip alone is unused
488  #ifdef DEVICE_FLIPMOUSE
489  //if(handler_hid_active(VB_SIP) || handler_vb_active(VB_SIP))
491 
492  if(0)
493  {
494  //if at least one strong action is defined and strong sip
495  //is unused, enter strong sip mode
496  D->strongmode = STRONG_SIP;
497  ESP_LOGI(LOG_TAG,"Enter STRONG SIP");
498  TONE(TONE_STRONGSIP_ENTER_FREQ,TONE_STRONGSIP_ENTER_DURATION);
499  } else {
500  #endif
501  if(fired[1] != 1)
502  {
503  //make a tone
504  TONE(TONE_STRONGSIP_ENTER_FREQ,TONE_STRONGSIP_ENTER_DURATION);
505  //either no strong sip + <yy> action is defined or strong
506  // is used, trigger strong sip VB.
507 
508  evt.type = VB_PRESS_EVENT;
509  evt.vb = VB_STRONGSIP;
510  xQueueSendToBack(debouncer_in,&evt,0);
511  //save fired stated
512  fired[1] = 1;
513  }
514  #ifdef DEVICE_FLIPMOUSE
515  }
516  #endif
517  } else {
518  if(fired[1] != 2)
519  {
520  //set/clear VBs
521  evt.type = VB_RELEASE_EVENT;
522  evt.vb = VB_STRONGSIP;
523  xQueueSendToBack(debouncer_in,&evt,0);
524  //track fired state
525  fired[1] = 2;
526  }
527  }
528 
529  //PUFF triggered
530  if(pressurevalue > cfg->adc.threshold_puff && \
531  pressurevalue < cfg->adc.threshold_strongpuff)
532  {
533 
534  if(fired[2] != 1)
535  {
536  //set/clear VBs
537  evt.type = VB_PRESS_EVENT;
538  evt.vb = VB_PUFF;
539  xQueueSendToBack(debouncer_in,&evt,0);
540  //save fired state
541  fired[2] = 1;
542  }
543  } else {
544  if(fired[2] == 1)
545  {
546  //set/clear VBs
547  evt.type = VB_RELEASE_EVENT;
548  evt.vb = VB_PUFF;
549  xQueueSendToBack(debouncer_in,&evt,0);
550  //track fired state
551  fired[2] = 0;
552  }
553  }
554 
555  //STRONGPUFF triggered
556  if(pressurevalue > cfg->adc.threshold_strongpuff)
557  {
558  //check if strong puff + up/down/left/right is set and
559  //strong puff alone is unused
560  #ifdef DEVICE_FLIPMOUSE
561  //if(handler_hid_active(VB_SIP) || handler_vb_active(VB_SIP))
563  if(0)
564  {
565  //if at least one strong action is defined and strong puff
566  //is unused, enter strong puff mode
567  D->strongmode = STRONG_PUFF;
568  ESP_LOGI(LOG_TAG,"Enter STRONG PUFF");
569  TONE(TONE_STRONGPUFF_ENTER_FREQ,TONE_STRONGPUFF_ENTER_DURATION);
570  } else {
571  #endif
572  if(fired[3] != 1)
573  {
574  //make a tone
575  TONE(TONE_STRONGPUFF_ENTER_FREQ,TONE_STRONGPUFF_ENTER_DURATION);
576  //either no strong puff + <yy> action is defined or strong
577  // is used, trigger strong puff VB.
578  evt.type = VB_PRESS_EVENT;
579  evt.vb = VB_STRONGPUFF;
580  xQueueSendToBack(debouncer_in,&evt,0);
581  //save fired state
582  fired[3] = 1;
583  }
584 
585  #ifdef DEVICE_FLIPMOUSE
586  }
587  #endif
588  } else {
589  if(fired[3] != 2)
590  {
591  //set/clear VBs
592  evt.type = VB_RELEASE_EVENT;
593  evt.vb = VB_STRONGPUFF;
594  xQueueSendToBack(debouncer_in,&evt,0);

598  }

616 void halAdcTaskMouse(void * pvParameters)
617 {
618  //analog values
619  adcData_t D;
620  D.strongmode = STRONG_NORMAL;
621  //int32_t x,y;
622  static uint16_t accelTimeX=0,accelTimeY=0;
623  int32_t tempX,tempY;
624  float moveVal, accumXpos = 0, accumYpos = 0;
625  //todo: use a define for the delay here... (currently used: value from vTaskDelay())
626  float accelFactor= 20 / 100000000.0f;
627  hid_cmd_t command,command2;
628  TickType_t xLastWakeTime;
629  //set adc data reference for timer
630  vTimerSetTimerID(adcStrongTimerHandle,&D);
631  uint32_t debug_out_cnt = 0;
632 
633  while(1)
634  {
635  //get mutex
636  if(xSemaphoreTake(adcSem, (TickType_t) 30) != pdTRUE)
637  {
638  ESP_LOGW(LOG_TAG,"Cannot obtain mutex for reading");
639  continue;
640  }
641 
642  //read out the analog voltages from all 5 channels (including deadzone)
643  //& set calibrate request to 0 before
644  D.calibrate_request = 0;
645  halAdcReadData(&D);
646 
647 
648  //if you want to slow down, uncomment following two lines
649  //ESP_LOGD(LOG_TAG,"X/Y square, X/Y ellipse: %d/%d, %d/%d",tempX,tempY,D.x,D.y);
650  //vTaskDelay(10);
651 
652  //report raw values.
653  halAdcReportRaw(D.up, D.down, D.left, D.right, D.pressure, D.x, D.y);
654 
655  //apply acceleration
656  if (D.x==0) accelTimeX=0;
657  else if (accelTimeX < ACCELTIME_MAX) accelTimeX+=adc_conf.acceleration;
658  if (D.y==0) accelTimeY=0;
659  else if (accelTimeY < ACCELTIME_MAX) accelTimeY+=adc_conf.acceleration;
660 
661  //calculate the current X movement by using acceleration, accel factor and sensitivity
662  moveVal = D.x * adc_conf.sensitivity_x * accelFactor * accelTimeX;
663  //limit value
664  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
665  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
666  //add to accumulated movement value
667  accumXpos+=moveVal;
668 
669  //do the same calculations for Y axis
670  moveVal = D.y * adc_conf.sensitivity_y * accelFactor * accelTimeY;
671  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
672  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
673  accumYpos+=moveVal;
674 
675  //cast to int again
676  tempX = accumXpos;
677  tempY = accumYpos;
678 
679  //limit to int8 values (to fit into mouse report)
680  if(tempX > 127) tempX = 127;
681  if(tempX < -127) tempX = -127;
682  if(tempY > 127) tempY = 127;
683  if(tempY < -127) tempY = -127;
684 
685  if(debug_out_cnt++%HAL_ADC_RAW_DIVIDER == 0)
686  {
687  ESP_LOGD(LOG_TAG,"mouse x/y %d/%d; ",tempX,tempY);
688  }
689 
690  //TODO: wenn D.strongmode != noraml > eigene fkt. mit berechneten werten.
691  //if we are in a special strong mode, do NOT send accumulated data
692  //to USB/BLE. Instead, call halAdcProcessStrongMode
693  if(D.strongmode == STRONG_NORMAL)
694  {
695  if(tempX != 0 && tempY != 0)
696  {
697  command.cmd[0] = 0x01;
698  command.cmd[1] = tempX;
699  command.cmd[2] = tempY;
700 
701  accumXpos -= tempX;
702  accumYpos -= tempY;
703 
704  //post values to mouse queue (USB and/or BLE)
705  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
706  { xQueueSend(hid_usb,&command,0); }
707 
708  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
709  { xQueueSend(hid_ble,&command,0); }
710  }
711 
712  if(tempX != 0 && tempY == 0)
713  {
714  command.cmd[1] = tempX;
715  accumXpos -= tempX;
716  command.cmd[0] = 0x10; //send X axis
717 
718  //post values to mouse queue (USB and/or BLE)
719  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
720  { xQueueSend(hid_usb,&command,0); }
721 
722  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
723  { xQueueSend(hid_ble,&command,0); }
724  }
725  if(tempY != 0 && tempX == 0)
726  {
727  command2.cmd[1] = tempY;
728  accumYpos -= tempY;
729  command2.cmd[0] = 0x11; //send Y axis
730 
731  //post values to mouse queue (USB and/or BLE)
732  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
733  { xQueueSend(hid_usb,&command2,0); }
734 
735  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
736  { xQueueSend(hid_ble,&command2,0); }
737  }
738 
739  //pressure sensor is handled in another function
740  halAdcProcessPressure(&D);
741  } else {
742  //in special mode, process strong mdoe
743  halAdcProcessStrongMode(&D);
744  }
745 
746  //give mutex
747  xSemaphoreGive(adcSem);
748 
749  //if OTF calibration is requested:
750  if(D.calibrate_request != 0) halAdcCalibrate();
751 
752  //delay the task.
753  vTaskDelayUntil( &xLastWakeTime, 20/portTICK_PERIOD_MS);
754  }
755 }

774 void halAdcTaskJoystick(void * pvParameters)
775 {
776  //analog values
777  adcData_t D;
778  D.strongmode = STRONG_NORMAL;
779  int32_t x,y;
780  //joystick_command_t command;
781  TickType_t xLastWakeTime;
782  //set adc data reference for timer
783  vTimerSetTimerID(adcStrongTimerHandle,&D);
784 
785  while(1)
786  {
787  //get mutex
788  if(xSemaphoreTake(adcSem, (TickType_t) 30) != pdTRUE)
789  {
790  ESP_LOGW(LOG_TAG,"Cannot obtain mutex for reading");
791  continue;
792  }
793 
794  //read out the analog voltages from all 5 channels
795  halAdcReadData(&D);
796

841 void halAdcCalibrate(void)
842 {
843  //check for initialized mutex
844  if(adcSem == NULL) return;
845 
846  //if we are updating now, do not calibrate.
847  if((xEventGroupGetBits(systemStatus) & SYSTEM_STABLECONFIG) == 0) return;
848 
849  //get mutex
850  if(xSemaphoreTake(adcSem, (TickType_t) 20))
851  {
852  if((xTaskGetTickCount() - adcCalibLast) < (HAL_ADC_CALIB_LOCKTIME / portTICK_PERIOD_MS))
853  {
854  ESP_LOGI(LOG_TAG,"Calibration lock time not passed yet");
855  //give mutex to enable tasks again
856  xSemaphoreGive(adcSem);
857  return;
858  }
859  ESP_LOGI(LOG_TAG,"Starting calibration, offsets: %d/%d",offsetx,offsety);
860  adcData_t D;
861 
862  uint32_t up = 0;
863  uint32_t down = 0;
864  uint32_t left = 0;
865  uint32_t right = 0;
866 
867  //save for next iteration
868  adcCalibLast = xTaskGetTickCount();
869 
870  //read values itself & accumulate (8sensor readings)
871  for(uint8_t i = 0; i<8; i++)
872  {
873 
874  //use read data for acquiring offset correction data
875  halAdcReadData(&D);
876 
877  //accumulate data
878  up+= D.up;
879  left+= D.left;
880  right+= D.right;
881  down+= D.down;
882 
883  //wait 2 ticks
884  vTaskDelay(2);
885  }
886 
887  //divide by 8 readings
888  up = up / 8;
889  down = down / 8;
890  left = left / 8;
891  right = right / 8;
892  //set as offset values
893  offsetx = left - right;
894  offsety = up - down;
895 
896  ESP_LOGI(LOG_TAG,"Finished calibration, offsets: %d/%d",offsetx,offsety);
897 
898  TONE(TONE_CALIB_FREQ,TONE_CALIB_DURATION);
899 
900  //give mutex to enable tasks again
901  xSemaphoreGive(adcSem);
902  } else {
903  ESP_LOGE(LOG_TAG,"Cannot calibrate, no mutex");
904  }
905 
906  return;
