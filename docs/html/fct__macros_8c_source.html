46 esp_err_t fct_macro(char *param)
47 {
48  //check for param struct
49  if(param == NULL)
50  {
51  ESP_LOGE(LOG_TAG,"param is NULL ");
52  return ESP_FAIL;
53  }
54 
55  int offset = 0;
56  int start = 0;
57 
58  while(offset < ATCMD_LENGTH)
59  {
60  //end loop if terminators are detected.
61  if(param[offset] == '\r' || param[offset] == '\n') break;
62 
63  //do we reach a command terminator?
64  if(param[offset] == ';')
65  {
66  //check if this character is escaped:
67  if(offset > 0)
68  {
69  if(param[offset-1] == '\\')
70  {
71  //if it escaped "\;", just continue.
72  offset++;
73  continue;
74  }
75  }
76 
77  //check if we hit an AT WA (wait)
78  if(memcmp(&param[start],"AT WA",5) == 0)
79  {
80  //if yes, delay this task.
81  uint32_t time = strtol((char*)&(param[start+6]),NULL,10);
82  if(time < 30000)
83  {
84  vTaskDelay(time / portTICK_PERIOD_MS);
85  ESP_LOGD(LOG_TAG,"Waiting: %d ms",time);
86  } else {
87  ESP_LOGE(LOG_TAG,"Hit AT WA with a delay time too high: %d",time);
88  }
89  } else {
90  //if not an AT WA, allocate memory and send to queue.
91  atcmd_t command;
92  uint16_t length = offset-start;
93  uint8_t *buffer = malloc(sizeof(uint8_t)*(length+1));
94  if(buffer != NULL)
95  {
96  //copy data
97  memcpy(buffer,&param[start],length);
98  //terminate
99  buffer[length] = 0;
100  //save to queue struct
101  command.buf = buffer;
102  command.len = length;
103  ESP_LOGD(LOG_TAG,"Sent AT cmd: %s",buffer);
104  //send to queue, wait maximum 10 ticks (100ms) for a free space.
105  if(xQueueSend(halSerialATCmds,(void*)&command,10) != pdTRUE)
106  {
107  ESP_LOGE(LOG_TAG,"Cmd queue is full, cannot send command");
108  }
109  } else {
110  ESP_LOGE(LOG_TAG,"Cannot allocate memory for command!");
111  }
112  }
113 
114  //save new start position for next command
115  start = offset + 1;
116  }
117 
118  //go to next character
119  offset++;
120  }
121 
122  return ESP_OK;
123 }
