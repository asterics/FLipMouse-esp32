46 void fct_infrared_send(char* cmdName)
47 {
48  //local IR struct
49  halIOIR_t *cfg = malloc(sizeof(halIOIR_t));
50  //transaction ID for IR data
51  uint32_t tid;
52  if(halStorageStartTransaction(&tid,20,LOG_TAG) == ESP_OK)
53  {
54  if(halStorageLoadIR(cmdName,cfg,tid) == ESP_OK)
55  {
56  //send pointer to IR send queue
57  if(cfg != NULL)
58  {
59  ESP_LOGI(LOG_TAG,"Triggering IR cmd, length %d",cfg->count);
60  SENDIRSTRUCT(cfg);
61  //free config afterwards (whole config is saved to queue)
62  free(cfg);
63  } else {
64  ESP_LOGE(LOG_TAG,"IR cfg is NULL!");
65  }
66  //create tone
67  TONE(TONE_IR_SEND_FREQ,TONE_IR_SEND_DURATION);
68  } else {
69  ESP_LOGE(LOG_TAG,"Error loading IR cmd");
70  }
71  halStorageFinishTransaction(tid);
72  } else {
73  ESP_LOGE(LOG_TAG,"Error starting transaction for IR cmd");
74  }
75 }

90 esp_err_t fct_infrared_record(char* cmdName, uint8_t outputtoserial)
91 {
92  uint16_t timeout = 0;
93  if(strnlen(cmdName,SLOTNAME_LENGTH) == SLOTNAME_LENGTH)
94  {
95  ESP_LOGE(LOG_TAG,"IR command name too long (maximum %d chars)",SLOTNAME_LENGTH);
96  return ESP_FAIL;
97  }
98  //transaction id
99  uint32_t tid;
100 
101  //create the IR struct
102  halIOIR_t *cfg = malloc(sizeof(halIOIR_t));
103  rmt_item32_t *buf = malloc(sizeof(rmt_item32_t)*TASK_HAL_IR_RECV_MAXIMUM_EDGES);
104  cfg->buffer = buf;
105  cfg->status = IR_RECEIVING;
106 
107  //check if memory was allocated
108  if(buf == NULL || cfg == NULL)
109  {
110  ESP_LOGE(LOG_TAG,"Error allocating IR memory");
111  if(buf != NULL) free(buf);
112  if(cfg != NULL) free(cfg);
113  return ESP_FAIL;
114  }
115 
116 
117  //put it to queue
118  //we assume there is space in the queue
119  xQueueSend(halIOIRRecvQueue, (void *)&cfg, (TickType_t)0);
120 
121  //check status every two ticks
122  while(cfg->status == IR_RECEIVING)
123  {
124  vTaskDelay(2);
125  timeout+=2;
126  if((timeout/portTICK_PERIOD_MS) > TASK_HAL_IR_RECV_TIMEOUT)
127  {
128  ESP_LOGW(LOG_TAG,"IR timeout waiting for status change");
129  //free buffers
130  free(cfg);
131  free(buf);
132  return ESP_FAIL;
133  }
134  }
135 
136  switch(cfg->status)
137  {
138  case IR_TOOSHORT:
139  ESP_LOGW(LOG_TAG,"IR cmd too short");
140  //free buffers
141  free(cfg);
142  free(buf);
143  return ESP_FAIL;
144  case IR_FINISHED:
145  //finished, storing
146  if(halStorageStartTransaction(&tid, 20,LOG_TAG) != ESP_OK)
147  {
148  ESP_LOGE(LOG_TAG,"Cannot start transaction");
149  //free buffers
150  free(cfg);
151  free(buf);
152  return ESP_FAIL;
153  }
154  if(halStorageStoreIR(tid, cfg, cmdName) != ESP_OK)
155  {
156  ESP_LOGE(LOG_TAG,"Cannot store IR cmd");
157  }
158  halStorageFinishTransaction(tid);
159  //create tone
160  TONE(TONE_IR_RECV_FREQ,TONE_IR_RECV_DURATION);
161  //send out a hex stream, if enabled
162  if(outputtoserial != 0)
163  {
164  char output[8*cfg->count + 4];
165  for(uint16_t i = 0; i<cfg->count; i++)
166  {
167  sprintf(&output[i*8],"%08X\r\n",cfg->buffer[i].val);
168  }
169  halSerialSendUSBSerial(output,strnlen(output,8*cfg->count + 4),10);
170  }
171  break;
172  case IR_OVERFLOW:
173  ESP_LOGW(LOG_TAG,"IR cmd too long");
174  //free buffers
175  free(cfg);
176  free(buf);
177  return ESP_FAIL;
178  default:
179  ESP_LOGE(LOG_TAG,"Unknown IR recv status");
180  //free buffers
181  free(cfg);
182  free(buf);
183  return ESP_FAIL;
184  }
185 
186  //everything fine...
187  //free buffers
188  free(cfg);
189  free(buf);
190  return ESP_OK;
191 }
