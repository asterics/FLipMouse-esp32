252 void halStorageCreateDefault(uint32_t tid)
253 {
254  char file[sizeof(base_path)+32];
255  //check tid
256  if(halStorageChecks(tid) != ESP_OK)
257  {
258  ESP_LOGE(LOG_TAG,"Cannot create default config, checks failed");
259  return;
260  }
261 
262  //create filename string to open the default slot (either FABI or FLipMouse)
263  #ifdef DEVICE_FLIPMOUSE
264  sprintf(file,"%s/flip.set",base_path);
265  #endif
266  #ifdef DEVICE_FABI
267  sprintf(file,"%s/fabi.set",base_path);
268  #endif
269  FILE *source = fopen(file, "rb");
270  if(source == NULL)
271  {
272  ESP_LOGE(LOG_TAG,"Cannot open default file for factory reset!");
273  return;
274  }
275 
276  //current slot number used for filename
277  int slotnr = -1;
278  //target file pointer
279  FILE *target = NULL;
280 
281  //credits:
282  //https://stackoverflow.com/questions/1006797/tried-and-true-simple-file-copying-code-in-c#1010207
283  char *buffer = malloc(512);
284  if(buffer == NULL)
285  {
286  ESP_LOGE(LOG_TAG,"Cannot alloc buf for default slot");
287  fclose(source);
288  return;
289  }
290 
291  //read file & open/close individual slot files
292  while (fgets(buffer, 512, source) != NULL)
293  {
294  //if we get a new slot by receiving a string "Slot X:<name>"
295  //we close the target file and open a new one.
296  //need to do it with strncmp to have a prefix check
297  if(strncmp("Slot", buffer, strlen("Slot")) == 0)
298  {
299  if(target != NULL) fclose(target);
300  slotnr++;
301  //create target filename (overwrite current config)
302  sprintf(file,"%s/%03d.set",base_path,slotnr);
303  target = fopen(file, "wb");
304  if(target == NULL)
305  {
306  ESP_LOGE(LOG_TAG,"Cannot open target config file \"%s\" for factory reset!",file);
307  free(buffer);
308  if(source != NULL) fclose(source);
309  return;
310  }
311  }
312 
313  //write the buffer to the new file
314  if(fputs(buffer, target) == EOF)
315  {
316  ESP_LOGE(LOG_TAG,"write failed for default slot");
317  free(buffer);
318  fclose(source);
319  fclose(target);
320  return;
321  }
322  }
323 
324  ESP_LOGI(LOG_TAG,"Factory reset, copied default file over config");
325  free(buffer);
326  fclose(source);
327  fclose(target);
328 }

460 esp_err_t halStorageDeleteIRCmd(int16_t slotnr, uint32_t tid)
461 {
462  char file[sizeof(base_path)+32];
463  char filenew[sizeof(base_path)+32];
464  int ret;
465 
466  if(slotnr > 250)
467  {
468  ESP_LOGE(LOG_TAG,"Cannot delete IR, slotnr too high");
469  return ESP_FAIL;
470  }
471 
472  //delete by starting & ending at given slotnumber
473  uint8_t from = slotnr;
474  uint8_t to = slotnr;
475 
476  //in case of deleting all slots, start at 0 and delete until the last slot
477  if(slotnr == -1)
478  {
479  from = 0;
480  if(halStorageGetNumberOfIRCmds(tid,&to) != ESP_OK)
481  {
482  ESP_LOGE(LOG_TAG,"Cannot get number of IR slots, cannot delete all");
483  return ESP_FAIL;
484  }
485  //if we have 2 slots, we need to delete IR_000.set & IR_001.set -> decrement here
486  to--;
487  }
488 
489  //check for valid storage handle
490  if(halStorageChecks(tid) != ESP_OK) return ESP_FAIL;
491 
492  //delete one or all slots
493  struct stat st;
494  for(uint8_t i = from; i<=to; i++)
495  {
496  sprintf(file,"%s/IR_%03d.set",base_path,i);
497  if (stat(file, &st) == 0) {
498  // Delete it if it exists
499  unlink(file);
500  }
501  //not necessary, ESP32 uses preemption
502  //taskYIELD();
503  }
504 
505  //re-arrange all following slots (of course, only if not deleting all)
506  if(slotnr != -1)
507  {
508 
509  for(uint8_t i = to+1; i<=249; i++)
510  {
511  sprintf(file,"%s/IR_%03d.set",base_path,i);
512  sprintf(filenew,"%s/IR_%03d.set",base_path,i-1);
513  ret = rename(file,filenew);
514  if(ret != 0)
515  {
516  ESP_LOGI(LOG_TAG,"Stopped renaming @ IR cmd %d",i);
517  break;
518  }
519  }
520  }
521  if(slotnr == 250)
522  {
523  ESP_LOGW(LOG_TAG,"Deleted all IR commands");
524  } else {
525  ESP_LOGI(LOG_TAG,"Deleted IR cmd %d",slotnr);
526  }
527  return ESP_OK;
528 }

816 esp_err_t halStorageLoad(hal_storage_load_action navigate, uint32_t tid)
817 {
818  uint8_t slotCount = 0;
819  esp_err_t ret;
820 
821  //fetch current slot count
822  if(halStorageGetNumberOfSlots(tid, &slotCount) != ESP_OK)
823  {
824  ESP_LOGE(LOG_TAG,"cannot get number of available slots");
825  return ESP_FAIL;
826  }
827 
828  //check what we should load
829  switch(navigate)
830  {
831  case NEXT:
832  //check if we are at the last slot, if yes load first, increment otherwise
833  if((slotCount-1) == storageCurrentSlotNumber) storageCurrentSlotNumber = 0;
834  else storageCurrentSlotNumber++;
835  ret = halStorageLoadNumber(storageCurrentSlotNumber,tid,0);
836  //if NEXT does not succeed (because of a deleted slot or something else)
837  //retry with storageCurrentSlotNumber = 1;
838  if(ret != ESP_OK)
839  {
840  ESP_LOGI(LOG_TAG,"Resetting current slot number to 0");
841  storageCurrentSlotNumber = 0;
842  ret = halStorageLoadNumber(storageCurrentSlotNumber,tid,0);
843  }
844  return ret;
845  break;
846  case PREV:
847  //check if we are at the first slot, if yes load last, decrement otherwise
848  if(storageCurrentSlotNumber == 0) storageCurrentSlotNumber = slotCount - 1;
849  else storageCurrentSlotNumber--;
850  ret = halStorageLoadNumber(storageCurrentSlotNumber,tid,0);
851  //if NEXT does not succeed (because of a deleted slot or something else)
852  //retry with storageCurrentSlotNumber = 1;
853  if(ret != ESP_OK)
854  {
855  ESP_LOGI(LOG_TAG,"Resetting current slot number to 0");
856  storageCurrentSlotNumber = 0;
857  ret = halStorageLoadNumber(storageCurrentSlotNumber,tid,0);
858  }
859  return ret;
860  break;
861  case DEFAULT:
862  storageCurrentSlotNumber = 0;
863  return halStorageLoadNumber(0,tid,0);
864  break;
865  case RESTOREFACTORYSETTINGS:
866  break;
867  default:
868  ESP_LOGE(LOG_TAG,"unknown navigate action in halStorageLoad!");
869  break;
870  }
871  return ESP_OK;
872 }
