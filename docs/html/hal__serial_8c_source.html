117 void halSerialRXTask(void *pvParameters)
118 {
119  uint16_t cmdoffset;
120  uint8_t *buf;
121  uint8_t *bufstatic = malloc(ATCMD_LENGTH);
122  uint8_t data;
123  uint8_t parserstate = 0;
124  atcmd_t currentcmd;
125 
126  if(bufstatic == NULL)
127  {
128  ESP_LOGE(LOG_TAG,"Cannot allocate mem");
129  vTaskDelete(NULL);
130  }
131 
132  while(1)
133  {
134  int len = uart_read_bytes(HAL_SERIAL_UART, &data, 1, portMAX_DELAY);
135 
136  //only parse if a valid byte is received
137  if(len == 1)
138  {
139  switch(parserstate)
140  {
141  case 0:
142  //check for leading "A"
143  if(data == 'A' || data == 'a')
144  {
146  parserstate++;
147  }
148  //if starting with '@', go to CIM mode.
149  if(data == '@')
150  {
152  ESP_LOGW(LOG_TAG,"CIM mode currently unsupported!");
153  }
154  break;
155 
156 
157  case 1:
158  //check for second 'T'/'t'
159  if(data == 'T' || data == 't')
160  {
161  parserstate++;
162  //reset command offset to beginning (but after "AT")
163  cmdoffset = 2;
164  bufstatic[0] = 'A';
165  bufstatic[1] = 'T';
166 
167  } else {
168  //reset parser of no leading "AT" is detected
169  parserstate = 0;
170  }
171  break;
172 
173 
174  case 2:
175  //now read data until we reach \r or \n
176  if(data == '\r' || data == '\n')
177  {
178  //terminate string
179  bufstatic[cmdoffset] = 0;
180  //allocate a new buffer for this command to be processed later
181  //free is either done in halSerialReceiveUSBSerial.
182  buf = malloc(cmdoffset+1);
183  if(buf == NULL)
184  {
185  ESP_LOGE(LOG_TAG,"Cannot allocate %d B buffer for new AT cmd",cmdoffset+1);
186  parserstate = 0;
187  break;
188  }
189  memcpy(buf,bufstatic,cmdoffset+1);
190 
191  //send buffer to queue
192  currentcmd.buf = buf;
193  currentcmd.len = cmdoffset+1;
194  if(halSerialATCmds != NULL)
195  {
196  if(xQueueSend(halSerialATCmds,(void*)&currentcmd,10) != pdTRUE)
197  {
198  ESP_LOGE(LOG_TAG,"AT cmd queue is full, cannot send cmd");
199  free(buf);
200  } else {
201  ESP_LOGI(LOG_TAG,"Sent AT cmd with len %d to queue: %s",cmdoffset,bufstatic);
202  }
203  } else {
204  ESP_LOGE(LOG_TAG,"AT cmd queue is NULL, cannot send cmd");
205  free(buf);
206  }
207  parserstate = 0;
208  break;
209  }
210 
211  //if everything is fine, just save this byte.
212  bufstatic[cmdoffset] = data;
213  cmdoffset++;
214 
215  //check for memory length
216  if(cmdoffset == ATCMD_LENGTH)
217  {
218  ESP_LOGW(LOG_TAG,"AT cmd too long, discarding");
219  parserstate = 0;
220  }
221  break;
222 
223  default:
224  ESP_LOGE(LOG_TAG,"Unknown parser state");
225  parserstate = 0;
226  break;
227  }
228  }
229  }
230 
231  //we should never be here...
232  vTaskDelete(NULL);
233 }

267 void halSerialHIDTask(void *param)
268 {
269  hid_cmd_t rx;
270  //RMT RAM has 64x32bit memory each block
271  rmt_item32_t *rmtBuf = malloc(sizeof(rmt_item32_t)*64);
272 
273  if(rmtBuf == NULL)
274  {
275  ESP_LOGE(LOG_TAG,"Cannot allocate memory for RMT");
276  vTaskDelete(NULL);
277  }
278 
279  uint8_t rmtCount = 0;
280 
281  //levels are always same, set here.
282  rmt_item32_t item;
283  item.level0 = 1;
284  item.level1 = 0;
285 
286  while(1)
287  {
288  //check if queue is initialized
289  if(hid_usb != NULL)
290  {
291  //pend on MQ, if timeout triggers, just wait again.
292  if(xQueueReceive(hid_usb,&rx,portMAX_DELAY))
293  {
294  //reset RMT offset
295  rmtCount = 0;
296 
297  //output if debug
298  #if LOG_LEVEL_SERIAL >= ESP_LOG_DEBUG
299  ESP_LOGD(LOG_TAG,"HID: %02X:%02X:%02X",rx.cmd[0],rx.cmd[1],rx.cmd[2]);
300  #endif
301 
302  //build RMT buffer
303  for(uint8_t i = 0; i<3; i++)
304  {
305  //process 2 bits at once (one rmt item)
306  for(uint8_t j = 0; j<4; j++)
307  {
308  //process even bit
309  if((rx.cmd[i] & (1<<(j*2))) != 0)
310  {
311  item.duration0 = HAL_SERIAL_HID_DURATION_1; //long timing
312  } else {
313  item.duration0 = HAL_SERIAL_HID_DURATION_0; //short timing
314  }
315  //process odd bit
316  if((rx.cmd[i] & (1<<(j*2+1))) != 0)
317  {
318  item.duration1 = HAL_SERIAL_HID_DURATION_1;
319  } else {
320  item.duration1 = HAL_SERIAL_HID_DURATION_0;
321  }
322  //fill buffer
323  //rmtCount = ((i*8+j*2)/2) + 1;
324  rmtBuf[rmtCount] = item;
325  rmtCount++;
326  }
327  }
328 
329  //signal the RMT the end of this transmission
330  item.duration0 = HAL_SERIAL_HID_DURATION_S; //stop bit timing
331  item.duration1 = 0;
332  rmtBuf[rmtCount] = item;
333  rmtCount++;
334 
335  #if LOG_LEVEL_SERIALHID>=ESP_LOG_VERBOSE
336  ESP_LOGV(LOG_TAG,"RMT dump:");
337  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,rmtBuf, sizeof(rmt_item32_t)*(rmtCount),ESP_LOG_VERBOSE);
338  #endif
339 
340  //put data into RMT buffer
341  if(rmt_fill_tx_items(HAL_SERIAL_HID_CHANNEL, rmtBuf, \
342  rmtCount, 0) != ESP_OK)
343  {
344  ESP_LOGE(LOG_TAG,"Cannot send data to RMT");
345  }
346  //start TX, reset memory index (always send from beginning)
347  if(rmt_tx_start(HAL_SERIAL_HID_CHANNEL, 1) != ESP_OK)
348  {
349  ESP_LOGE(LOG_TAG,"Cannot start RMT TX");
350  }
351  rmt_wait_tx_done(HAL_SERIAL_HID_CHANNEL,portMAX_DELAY);
353  vTaskDelay(1);
354  }
355  } else {
356  ESP_LOGW(LOG_TAG,"usb hid queue not initialized, retry in 1s");
357  vTaskDelay(1000/portTICK_PERIOD_MS);
358  }
359  }
360 }

417 int halSerialSendUSBSerial(char *data, uint32_t length, TickType_t ticks_to_wait)
418 {
419  //check if sem is already initialized
420  if(serialsendingsem == NULL)
421  {
422  ESP_LOGW(LOG_TAG,"Sem not ready, waiting for 1s");
423  vTaskDelay(1000/portTICK_PERIOD_MS);
424  }
425 
426  //if an additional stream is registered, send data there as well
427  //no HID commands will be sent there.
428  if(outputcb != NULL)
429  {
430  if(outputcb(data,length) != ESP_OK)
431  {
432  ESP_LOGE(LOG_TAG,"Additional stream cannot be sent,removing stream!");
433  outputcb = NULL;
434  }
435  }
436 
437  //acquire mutex to have TX permission on UART
438  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
439  {
440  //send the data
441  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
442  uart_write_bytes(HAL_SERIAL_UART, HAL_SERIAL_LINE_ENDING, strnlen(HAL_SERIAL_LINE_ENDING,4));
443  //release mutex
444  xSemaphoreGive(serialsendingsem);
445 
446  return txBytes;
447  } else return -1;
448 }

464 {

466 
467  //if all HID types should be resetted,
468  //send global reset command (don't send 3 different commands)

473  //send to queue
474  xQueueSend(hid_usb, &m, 0);

491  //send to queue

496  {
497  hid_cmd_t j;
498  j.cmd[0] = 0x3F;
499  //send to queue
500  xQueueSend(hid_usb, &j, 0);

539 esp_err_t halSerialInit(void)
540 {
541  //set log level to given log level
542  esp_log_level_set(LOG_TAG,LOG_LEVEL_SERIAL);
543 
544  /*++++ UART config (UART to CDC) ++++*/
545 
546  esp_err_t ret = ESP_OK;
547  const uart_config_t uart_config = {
548  .baud_rate = 230400,
549  .data_bits = UART_DATA_8_BITS,
550  .parity = UART_PARITY_DISABLE,
551  .stop_bits = UART_STOP_BITS_1,
552  .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
553  };
554 
555  //update UART config
556  ret = uart_param_config(HAL_SERIAL_UART, &uart_config);
557  if(ret != ESP_OK)
558  {
559  ESP_LOGE(LOG_TAG,"UART param config failed");
560  return ret;
561  }
562 
563  //set IO pins
564  ret = uart_set_pin(HAL_SERIAL_UART, HAL_SERIAL_TXPIN, HAL_SERIAL_RXPIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
565  if(ret != ESP_OK)
566  {
567  ESP_LOGE(LOG_TAG,"UART set pin failed");
568  return ret;
569  }
570 
571  //Install UART driver with RX and TX buffers
572  ret = uart_driver_install(HAL_SERIAL_UART, BUF_SIZE_RX, BUF_SIZE_TX, 0, NULL, 0);
573  if(ret != ESP_OK)
574  {
575  ESP_LOGE(LOG_TAG,"UART driver install failed");
576  return ret;
577  }
578 
579  //create mutex for all tasks sending to serial TX queue.
580  //avoids splitting of different packets due to preemption
581  serialsendingsem = xSemaphoreCreateMutex();
582  if(serialsendingsem == NULL)
583  {
584  ESP_LOGE(LOG_TAG,"Cannot create semaphore for TX");
585  return ESP_FAIL;
586  }
587  //mutex for sending HID commands
588  //acquired before sending, released in RMT finish callback
589  hidsendingsem = xSemaphoreCreateMutex();
590  if(hidsendingsem == NULL)
591  {
592  ESP_LOGE(LOG_TAG,"Cannot create semaphore for HID");
593  return ESP_FAIL;
594  }
595 
596  //create the AT command queue
597  halSerialATCmds = xQueueCreate(CMDQUEUE_SIZE,sizeof(atcmd_t));
598 
599  /*++++ RMT config (sending HID commands) ++++*/
600  rmt_config_t hidoutput_rmt;
601 
602  hidoutput_rmt.rmt_mode = RMT_MODE_TX; //TX mode
603  hidoutput_rmt.channel = HAL_SERIAL_HID_CHANNEL; //use define RMT channel
604  hidoutput_rmt.clk_div = 10; //8MHz clock; we need to be fast...
605  hidoutput_rmt.gpio_num = HAL_SERIAL_HIDPIN; //output pin
606  hidoutput_rmt.mem_block_num = 1;
607  hidoutput_rmt.tx_config.loop_en = 0; //do not loop
608  hidoutput_rmt.tx_config.carrier_en = 0; //we don't need a carrier
609  hidoutput_rmt.tx_config.idle_output_en = 1;
610  hidoutput_rmt.tx_config.idle_level = 0;
611 
612  if(rmt_config(&hidoutput_rmt) != ESP_OK)
613  {
614  ESP_LOGE(LOG_TAG,"Cannot init RMT for HID output");
615  return ESP_FAIL;
616  }
617 
618  if(rmt_driver_install(HAL_SERIAL_HID_CHANNEL,0,0) != ESP_OK)
619  {
620  ESP_LOGE(LOG_TAG,"Cannot install driver for HID output");
621  return ESP_FAIL;
622  }
623 
624  /*++++ task setup ++++*/
625  //task for sending HID commands via RMT
626  xTaskCreate(halSerialHIDTask, "serialHID", HAL_SERIAL_TASK_STACKSIZE+256, NULL, configMAX_PRIORITIES-3, NULL);
627 
628  //Create a task to handler UART event from ISR
629  xTaskCreate(halSerialRXTask, "serialRX", HAL_SERIAL_TASK_STACKSIZE, NULL, 5, NULL);
630 
631  //everything went fine
632  ESP_LOGI(LOG_TAG,"Driver installation complete");
633  return ESP_OK;
634 }
