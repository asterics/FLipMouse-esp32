95 static esp_err_t wifi_sh_event_handler(void *ctx, system_event_t *event)
96 {
97  system_event_sta_disconnected_t *disconnected ;
98 
99  switch(event->event_id) {
100  case SYSTEM_EVENT_STA_START:
101  esp_wifi_connect();
102  ESP_LOGI(LOG_TAG_WIFI,"STA_START, now connecting");
103  break;
104  case SYSTEM_EVENT_STA_GOT_IP:
105  ESP_LOGI(LOG_TAG_WIFI,"GOT_IP, now active");
106  xEventGroupSetBits(smarthomestatus, SH_WIFI_ACTIVE);
107  break;
108  case SYSTEM_EVENT_STA_DISCONNECTED:
109  esp_wifi_connect();
110  disconnected = &event->event_info.disconnected;
111  ESP_LOGD(LOG_TAG_WIFI, "SYSTEM_EVENT_STA_DISCONNECTED, ssid:%s, ssid_len:%d, bssid:" MACSTR ", reason:%d,%s", \
112  disconnected->ssid, disconnected->ssid_len, MAC2STR(disconnected->bssid), disconnected->reason, wifi_get_reason(disconnected->reason));
113  ESP_LOGI(LOG_TAG_WIFI,"STA_DISCONNECT, now connecting");
114  xEventGroupClearBits(smarthomestatus, SH_WIFI_ACTIVE);
115  break;
116  default:
117  break;
118  }
119  return ESP_OK;
120 }

128 static esp_err_t mqtt_event_handler(esp_mqtt_event_handle_t event)
129 {
130  int msg_id = 0;
131  switch (event->event_id) {
132  case MQTT_EVENT_CONNECTED:
133  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_CONNECTED");
134  xEventGroupSetBits(smarthomestatus, SH_MQTT_ACTIVE);
135  //if we got connected -> start
136  esp_mqtt_client_start(mqtt_client);
137  break;
138  case MQTT_EVENT_DISCONNECTED:
139  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_DISCONNECTED");
140  xEventGroupClearBits(smarthomestatus, SH_MQTT_ACTIVE);
141  break;
142  case MQTT_EVENT_SUBSCRIBED:
143  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_SUBSCRIBED, msg_id=%d", event->msg_id);
144  ESP_LOGI(LOG_TAG_MQTT, "sent publish successful, msg_id=%d", msg_id);
145  break;
146  case MQTT_EVENT_UNSUBSCRIBED:
147  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_UNSUBSCRIBED, msg_id=%d", event->msg_id);
148  break;
149  case MQTT_EVENT_PUBLISHED:
150  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_PUBLISHED, msg_id=%d", event->msg_id);
151  break;
152  case MQTT_EVENT_DATA:
153  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_DATA");
154  printf("TOPIC=%.*s\r\n", event->topic_len, event->topic);
155  printf("DATA=%.*s\r\n", event->data_len, event->data);
156  break;
157  case MQTT_EVENT_ERROR:
158  ESP_LOGI(LOG_TAG_MQTT, "MQTT_EVENT_ERROR");
159  break;
160  default:
161  ESP_LOGI(LOG_TAG_MQTT, "Other event id:%d", event->event_id);
162  break;
163  }
164  return ESP_OK;
165 }

178 esp_err_t taskMQTTInit(void)
179 {
180  if(checkeventgroup() != ESP_OK) return ESP_FAIL;
181 
182  //check if already initialized, if yes we are finished here, if no:
183  //set bit immediately and continue
184  if(xEventGroupGetBits(smarthomestatus) & SH_MQTT_INITIALIZED) return ESP_OK;
185  xEventGroupSetBits(smarthomestatus, SH_MQTT_INITIALIZED);
186 
187  //check if we need to initialize wifi before
188  if(!(xEventGroupGetBits(smarthomestatus) & SH_WIFI_INITIALIZED))
189  {
190  taskWifiInit();
191  }
192 
193  //set the log level for MQTT
194  esp_log_level_set(LOG_TAG_MQTT, LOG_LEVEL_MQTT);
195 
197  char mqttbroker[101];
198  esp_err_t ret;
199 
200  //get mqtt broker URL
201  ret = halStorageNVSLoadString(NVS_MQTT_BROKER,mqttbroker);
202  if(ret != ESP_OK)
203  {
204  ESP_LOGE(LOG_TAG_MQTT,"Error reading MQTT broker, cannot connect: %d",ret);
205  return ESP_FAIL;
206  }
207 
208  //init mqtt
209  esp_mqtt_client_config_t mqtt_cfg = {
210  .uri = mqttbroker,
211  .event_handle = mqtt_event_handler,
212  };
213  mqtt_client = esp_mqtt_client_init(&mqtt_cfg);
214 
215  return ESP_OK;
216 }

366 esp_err_t taskMQTTPublish(char* topic_payload)
367 {
368  //we can call it multiple times, if the flag is set, init does nothing
369  if(taskMQTTInit() != ESP_OK)
370  {
371  ESP_LOGE(LOG_TAG_MQTT,"Error init MQTT");
372  return ESP_FAIL;
373  }
374 
375  char delim[3] = ":";
376  esp_err_t ret = halStorageNVSLoadString(NVS_MQTT_DELIM,delim);
377  if(ret != ESP_OK)
378  {
379  ESP_LOGI(LOG_TAG_MQTT,"Using default delimiter");
380  delim[0] = ':';
381  delim[1] = '\0';
382  }
383 
384 
385  //split into topic & payload
386  char* sep = strpbrk(topic_payload,&delim[0]);
387  if(sep == NULL)
388  {
389  ESP_LOGE(LOG_TAG_MQTT,"Wrong delimiter, cannot send MQTT message");
390  return ESP_FAIL;
391  }
392  //save topic data
393  char* topic = malloc(sep-topic_payload+1);
394  strncpy(topic,topic_payload,sep-topic_payload+1);
395  topic[sep-topic_payload] = '\0';
396 
397  char* payload = malloc(strnlen(sep+1,200)+1);
398  strncpy(payload,sep+1,strnlen(sep+1,200)+1);
399  payload[strnlen(sep+1,200)] = '\0';
400 
401  ESP_LOGI(LOG_TAG_MQTT,"Publish: %s @ %s",payload,topic);
402 
403  //publish message
404  esp_mqtt_client_publish(mqtt_client, topic, payload, 0, 0, 0);
405 
406  //free allocated strings
407  if(topic != NULL) free(topic);
408  if(payload != NULL) free(payload);
409 
410  return ESP_OK;
411 }

424 esp_err_t taskREST(char* uri)
425 {
426  //we can call it multiple times, if the flag is set, init does nothing
427  if(taskWifiInit() != ESP_OK)
428  {
429  ESP_LOGE(LOG_TAG_REST,"Error init WiFi");
430  return ESP_FAIL;
431  }
432  //wait ~2s for the WIFI_ACTIVE flag to be set.
433  //if not set in time, just try anyway sending the REST call.
434  xEventGroupWaitBits(smarthomestatus,SH_WIFI_ACTIVE,pdFALSE, \
435  pdFALSE,5000/portTICK_PERIOD_MS);
436 
437  //set log level
438  esp_log_level_set(LOG_TAG_REST, LOG_LEVEL_REST);
439 
440  //copy the given URL, because we cannot be sure how
441  //this string will be valid.
442  char* url = malloc(strnlen(uri,ATCMD_LENGTH)+1);
443  strncpy(url,uri,strnlen(uri,ATCMD_LENGTH)+1);
444 
445  //init with empty config
446  esp_http_client_config_t config = {
447  .cert_pem = "",
448  .url = url
449  };
450 
451  esp_http_client_handle_t client = esp_http_client_init(&config);
452  if(client == NULL)
453  {
454  ESP_LOGE(LOG_TAG_REST,"Error init http client");
455  return ESP_FAIL;
456  }
457  //set the given URL
458  if(esp_http_client_set_url(client,uri) != ESP_OK)
459  {
460  ESP_LOGE(LOG_TAG_REST,"Error setting URI");
461  return ESP_FAIL;
462  }
463  //preform call
464  if(esp_http_client_perform(client) != ESP_OK)
465  {
466  ESP_LOGE(LOG_TAG_REST,"Error http_client_perform");
467  return ESP_FAIL;
468  }
469  //cleanup & return
470  esp_http_client_cleanup(client);
471  return ESP_OK;
472 }
