262 void fat_serve(char* resource, int fd) {
263  //if we serve a html file for a different URL, Content Type
264  //is not set correctly (no file ending given).
265  //Set this value to != 0 to force ContentType = text/html
266  uint8_t force_html = 0;
267 
268  //testing for special URLS, which should be handled by
269  //redirects or other captive portal logic
270  //URL source: https://stackoverflow.com/questions/46289283/esp8266-captive-portal-with-pop-up
271 
272  if(strcmp(resource,"/fwlink") == 0) { redirect(resource, fd); return; }
273  if(strcmp(resource,"/connecttest.txt") == 0) { redirect(resource, fd); return; }
274  if(strcmp(resource,"/hotspot-detect.html") == 0) { redirect(resource, fd); return; }
275  if(strcmp(resource,"/library/test/success.html") == 0) { redirect(resource, fd); return; }
276  if(strcmp(resource,"/kindle-wifi/wifistub.html") == 0) { redirect(resource, fd); return; }
277 
278  //do NOT redirect for Android, instead serve root file.
279  //if(strcmp(resource,"/generate_204") == 0) { redirect(resource, fd); return; }
280  //if(strcmp(resource,"/gen_204") == 0) { redirect(resource, fd); return; }
281 
282  //if we have a very long filename, rewrite for index.htm...
283  //we had a DoubleExceptionVector for long names (not supported by FAT)
284  if(strlen(resource) > 32) sprintf(resource,"/index.htm");
285 
286  if(xSemaphoreTake(fsSem,200) == pdTRUE)
287  {
288  //basepath + 8.3 file + folder + margin
289  char file[sizeof(base_path)+32];
290 
291  //Captive portal:
292  //for all devices which do NOT want a redirect, serve index file
293  if(strcmp(resource,"/generate_204") == 0) {
294  sprintf(file,"%s%s",base_path,"/index.htm");
295  force_html = 1; //force content type
296  } else if(strcmp(resource,"/gen_204") == 0) {
297  sprintf(file,"%s%s",base_path,"/index.htm");
298  force_html = 1; //force content type
299  } else {
300  sprintf(file,"%s%s",base_path,resource);
301  }
302  ESP_LOGI(LOG_TAG,"serving from FAT: %s",file);
303 
304  // open the file for reading
305  FILE* f = fopen(file, "r");
306  if(f == NULL) {
307  ESP_LOGW(LOG_TAG,"Resource not found: %s, opening index.htm", file);
308  sprintf(file,"%s%s",base_path,"/index.htm");
309  f = fopen(file, "r");
310  force_html = 1; //force content type
311  if(f == NULL)
312  {
313  ESP_LOGE(LOG_TAG,"Index not found? Sending redirect...");
314  send(fd, http_redir_hdr, sizeof(http_redir_hdr) - 1, 0);
315  xSemaphoreGive(fsSem);
316  return;
317  }
318  }
319 
320  //get the size of this file (for html header)
321  fseek(f, 0L, SEEK_END);
322  int sz = ftell(f);
323  rewind(f);
324 
325  //send http header (200)
326  send(fd, http_html_hdr, sizeof(http_html_hdr) - 1,MSG_MORE);
327 
328  //send http header (content type)
329  int reslen = strlen(resource);
330  char hdr[36];
331  if(strcmp(&resource[reslen-4],".css") == 0)
332  {
333  snprintf(hdr,36,"Content-Type: text/css\r\n");
334  } else if(strcmp(&resource[reslen-4],".htm") == 0 || force_html != 0) {
335  snprintf(hdr,36,"Content-Type: text/html\r\n");
336  } else if(strcmp(&resource[reslen-3],".js") == 0) {
337  snprintf(hdr,36,"Content-Type: text/javascript\r\n");
338  } else {
339  snprintf(hdr,36,"Content-Type: text/plain\r\n");
340  }
341  send(fd, hdr, strlen(hdr),MSG_MORE);
342  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
343  ESP_LOGD(LOG_TAG,"Hdr: %s",hdr);
344  #endif
345 
346  //send http header (content length)
347  snprintf(hdr,36,"Content-Length: %d\r\n\r\n",sz);
348  send(fd, hdr, strlen(hdr),MSG_MORE);
349  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
350  ESP_LOGD(LOG_TAG,"Hdr: %s",hdr);
351  #endif
352 
353  //allocate a buffer in size of one FAT block
354  char* buffer = malloc(512);
355  int i=0, len=0;
356  do {
357  len=fread(buffer, 1, 512, f);
358  i+=len;
359  send(fd, buffer, len, 0);
360  vTaskDelay(1);
361  } while (len==512);
362 
363  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
364  ESP_LOGD(LOG_TAG,"Sent %u bytes",i);
365  #endif
366  fclose(f);
367  free(buffer);
368  xSemaphoreGive(fsSem);
369  } else {
370  ESP_LOGE(LOG_TAG,"Timeout waiting for fat mutex!");
371  }
372 }

381 static void http_server_netconn_serve(int fd) {
382  if(fd < 0) return;
383  #define MAX_BUFF_SIZE 512
384 
385  char *buf = malloc(MAX_BUFF_SIZE);
386  if(buf == NULL)
387  {
388  ESP_LOGE(LOG_TAG,"Cannot allocate buf for recv");
389  return;
390  }
391 
392  //read incoming
393  int size = recv(fd,buf,MAX_BUFF_SIZE-1,0);
394  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
395  ESP_LOGD(LOG_TAG,"recv size: %d",size);
396  #endif
397  if(size > 0)
398  {
399  buf[size] = '\0';
400  char *request_line = strtok(buf, "\n");
401 
402  if(request_line) {
403  // default page -> redirect to index.html
404  if(strstr(request_line, "GET / ")) {
405  fat_serve((char*)"/index.htm", fd);
406  } else {
407  // get the requested resource
408  char* method = strtok(request_line, " ");
409  //get rid of "unused" variable.
410  //strtok although needs this call.
411  (void)(method);
412  char* resource = strtok(NULL, " ");
413  fat_serve(resource, fd);
414  }
415  }
416  }
417  free(buf);
418 }

431 static void http_server(void *pvParameters) {
432  socklen_t addr_len;
433  struct sockaddr_in sock_addr;
434  int ret;
435  //we will wait until the flag for WIFI enable request is set.
436  while(((xEventGroupWaitBits(connectionRoutingStatus,WIFI_TO_ACTIVATE,pdTRUE, \
437  pdTRUE, portMAX_DELAY)) & WIFI_TO_ACTIVATE) == 0);
438  //lets start the Wifi stack
439  wifiEnDisable(1);
440 
441  //create a new TCP connection
442  http_sockfd = socket(AF_INET, SOCK_STREAM, 0);
443  if (http_sockfd < 0) {
444  ESP_LOGE(LOG_TAG,"Cannot create socket");
445  }
446  //bind it to incoming port 80 (HTTP)
447  memset(&sock_addr, 0, sizeof(sock_addr));
448  sock_addr.sin_family = AF_INET;
449  sock_addr.sin_addr.s_addr = 0;
450  sock_addr.sin_port = htons(80);
451  ret = bind(http_sockfd, (struct sockaddr*)&sock_addr, sizeof(sock_addr));
452  if(ret) {
453  ESP_LOGE(LOG_TAG,"Failed to bind: %d",ret);
454  close(http_sockfd);
455  http_sockfd = -1;
456  vTaskDelete(NULL);
457  }
458  ret = listen(http_sockfd, 1);
459  if(ret) {
460  ESP_LOGE(LOG_TAG,"Failed to listen: %d",ret);
461  close(http_sockfd);
462  http_sockfd = -1;
463  vTaskDelete(NULL);
464  }
465  ESP_LOGI(LOG_TAG,"http_server task started");
466 
467  do {
468  //accept connection & save netconn handle
469  http_new_sockfd = accept(http_sockfd, (struct sockaddr *)&sock_addr, &addr_len);
470  if (http_new_sockfd < 0) {
471  ESP_LOGE(LOG_TAG, "Failed to accept: %d",http_new_sockfd);
472  } else {
473  //serve
474  http_server_netconn_serve(http_new_sockfd);
475  //close
476  close(http_new_sockfd);
477  }
478  vTaskDelay(1); //allows task to be pre-empted
479  } while(1);
480  //if we run into an error or this task is going to be removed
481  close(http_new_sockfd);
482  close(http_sockfd);
483  //delete task
484  vTaskDelete(NULL);
485 }

502 static esp_err_t wifi_event_handler(void *ctx, system_event_t *event) {
503  //check if event flags are active
504  if(connectionRoutingStatus == NULL)
505  {
506  ESP_LOGE(LOG_TAG,"Status flags for wifi are NULL! Should not receive an event!");
507  return ESP_FAIL;
508  }
509 
510  //determine event: print information and set flags accordingly
511  switch(event->event_id) {
512 
513  case SYSTEM_EVENT_STA_START:
514  ESP_LOGE(LOG_TAG,"Shit!");
515  esp_wifi_connect();
516  break;
517 
518  case SYSTEM_EVENT_AP_START:
519  ESP_LOGD(LOG_TAG,"Access point started");
520  xEventGroupSetBits(connectionRoutingStatus, WIFI_ACTIVE);
521  //start off timer (switch off wifi, if no client connects in timeout)
522  wifiStartStopOffTimer(1);
523  break;
524 
525  case SYSTEM_EVENT_AP_STOP:
526  ESP_LOGD(LOG_TAG,"Access point stopped");
527  xEventGroupClearBits(connectionRoutingStatus, WIFI_ACTIVE);
528  break;
529 
530  case SYSTEM_EVENT_AP_STACONNECTED:
531  //a new client connected
532  ESP_LOGI(LOG_TAG,"Client connected, currently connected: %d",getNumberOfWifiStations());
533  //set client connected flag
534  xEventGroupSetBits(connectionRoutingStatus, WIFI_CLIENT_CONNECTED);
535 
536  //if the wifi timer is active, disable the timer (client is connected)
537  wifiStartStopOffTimer(0);
538  break;
539 
540  case SYSTEM_EVENT_AP_STADISCONNECTED:
541  //a client got disconnected
542  ESP_LOGI(LOG_TAG,"Client disconnected, currently connected: %d",getNumberOfWifiStations());
543  if(getNumberOfWifiStations() == 0)
544  {
545  //remove the callback, not necessary if no client is connected
546  halSerialRemoveOutputStream();
547  //clear client connected flag
548  xEventGroupClearBits(connectionRoutingStatus, WIFI_CLIENT_CONNECTED);
549  //start the auto-disable timer
550  wifiStartStopOffTimer(1);
551  }
552  break;
553  default:
554  break;
555  }
556  return ESP_OK;
557 }

575 esp_err_t taskWebGUIEnDisable(int onoff, bool fromISR)
576 {
577  //check first if wifi was already used...
578  EventBits_t status;
579  if(fromISR) status = xEventGroupGetBits(connectionRoutingStatus);
580  else status = xEventGroupGetBitsFromISR(connectionRoutingStatus);
581  if((status & WIFI_LOCKED) != 0)
582  {
583  ESP_LOGW(LOG_TAG,"Wifi can be enabled/disabled only once each powercycle!");
584  return ESP_FAIL;
585  }
586 
587  BaseType_t xHigherPriorityTaskWoken, xResult;
588  if(onoff != 0)
589  {
590  if(fromISR)
591  {
592  xHigherPriorityTaskWoken = pdFALSE;
593  xResult = xEventGroupSetBitsFromISR(connectionRoutingStatus, \
594  WIFI_TO_ACTIVATE, &xHigherPriorityTaskWoken);
595  /* Was the message posted successfully? */
596  if(xResult != pdFAIL && xHigherPriorityTaskWoken) portYIELD_FROM_ISR();
597  } else {
598  //en- /disable wifi in non-ISR mode
599  xEventGroupSetBits(connectionRoutingStatus,WIFI_TO_ACTIVATE);
600  }
601  } else {
602  //disable wifi by killing all that stuff...
603  wifiEnDisable(0);
604  }
605  return ESP_OK;
606 }

627 esp_err_t wifiEnDisable(int onoff)
628 {
629  esp_err_t ret;
630  //should we enable or disable wifi?
631  if(onoff == 0)
632  {

634  xEventGroupClearBits(connectionRoutingStatus, WIFI_ACTIVE | WIFI_CLIENT_CONNECTED);
635

638  if(wifiHTTPServerHandle_t != NULL) vTaskDelete(wifiHTTPServerHandle_t);
639  if(wifiWSServerHandle_t != NULL) vTaskDelete(wifiWSServerHandle_t);
640  //close remaining sockets
641  close(http_new_sockfd);
642  close(http_sockfd);
643  netconn_close(ws_conn);
644 
645  //signal an already used wifi
646  xEventGroupSetBits(connectionRoutingStatus,WIFI_LOCKED);
647 
648  //disable, call wifi_stop
649  if(esp_wifi_stop() == ESP_OK)
650  {
651  ret = esp_wifi_deinit();
652  } else {
653  ret = ESP_OK;
654  }
655  //check return value
656  if(ret != ESP_OK)
657  {
658  ESP_LOGE(LOG_TAG,"Please initialize WiFi prior to disable it!");
659  return ESP_FAIL;
660  } else {
661  return ESP_OK;
662  }
663  } else {
664  //based on softAP example of Espressif.
665  tcpip_adapter_init();
666  //if event loop init fails, we might have an already initialized event loop
667  //in this case, just add the handler of this file.

682  };

684  esp_wifi_set_config(ESP_IF_WIFI_AP, &wifi_config);
685  ret = esp_wifi_start();
686 
687  ESP_LOGI(LOG_TAG,"AP - PW: %s (%d)",wifi_config.ap.password,strnlen(wifipw,32));
688  ESP_LOGI(LOG_TAG,"AP - SSID: %s (%d)",wifi_config.ap.ssid, wifi_config.ap.ssid_len);
689 
690  //check esp_wifi_start's return value
691  switch(ret)
692  {
693  //everything ok
694  case ESP_OK: break;
695  //determine error
696  case ESP_ERR_WIFI_CONN:
697  ESP_LOGE(LOG_TAG,"Wifi internal error, control block invalid");
698  return ESP_FAIL;
699  case ESP_ERR_NO_MEM:
700  ESP_LOGE(LOG_TAG,"Wifi internal error, out of memory");
701  return ESP_FAIL;
702  case ESP_ERR_WIFI_NOT_INIT:
703  ESP_LOGE(LOG_TAG,"Please initialize WiFi prior to enable/disable it!");
704  return ESP_FAIL;
705  case ESP_FAIL:
706  case ESP_ERR_INVALID_ARG:
707  default:
708  ESP_LOGE(LOG_TAG,"Unknown internal Wifi error");
709  return ESP_FAIL;
710  }
711 
712  //wait 200ms for WiFi stack to settle
713  vTaskDelay(200/portTICK_PERIOD_MS);
714 
715  /*++++ initialize capitive portal dns server ++++*/
716  captdnsInit();
717 
718  /*++++ start Websocket task ++++*/
719  xTaskCreate(&ws_server, "ws_server", TASK_WEBGUI_WEBSOCKET_STACKSIZE, NULL, 5, &wifiWSServerHandle_t);
720 
721  return ESP_OK;
722  }
