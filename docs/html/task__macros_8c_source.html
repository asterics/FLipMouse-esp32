50 void task_macro(taskMacrosConfig_t *param)
51 {
52  //check for param struct
53  if(param == NULL)
54  {
55  ESP_LOGE(LOG_TAG,"param is NULL ");
56  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
57  return;
58  }
59  //event bits used for pending on debounced buttons
60  EventBits_t uxBits = 0;
61  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
62  uint8_t evGroupIndex = param->virtualButton / 4;
63  //calculate bitmask offset within the EventGroup
64  uint8_t evGroupShift = param->virtualButton % 4;
65  //save virtualbutton
66  uint vb = param->virtualButton;
67  //final pointer to the EventGroup used by this task
68  EventGroupHandle_t *evGroup = NULL;
69  //local copy of macro (much safer, caller might free the config)
70  char macro[SLOTNAME_LENGTH];
71  strcpy(macro,param->macro);
72 
73  //do all the eventgroup checking only if this is a persistent task
74  //-> virtualButton is NOT set to VB_SINGLESHOT
75  if(param->virtualButton != VB_SINGLESHOT)
76  {
77  //check for correct offset
78  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
79  {
80  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
81  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
82  return;
83  }
84 
85  //test if event groups are already initialized, otherwise exit immediately
86  while(virtualButtonsOut[evGroupIndex] == 0)
87  {
88  ESP_LOGE(LOG_TAG,"uninitialized event group, retry in 1s");
89  vTaskDelay(1000/portTICK_PERIOD_MS);
90  }
91 
92  //test if event groups are already initialized, otherwise exit immediately
93  while(halSerialATCmds == NULL)
94  {
95  ESP_LOGE(LOG_TAG,"uninitialized event group, retry in 1s");
96  vTaskDelay(1000/portTICK_PERIOD_MS);
97  }
98  //event group intialized, store for later usage
99  evGroup = virtualButtonsOut[evGroupIndex];
100  }
101 
102  while(1)
103  {
104  //wait for the flag
105  if(vb != VB_SINGLESHOT)
106  {
107  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift)|(1<<(evGroupShift+4)),pdTRUE,pdFALSE,1000);
108  }
109  //test for a valid set flag or trigger if we are using singleshot
110  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
111  {
112  int offset = 0;
113  int start = 0;
114 
115 
116  while(offset < SLOTNAME_LENGTH)
117  {
118  //end loop if terminators are detected.
119  if(macro[offset] == '\r' || macro[offset] == '\n') break;
120 
121  //do we reach a command terminator?
122  if(macro[offset] == ';')
123  {
124  //check if this character is escaped:
125  if(offset > 0)
126  {
127  if(macro[offset-1] == '\\')
128  {
129  //if it escaped "\;", just continue.
130  offset++;
131  continue;
132  }
133  }
134 
135  //check if we hit an AT WA (wait)
136  if(memcmp(&macro[start],"AT WA",5) == 0)
137  {
138  //if yes, delay this task.
139  uint32_t time = strtol((char*)&(macro[start+6]),NULL,10);
140  if(time < 30000)
141  {
142  vTaskDelay(time / portTICK_PERIOD_MS);
143  } else {
144  ESP_LOGE(LOG_TAG,"Hit AT WA with a delay time too high: %d",time);
145  }
146  } else {
147  //if not an AT WA, allocate memory and send to queue.
148  atcmd_t command;
149  uint16_t length = offset-start;
150  uint8_t *buffer = malloc(sizeof(uint8_t)*(length+1));
151  if(buffer != NULL)
152  {
153  //copy data
154  memcpy(buffer,&macro[start],length);
155  //terminate
156  buffer[length] = 0;
157  //save to queue struct
158  command.buf = buffer;
159  command.len = length;
160  ESP_LOGD(LOG_TAG,"Sent AT cmd: %s",buffer);
161  //send to queue, wait maximum 10 ticks (100ms) for a free space.
162  if(xQueueSend(halSerialATCmds,(void*)&command,10) != pdTRUE)
163  {
164  ESP_LOGE(LOG_TAG,"Cmd queue is full, cannot send command");
165  }
166  //we do NOT need a taskYIELD here, because ESP32 is running with preemption:
167  //https://www.freertos.org/a00020.html#taskYIELD
168  //taskYIELD();
169  } else {
170  ESP_LOGE(LOG_TAG,"Cannot allocate memory for command!");
171  }
172  }
173 
174  //save new start position for next command
175  start = offset + 1;
176  }
177 
178  //go to next character
179  offset++;
180  }
181  }
182 
183  //function tasks in single shot mode MUST return to its caller
184  if(vb == VB_SINGLESHOT) return;
185  }
186 }
