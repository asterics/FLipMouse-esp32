35 static void gpio_isr_handler(void* arg)
36 {
37  uint32_t pin = (uint32_t) arg;
38  uint8_t vb = 0;
39  BaseType_t xResult, xHigherPriorityTaskWoken = pdFALSE;
40 
41  switch(pin)
42  {
43  case HAL_IO_PIN_BUTTON_EXT1: vb = VB_EXTERNAL1; break;
44  case HAL_IO_PIN_BUTTON_EXT2: vb = VB_EXTERNAL2; break;
45  case HAL_IO_PIN_BUTTON_INT1: vb = VB_INTERNAL1; break;
46  case HAL_IO_PIN_BUTTON_INT2: vb = VB_INTERNAL2; break;
47  default: return;
48  }
49 
50  if(gpio_get_level(pin) == 0)
51  {
52  //set press flag
53  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)),&xHigherPriorityTaskWoken);
54  //clear release flag
55  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)));
56  } else {
57  //set release flag
58  xResult = xEventGroupSetBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4 + 4)),&xHigherPriorityTaskWoken);
59  //clear press flag
60  xResult = xEventGroupClearBitsFromISR(virtualButtonsIn[vb/4],(1<<(vb%4)));
61  }
62  if(xResult == pdPASS) portYIELD_FROM_ISR();
63 }

72 esp_err_t halIOInit(void)
73 {
74  /*++++ init GPIO interrupts for 2 external & 2 internal buttons ++++*/
75  gpio_config_t io_conf;
76  //disable pull-down mode
77  io_conf.pull_down_en = 0;
78  //interrupt of rising edge
79  io_conf.intr_type = GPIO_PIN_INTR_ANYEDGE;
80  //bit mask of the pins
81  io_conf.pin_bit_mask = (1<<HAL_IO_PIN_BUTTON_EXT1) | (1<<HAL_IO_PIN_BUTTON_EXT2) | \
82  (1<<HAL_IO_PIN_BUTTON_INT1) | (1<<HAL_IO_PIN_BUTTON_INT2);
83  //set as input mode
84  io_conf.mode = GPIO_MODE_INPUT;
85  //enable pull-up mode
86  io_conf.pull_up_en = 1;
87  gpio_config(&io_conf);
88 
89  //TODO: ret vals pr√ºfen

91  //install gpio isr service
