120 static void gpio_isr_handler(void* arg)
121 {
122  uint32_t pin = (uint32_t) arg;
123  uint8_t vb = 0;
124  raw_action_t evt;
125  BaseType_t xHigherPriorityTaskWoken = pdFALSE;
126 
127  //determine pin of ISR reason
128  switch(pin)
129  {
130  case HAL_IO_PIN_BUTTON_EXT1: vb = VB_EXTERNAL1; break;
131  case HAL_IO_PIN_BUTTON_EXT2: vb = VB_EXTERNAL2; break;
132  #ifdef DEVICE_FABI
133  case HAL_IO_PIN_BUTTON_EXT3: vb = VB_EXTERNAL3; break;
134  case HAL_IO_PIN_BUTTON_EXT4: vb = VB_EXTERNAL4; break;
135  case HAL_IO_PIN_BUTTON_EXT5: vb = VB_EXTERNAL5; break;
136  case HAL_IO_PIN_BUTTON_EXT6: vb = VB_EXTERNAL6; break;
137  case HAL_IO_PIN_BUTTON_EXT7: vb = VB_EXTERNAL7; break;
138  #endif
139  case HAL_IO_PIN_BUTTON_INT1: vb = VB_INTERNAL1; break;
140  #ifdef DEVICE_FLIPMOUSE
141  case HAL_IO_PIN_BUTTON_INT2: vb = VB_INTERNAL2; break;
142  #endif
143  default: return;
144  }
145 
146  //set VB number to event.
147  evt.vb = vb;
148  if(debouncer_in == NULL) return;
149 
150  //press or release?
151  if(gpio_get_level(pin) == 0)
152  {
153  //set an press event
154  evt.type = VB_PRESS_EVENT;
155  //send event
156  xQueueSendToBackFromISR(debouncer_in,&evt,&xHigherPriorityTaskWoken);
157  //extra handling for long press
158  if(pin == HAL_IO_PIN_LONGACTION)
159  {
160  //check if timer is initialized, if yes reset & start
161  if(longactiontimer != NULL)
162  {
163  xTimerResetFromISR(longactiontimer,&xHigherPriorityTaskWoken);
164  xTimerStartFromISR(longactiontimer,&xHigherPriorityTaskWoken);
165  }
166  }
167  } else {
168  //set an press event
169  evt.type = VB_RELEASE_EVENT;
170  //send event
171  xQueueSendToBackFromISR(debouncer_in,&evt,&xHigherPriorityTaskWoken);
172  //extra handling for long press
173  if(pin == HAL_IO_PIN_LONGACTION)
174  {
175  //check if timer is initialized, if yes reset & start
176  if(longactiontimer != NULL) xTimerStopFromISR(longactiontimer,&xHigherPriorityTaskWoken);
177  }
178  }
179  if(xHigherPriorityTaskWoken) portYIELD_FROM_ISR();
180 }

207 void halIOIRRecvTask(void * param)
208 {
209  halIOIR_t* recv;
210  RingbufHandle_t rb = NULL;
211  //get RMT RX ringbuffer
212  rmt_get_ringbuf_handle(4, &rb);
213 
214  if(halIOIRRecvQueue == NULL)
215  {
216  ESP_LOGW(LOG_TAG, "halIOIRRecvQueue not initialised");
217  while(halIOIRRecvQueue == NULL) vTaskDelay(1000/portTICK_PERIOD_MS);
218  }
219 
220  while(1)
221  {
222  //wait for updates (triggered receiving)
223  if(xQueueReceive(halIOIRRecvQueue,&recv,portMAX_DELAY))
224  {
225  ESP_LOGI(LOG_TAG,"IR recv triggered.");
226 
227  //acquire buffer for ALL possible receiving elements
228  uint16_t offset = 0;
229  size_t rx_size = 0;
230  uint16_t local_timeout = 0;
231 
232  //check buffer
233  if(recv->buffer == NULL)
234  {
235  ESP_LOGE(LOG_TAG,"Please provide a buffer for receiving IR!");
236  continue;
237  }
238 
239  //start receiving on channel 4, flush all buffer elements
240  rmt_rx_start(4, 1);
241  //set target struct
242  recv->status = IR_RECEIVING;
243 
244  do{
245  //wait for one item until timeout or data is valid
246  rmt_item32_t* item = (rmt_item32_t*) xRingbufferReceive(rb, &rx_size, TASK_HAL_IR_RECV_EDGE_TIMEOUT/portTICK_PERIOD_MS);
247  //got one item
248  if(item != NULL) {
249  //put data into buffer
250  memcpy(&recv->buffer[offset], item, sizeof(rmt_item32_t)*rx_size);
251 
252  //invert active mode by inverting bits level0 and level1 in each item
253  for(size_t i = 0; i<rx_size;i++)
254  {
255  recv->buffer[offset+i].level0 = !recv->buffer[offset+i].level0;
256  recv->buffer[offset+i].level1 = !recv->buffer[offset+i].level1;
257  }
258  //increase offset for next received item
259  offset+= rx_size;
260 
261  //give item back to ringbuffer
262  vRingbufferReturnItem(rb, (void*) item);
263  //too much
264  if(offset >= TASK_HAL_IR_RECV_MAXIMUM_EDGES)
265  {
266  ESP_LOGE(LOG_TAG,"Too much IR edges, finished");
267  recv->status = IR_OVERFLOW;
268  recv->count = 0;
269  break;
270  }
271  } else {
272  local_timeout += TASK_HAL_IR_RECV_EDGE_TIMEOUT;
273  //check if timeout is the long one and no edges received...
274  if(local_timeout >= TASK_HAL_IR_RECV_TIMEOUT && offset == 0)
275  {
276  //timeout, cancel
277  rmt_rx_stop(4);
278  recv->status = IR_TOOSHORT;
279  recv->count = 0;
280  ESP_LOGE(LOG_TAG,"No cmd");
281  break;
282  }
283  //if we received already something and timeout triggers ->
284  //command finished
285  if(offset != 0)
286  {
287  //update status accordingly
288  if(offset > TASK_HAL_IR_RECV_MINIMUM_EDGES)
289  {
290  //save everything necessary to pointer from queue
291  recv->count = offset;
292  recv->status = IR_FINISHED;
293  ESP_LOGI(LOG_TAG,"Recorded @%d %d edges",(uint32_t)recv->buffer,offset);
294  } else {
295  //timeout, cancel
296  recv->status = IR_TOOSHORT;
297  recv->count = 0;
298  ESP_LOGE(LOG_TAG,"IR cmd too short");
299  }
300  rmt_rx_stop(4);
301  break;
302  }
303  }
304  } while(1);
305  }
306  }
