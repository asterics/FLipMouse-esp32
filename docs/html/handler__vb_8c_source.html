71 static void handler_vb(void *event_handler_arg, esp_event_base_t event_base, int32_t event_id, void *event_data)
72 {
73  //if we don't have a stable config, simply return...
74  if((xEventGroupGetBits(systemStatus) & SYSTEM_STABLECONFIG) == 0) return;
75  //still commands to be processed, shouldn't continue
76  if((xEventGroupGetBits(systemStatus) & SYSTEM_EMPTY_CMD_QUEUE) == 0) return;
77 
78  //use the mutex to ensure a valid chained list.
79  if(xSemaphoreTake(vbCmdSem,4) != pdTRUE)
80  {
81  ESP_LOGW(LOG_TAG,"VB mutex not free for handler");
82  return;
83  }
84  //if command chain is empty, we cannot do anything.
85  if(cmd_chain == NULL)
86  {
87  xSemaphoreGive(vbCmdSem);
88  return;
89  }
90 
91  uint8_t vb = 0;
92  uint32_t count = 0;
93  switch(event_id)
94  {
95  case VB_PRESS_EVENT: vb |= 0x80; //set uppermost bit for press event.
96  case VB_RELEASE_EVENT:
97  break;
98  default: //might be another type of event, we don't care of.
99  return;
100  }
101 
102  //we need an VB number as event data.
103  if(event_data == 0)
104  {
105  ESP_LOGE(LOG_TAG,"Empty event data, cannot proceed!");
106  return;
107  }
108 
109  vb |= (*((uint32_t*) event_data)) & 0x7F;
110 
111  //begin with head of chain
112  vb_cmd_t *current = cmd_chain;
113  //iterate through all available vb cmds
114  while(current != NULL)
115  {
116  //send VB command(s), if VB matches
117  //this way, we can do more actions on one VB
118  if(current->vb == vb)
119  {
120  count++;
121  /* determine action to be triggered */
122  switch(current->cmd)
123  {
124  case T_MACRO:
125  if(current->cmdparam == NULL)
126  {
127  ESP_LOGE(LOG_TAG,"Param is null, cannot execute macro");
128  } else {
129  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
130  ESP_LOGD(LOG_TAG,"Sent macro %s for VB %d", (char*)current->cmdparam, vb & 0x7F);
131  #endif
132  fct_macro(current->cmdparam);
133  }
134  break;
135  case T_CONFIGCHANGE:
136  if(current->cmdparam == NULL)
137  {
138  ESP_LOGE(LOG_TAG,"Param is null, cannot request config change");
139  } else {
140  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
141  ESP_LOGD(LOG_TAG,"Sent CFG change %s for VB %d", (char*)current->cmdparam, vb & 0x7F);
142  #endif
143  xQueueSend(config_switcher,(void*)current->cmdparam,(TickType_t)10);
144  }
145  break;
146  case T_CALIBRATE:
147  halAdcCalibrate();
148  break;
149  case T_SENDIR:
150  if(current->cmdparam == NULL)
151  {
152  ESP_LOGE(LOG_TAG,"Param is null, cannot send IR");
153  } else {
154  fct_infrared_send(current->cmdparam);
155  }
156  break;
157  default:
158  ESP_LOGE(LOG_TAG,"Unknown VB cmd type");
159  break;
160  }
161  }
162  current = current->next;
163  }
164  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
165  if(count == 0) ESP_LOGD(LOG_TAG,"Sent %d cmds for VB %d", count, vb & 0x7F);
166  #endif
167  if(count != 0) ESP_LOGI(LOG_TAG,"Sent %d cmds for VB %d", count, vb & 0x7F);
168  xSemaphoreGive(vbCmdSem);
169 }

175 esp_err_t handler_vb_init(void)
176 {
177  //init VB mutex
178  if(vbCmdSem != NULL) vSemaphoreDelete(vbCmdSem);
179  vbCmdSem = xSemaphoreCreateMutex();
180  if(vbCmdSem == NULL)
181  {
182  ESP_LOGE(LOG_TAG,"Cannot create mutex, exiting!");
183  return ESP_FAIL;
184  }
185  xSemaphoreGive(vbCmdSem);
186  //set log level to given log level
187  esp_log_level_set(LOG_TAG,LOG_LEVEL_VB);
188 
189  return esp_event_handler_register(VB_EVENT,ESP_EVENT_ANY_ID,handler_vb,NULL);
190 }

199 esp_err_t handler_vb_delCmd(uint8_t vb)
200 {
201  //existing chain
202  vb_cmd_t *current = cmd_chain;
203  vb_cmd_t *prev = NULL;
204  uint count = 0;
205 
206  heap_caps_check_integrity_all(true);
207  //do as long as we don't have a null pointer
208  while(current != NULL)
209  {
210  //if the VB number matches (discarding press/release flag)
211  if((current->vb & 0x7F) == (vb & 0x7F))
212  {
213  //set pointer from previous element to next one
214  //but only if we are not at the head (no previous element)
215  if(prev != NULL) prev->next = current->next;
216  //if no previous element -> replace
217  else cmd_chain = current->next;
218  //free an AT string
219  if(current->atoriginal != NULL) free(current->atoriginal);
220  //if set, free param string
221  if(current->cmdparam != NULL) free(current->cmdparam);
222  //free this element
223  free(current);
224  //just begin at the front again (easiest way if we removed the head)
225  current = cmd_chain;
226  prev = NULL;
227  if((vb & 0x7F) <= 63) vb_active &= ~(1<<(vb & 0x7F)); //delete active flag
228  } else { //does not match
229  //set pointers to next element
230  prev = current;
231  current = current->next;
232  }
233  }
234  if(count != 0) return ESP_OK;
235  else return ESP_FAIL;
236 }

249 esp_err_t handler_vb_addCmd(vb_cmd_t *newCmd, uint8_t replace)
250 {
251  //sanitizing...
252  if(newCmd == NULL)
253  {
254  ESP_LOGE(LOG_TAG,"newCmd is NULL");
255  return ESP_FAIL;
256  }
257  if(vbCmdSem == NULL)
258  {
259  ESP_LOGE(LOG_TAG,"vbCmdSem is NULL");
260  return ESP_FAIL;
261  }
262  if((newCmd->vb & 0x7F) >= VB_MAX)
263  {
264  ESP_LOGE(LOG_TAG,"newCmd->vb out of range");
265  return ESP_FAIL;
266  }
267 
268  //take mutex for modifying
269  if(xSemaphoreTake(vbCmdSem,50) != pdTRUE)
270  {
271  ESP_LOGE(LOG_TAG,"VB mutex not free for adding");
272  return ESP_FAIL;
273  }
274 
275  //existing chain, add to end
276  vb_cmd_t *current = cmd_chain;
277  int count = 0;
278 
279  //debugging...
280  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
281  while(current!=NULL)
282  {
283  count++;
284  ESP_LOGD(LOG_TAG,"%d:%2d:0x%X",count,current->vb,(uint32_t)current);
285  current = current->next;
286  }
287  current = cmd_chain;
288  count = 0;
289  #endif
290 
291  //if set, remove any previously set commands.
292  if(replace) handler_vb_delCmd(newCmd->vb);
293 
294  //allocate new command
295  current = cmd_chain;
296  vb_cmd_t *new;
297  new = malloc(sizeof(vb_cmd_t));
298 
299  if(new != NULL)
300  {
301  memcpy(new, newCmd, sizeof(vb_cmd_t));
302  //save pointer of new config to end of chain
303  new->next = NULL;
304  //if we don't have a head, use this one
305  if(cmd_chain == NULL) {
306  cmd_chain = new;
307  } else {
308  //otherwise append to tail.
309  while(current->next != NULL)
310  {
311  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
312  ESP_LOGD(LOG_TAG,"Nr %d, @0x%8X",count,(uint32_t)current);
313  #endif
314  count++;
315  current = current->next;
316  }
317  current->next = new;
318  }
319  if((new->vb & 0x7F) <= 63) vb_active |= (1<<(new->vb & 0x7F)); //set active flag
320  count++;
321  #if LOG_LEVEL_VB >= ESP_LOG_DEBUG
322  ESP_LOGI(LOG_TAG,"Added new cmd nr %d, new: 0x%8X, prev: 0x%8X",count,(uint32_t)new,(uint32_t)current);
323  #endif
324 
325  current = new;
326  xSemaphoreGive(vbCmdSem);
327  return ESP_OK;
328  } else {
329  ESP_LOGE(LOG_TAG,"Cannot allocate memory for new VB cmd!");
330  xSemaphoreGive(vbCmdSem);
331  return ESP_FAIL;
332  }
333 
334  //should not be here, but return ok anyway
335  xSemaphoreGive(vbCmdSem);
336  return ESP_OK;
337 }

354 esp_err_t handler_vb_setCmdChain(vb_cmd_t *chain)
355 {
356  esp_err_t ret = ESP_OK;
357 
358  //check if chain is currently active,
359  //if yes clear it before setting a new one
360  if(cmd_chain != NULL)
361  {
362  ret = handler_vb_clearCmds();
363  if(ret != ESP_OK)
364  {
365  ESP_LOGE(LOG_TAG,"Cannot clear old chain");
366  return ret;
367  }
368  }
369 
370  //enter critical section for setting new cmd chain
371  if(vbCmdSem == NULL)
372  {
373  ESP_LOGE(LOG_TAG,"vbCmdSem is NULL");
374  return ESP_FAIL;
375  }
376  //take mutex for modifying
377  if(xSemaphoreTake(vbCmdSem,50) != pdTRUE)
378  {
379  ESP_LOGE(LOG_TAG,"cannot enter critical section");
380  return ESP_FAIL;
381  }
382  //save the pointer
383  cmd_chain = chain;
384  //release mutex
385  xSemaphoreGive(vbCmdSem);
386  return ESP_OK;

395 esp_err_t handler_vb_clearCmds(void)
396 {
397  if(cmd_chain == NULL)
398  {
399  ESP_LOGW(LOG_TAG,"VB cmds already empty");
400  return ESP_FAIL;
401  }
402  if(vbCmdSem == NULL)
403  {
404  ESP_LOGE(LOG_TAG,"vbCmdSem is NULL");
405  return ESP_FAIL;
406  }
407 
408  //take mutex for modifying
409  if(xSemaphoreTake(vbCmdSem,50) != pdTRUE)
410  {
411  ESP_LOGE(LOG_TAG,"VB mutex not free for clearing");
412  return ESP_FAIL;
413  }
414 
415  //pointers for next and current command
416  vb_cmd_t *next = NULL;
417  vb_cmd_t *current = cmd_chain;
418  int count = 0;
419 
420  do {
421  //load next block
422  next = current->next;
423  //if set, free the original AT command
424  if(current->atoriginal != NULL) free(current->atoriginal);
425  current->atoriginal = NULL;
426  //if set, free param string
427  if(current->cmdparam != NULL) free(current->cmdparam);
428  current->cmdparam = NULL;
429  //free the current one
430  free(current);
431  //count for statistics
432  count++;
433  //previous next is current for next while iteration
434  current = next;
435  //break the loop if current is NULL (we reached end of chain)
436  } while(current != NULL);
437 
438  #if LOG_LEVEL_VB >= ESP_LOG_INFO
439  ESP_LOGI(LOG_TAG,"Cleared %d VB cmds",count);
440  #endif
441 
442  cmd_chain = NULL;
443  vb_active = 0;
444  //release mutex
445  xSemaphoreGive(vbCmdSem);
446  return ESP_OK;
447 }

457 esp_err_t handler_vb_getAT(char* output, uint8_t vb)
458 {
459  if(vbCmdSem == NULL)
460  {
461  ESP_LOGE(LOG_TAG,"vbCmdSem is NULL");
462  return ESP_FAIL;
463  }
464 
465  if(cmd_chain == NULL)
466  {
467  ESP_LOGE(LOG_TAG,"Chain empty!");
468  return ESP_FAIL;
469  }
470  //take mutex for reading
471  if(xSemaphoreTake(vbCmdSem,50) != pdTRUE)
472  {
473  ESP_LOGE(LOG_TAG,"VB mutex not free for getting");
474  return ESP_FAIL;
475  }
476 
477  //pointers for current command
478  vb_cmd_t *current = cmd_chain;
479 
480  //iterate through all available vb cmds
481  while(current != NULL)
482  {
483  //is this the requested button?
484  if((current->vb & 0x7F) == vb)
485  {
486  //check if we found an AT string
487  if(current->atoriginal != NULL)
488  {
489  strncpy(output,current->atoriginal,ATCMD_LENGTH);
490  #if LOG_LEVEL_VB >= ESP_LOG_INFO
491  ESP_LOGI(LOG_TAG,"BM%02d: %s",vb,output);
492  #endif
493  xSemaphoreGive(vbCmdSem);
494  return ESP_OK;
495  }
496  }
497  current = current->next;
498  }
499  ESP_LOGD(LOG_TAG,"No AT command found");
500  xSemaphoreGive(vbCmdSem);
501  return ESP_FAIL;
502 }

512 bool handler_vb_active(uint8_t vb)
513 {
514  //we check for VB_MAX (firmware specfic) and 63 (size of vb_active)
515  if(vb >= VB_MAX || vb >= 63)
516  {
517  ESP_LOGE(LOG_TAG,"Cannot detect state of VB %d, out of range!",vb);
518  return false;
519  } else {
520  if((vb_active & (1<<vb)) != 0) return true;
521  else return false;
522  }
523  return false;
524 }
