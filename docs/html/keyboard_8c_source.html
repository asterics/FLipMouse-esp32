2054  static uint16_t unicode_wchar = 0;

2056  static uint8_t deadkey = 0;
2057 
2058  //avoid accessing arrays out of bound
2059  if(locale >= LAYOUT_MAX) return 0;
2060 
2061  if (inputdata < 0x80) {
2062  // single byte encoded, 0x00 to 0x7F
2063  utf8_state = 0;
2064  keycode = unicode_to_keycode(inputdata, locale);
2065  deadkey = deadkey_to_keycode(keycode,locale);
2066  *keycode_modifier = keycode_to_modifier(keycode,locale);

2068  else *deadkey_first_keycode = 0;
2069  return keycode_to_key(keycode & keycodes_masks[locale][3]);
2070  } else if (inputdata < 0xC0) {
2071  // 2nd, 3rd or 4th byte, 0x80 to 0xBF
2072  inputdata &= 0x3F;
2073  if (utf8_state == 1) {
2074  utf8_state = 0;
2075  keycode = unicode_to_keycode(unicode_wchar | inputdata, locale);
2076  deadkey = deadkey_to_keycode(keycode,locale);
2077  *keycode_modifier = keycode_to_modifier(keycode,locale);

2079  else *deadkey_first_keycode = 0;
2080  return keycode_to_key(keycode & keycodes_masks[locale][3]);
2081  } else if (utf8_state == 2) {
2082  unicode_wchar |= ((uint16_t)inputdata << 6);
2083  utf8_state = 1;
2084  }
2085  } else if (inputdata < 0xE0) {
2086  // begin 2 byte sequence, 0xC2 to 0xDF
2087  // or illegal 2 byte sequence, 0xC0 to 0xC1
2088  unicode_wchar = (uint16_t)(inputdata & 0x1F) << 6;
2089  utf8_state = 1;
2090  } else if (inputdata < 0xF0) {
2091  // begin 3 byte sequence, 0xE0 to 0xEF
2092  unicode_wchar = (uint16_t)(inputdata & 0x0F) << 12;
2093  utf8_state = 2;
2094  } else {
2095  // begin 4 byte sequence (not supported), 0xF0 to 0xF4
2096  // or illegal, 0xF5 to 0xFF
2097  utf8_state = 255;
2098  }
2099  return 0;
2100 }
