192 void fat_serve(char* resource, int fd) {
193 
194  if(xSemaphoreTake(fatSem,200) == pdTRUE)
195  {
196 
197  //basepath + 8.3 file + folder + margin
198  char file[sizeof(base_path)+32];
199 
200  sprintf(file,"%s%s",base_path,resource);
201  ESP_LOGI(LOG_TAG,"serving from FAT: %s",file);
202 
203  // open the file for reading
204  FILE* f = fopen(file, "r");
205  if(f == NULL) {
206  ESP_LOGW(LOG_TAG,"Resource not found: %s, sending redirect", file);
207  send(fd, http_redir_hdr, sizeof(http_redir_hdr) - 1, 0);
208  xSemaphoreGive(fatSem);
209  return;
210  }
211 
212  //get the size of this file (for html header)
213  fseek(f, 0L, SEEK_END);
214  int sz = ftell(f);
215  rewind(f);
216 
217  //send http header (200)
218  send(fd, http_html_hdr, sizeof(http_html_hdr) - 1,MSG_MORE);
219 
220  //send http header (content type)
221  int reslen = strlen(resource);
222  char hdr[36];
223  if(strcmp(&resource[reslen-4],".css") == 0)
224  {
225  snprintf(hdr,36,"Content-Type: text/css\r\n");
226  } else if(strcmp(&resource[reslen-4],".htm") == 0) {
227  snprintf(hdr,36,"Content-Type: text/html\r\n");
228  } else if(strcmp(&resource[reslen-3],".js") == 0) {
229  snprintf(hdr,36,"Content-Type: text/javascript\r\n");
230  } else {
231  snprintf(hdr,36,"Content-Type: text/plain\r\n");
232  }
233  send(fd, hdr, strlen(hdr),MSG_MORE);
234  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
235  ESP_LOGD(LOG_TAG,"Hdr: %s",hdr);
236  #endif
237 
238  //send http header (content length)
239  snprintf(hdr,36,"Content-Length: %d\r\n\r\n",sz);
240  send(fd, hdr, strlen(hdr),MSG_MORE);
241  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
242  ESP_LOGD(LOG_TAG,"Hdr: %s",hdr);
243  #endif
244 
245  //allocate a buffer in size of one FAT block
246  char* buffer = malloc(512);
247  int i=0, len=0;
248  do {
249  len=fread(buffer, 1, 512, f);
250  i+=len;
251  send(fd, buffer, len, 0);
252  vTaskDelay(1);
253  } while (len==512);
254 
255  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
256  ESP_LOGD(LOG_TAG,"Sent %u bytes",i);
257  #endif
258  fclose(f);
259  free(buffer);
260  xSemaphoreGive(fatSem);
261  } else {
262  ESP_LOGE(LOG_TAG,"Timeout waiting for fat mutex!");
263  }
264 }

273 static void http_server_netconn_serve(int fd) {
274  if(fd < 0) return;
275  #define MAX_BUFF_SIZE 512
276 
277  char *buf = malloc(MAX_BUFF_SIZE);
278  if(buf == NULL)
279  {
280  ESP_LOGE(LOG_TAG,"Cannot allocate buf for recv");
281  return;
282  }
283 
284  //read incoming
285  int size = recv(fd,buf,MAX_BUFF_SIZE,0);
286  #if (LOG_LEVEL_WEB >= ESP_LOG_DEBUG)
287  ESP_LOGD(LOG_TAG,"recv size: %d",size);
288  #endif
289  if(size > 0)
290  {
291  buf[size] = '\0';
292  char *request_line = strtok(buf, "\n");
293 
294  if(request_line) {
295  // default page -> redirect to index.html
296  if(strstr(request_line, "GET / ")) {
297  fat_serve((char*)"/index.htm", fd);
298  } else {
299  // get the requested resource
300  char* method = strtok(request_line, " ");
301  //get rid of "unused" variable.
302  //strtok although needs this call.
303  (void)(method);
304  char* resource = strtok(NULL, " ");
305  fat_serve(resource, fd);
306  }
307  }
308  }
309  free(buf);
310 }

323 static void http_server(void *pvParameters) {
324  int sockfd, new_sockfd;
325  socklen_t addr_len;
326  struct sockaddr_in sock_addr;
327  int ret;
328  //create a new TCP connection
329  sockfd = socket(AF_INET, SOCK_STREAM, 0);
330  if (sockfd < 0) {
331  ESP_LOGE(LOG_TAG,"Cannot create socket");
332  }
333  //bind it to incoming port 80 (HTTP)
334  memset(&sock_addr, 0, sizeof(sock_addr));
335  sock_addr.sin_family = AF_INET;
336  sock_addr.sin_addr.s_addr = 0;
337  sock_addr.sin_port = htons(80);
338  ret = bind(sockfd, (struct sockaddr*)&sock_addr, sizeof(sock_addr));
339  if(ret) {
340  ESP_LOGE(LOG_TAG,"Failed to bind");
341  close(sockfd);
342  sockfd = -1;
343  }
344  ret = listen(sockfd, 1);
345  if(ret) {
346  ESP_LOGE(LOG_TAG,"Failed to listen");
347  close(sockfd);
348  sockfd = -1;
349  }
350  ESP_LOGI(LOG_TAG,"http_server task started");
351 
352  do {
353  //accept connection & save netconn handle
354  new_sockfd = accept(sockfd, (struct sockaddr *)&sock_addr, &addr_len);
355  if (new_sockfd < 0) {
356  ESP_LOGE(LOG_TAG, "Failed to accept" );
357  } else {
358  //serve
359  http_server_netconn_serve(new_sockfd);
360  //close
361  close(new_sockfd);
362  }
363  vTaskDelay(1); //allows task to be pre-empted
364  } while(wifiActive);
365  ESP_LOGI(LOG_TAG,"Killing http task");
366  //if we run into an error
367  close(new_sockfd);
368  close(sockfd);
369  //delete task
370  vTaskDelete(NULL);
371 }

388 static esp_err_t wifi_event_handler(void *ctx, system_event_t *event) {
389  //check if event flags are active
390  if(connectionRoutingStatus == NULL)
391  {
392  ESP_LOGE(LOG_TAG,"Status flags for wifi are NULL! Should not receive an event!");
393  return ESP_FAIL;
394  }
395 
396  //determine event: print information and set flags accordingly
397  switch(event->event_id) {
398 
399  case SYSTEM_EVENT_STA_START:
400  esp_wifi_connect();
401  break;
402 
403  case SYSTEM_EVENT_AP_START:
404  ESP_LOGD(LOG_TAG,"Access point started");
405  xEventGroupSetBits(connectionRoutingStatus, WIFI_ACTIVE);
406  //start off timer (switch off wifi, if no client connects in timeout)
407  wifiStartStopOffTimer(1);
408  break;
409 
410  case SYSTEM_EVENT_AP_STOP:
411  ESP_LOGD(LOG_TAG,"Access point stopped");
412  xEventGroupClearBits(connectionRoutingStatus, WIFI_ACTIVE);
413  break;
414 
415  case SYSTEM_EVENT_AP_STACONNECTED:
416  //a new client connected
417  ESP_LOGI(LOG_TAG,"Client connected, currently connected: %d",getNumberOfWifiStations());
418  //set client connected flag
419  xEventGroupSetBits(connectionRoutingStatus, WIFI_CLIENT_CONNECTED);
420 
421  //if the wifi timer is active, disable the timer (client is connected)
422  wifiStartStopOffTimer(0);
423  break;
424 
425  case SYSTEM_EVENT_AP_STADISCONNECTED:
426  //a client got disconnected
427  ESP_LOGI(LOG_TAG,"Client disconnected, currently connected: %d",getNumberOfWifiStations());
428  if(getNumberOfWifiStations() == 0)
429  {
430  //remove the callback, not necessary if no client is connected
431  halSerialRemoveOutputStream();
432  //clear client connected flag
433  xEventGroupClearBits(connectionRoutingStatus, WIFI_CLIENT_CONNECTED);
434  //start the auto-disable timer
435  wifiStartStopOffTimer(1);
436  }
437  break;
438  default:
439  break;
440  }
441  return ESP_OK;
442 }

459 esp_err_t taskWebGUIEnDisable(int onoff)
460 {
461  esp_err_t ret;
462  //should we enable or disable wifi?
463  if(onoff == 0)
464  {
465  //clear wifi flags
466  xEventGroupClearBits(connectionRoutingStatus, WIFI_ACTIVE | WIFI_CLIENT_CONNECTED);
467 
468  //stop tasks (free 16k RAM)
469  wifiActive = 0;
470 
471  //disable, call wifi_stop
472  if(esp_wifi_stop() == ESP_OK)
473  {
474  ret = esp_wifi_deinit();
475  } else {
476  ret = ESP_OK;
477  }
478  //check return value
479  if(ret != ESP_OK)
480  {
481  ESP_LOGE(LOG_TAG,"Please initialize WiFi prior to enable/disable it!");
482  return ESP_FAIL;
483  } else return ESP_OK;
484  } else {
485  //Wifi config
486  wifi_init_config_t wifi_init_config = WIFI_INIT_CONFIG_DEFAULT();
487  //init wifi & set AP mode
488  ESP_ERROR_CHECK(esp_wifi_init(&wifi_init_config));
489  //bring into AP mode
490  ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_AP));
491  //apply config
492  ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &ap_config));
493 
494  //start wifi
495  ret = esp_wifi_start();
496  switch(ret)
497  {
498  //everything ok
499  case ESP_OK: break;
500  //determine error
501  case ESP_ERR_WIFI_CONN:
502  ESP_LOGE(LOG_TAG,"Wifi internal error, control block invalid");
503  return ESP_FAIL;
504  case ESP_ERR_NO_MEM:
505  ESP_LOGE(LOG_TAG,"Wifi internal error, out of memory");
506  return ESP_FAIL;
507  case ESP_ERR_WIFI_NOT_INIT:
508  ESP_LOGE(LOG_TAG,"Please initialize WiFi prior to enable/disable it!");
509  return ESP_FAIL;
510  case ESP_FAIL:
511  case ESP_ERR_INVALID_ARG:
512  default:
513  ESP_LOGE(LOG_TAG,"Unknown internal Wifi error");
514  return ESP_FAIL;
515  }
516 
517  // start the HTTP Server tasks
518  wifiActive = 1;
519  xTaskCreate(&http_server, "http_server", TASK_WEBGUI_SERVER_STACKSIZE, NULL, 5, &wifiHTTPServerHandle_t);
520  xTaskCreate(&ws_server, "ws_server", TASK_WEBGUI_WEBSOCKET_STACKSIZE, NULL, 5, &wifiWSServerHandle_t);
521 
522  //wait 250ms for WiFi stack to settle
523  vTaskDelay(200/portTICK_PERIOD_MS);
524 
525  return ESP_OK;
526  }
